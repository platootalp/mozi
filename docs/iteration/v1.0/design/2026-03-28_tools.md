## 1. 模块概述

Tools 模块是系统的工具层，提供 Agent 执行任务所需的各种工具。通过 HashAnchorEdit 实现安全的文件编辑，通过内置工具提供基础能力，通过 LSPBridge 和 MCPClient 提供扩展能力。

### 1.1 核心职责

- 工具定义与抽象
- 工具注册与管理
- 工具执行与结果处理
- 哈希锚定安全编辑

### 1.2 差异化特性

| 特性 | 说明 |
|------|------|
| HashAnchorEdit | 带哈希验证的安全编辑，防止并发冲突 |
| 工具沙箱 | 工具在受限环境中执行 |
| 原子执行 | 工具执行是原子的 |
| 扩展机制 | 支持 MCP/Skill 扩展 |

## 3. Tool 基类

### 3.1 抽象基类

```python
class Tool(ABC):
    """工具抽象基类"""

    @property
    @abstractmethod
    def name(self) -> str:
        """工具名称"""
        pass

    @property
    @abstractmethod
    def description(self) -> str:
        """工具描述"""
        pass

    @property
    def input_schema(self) -> Dict[str, Any]:
        """输入 Schema (JSON Schema)"""
        return {}

    @property
    def output_schema(self) -> Dict[str, Any]:
        """输出 Schema"""
        return {}

    @property
    def timeout(self) -> int:
        """超时时间（秒）"""
        return 30

    @abstractmethod
    async def execute(self, params: Dict[str, Any], context: ToolContext) -> ToolResult:
        """执行工具"""
        pass

    def validate_params(self, params: Dict[str, Any]) -> bool:
        """验证参数"""
        try:
            jsonschema.validate(params, self.input_schema)
            return True
        except jsonschema.ValidationError:
            return False
```

### 3.2 工具结果

```python
@dataclass
class ToolResult:
    """工具执行结果"""
    success: bool
    content: Optional[str] = None
    error: Optional[str] = None
    metadata: Dict[str, Any] = field(default_factory=dict)

    @property
    def is_error(self) -> bool:
        return not self.success

    def to_dict(self) -> Dict[str, Any]:
        return {
            "success": self.success,
            "content": self.content,
            "error": self.error,
            "metadata": self.metadata,
        }
```

## 4. 工具注册表

### 4.1 ToolRegistry

```python
class ToolRegistry:
    """工具注册表"""

    def __init__(self):
        self._tools: Dict[str, Tool] = {}
        self._builtin_registered = False

    def register(self, tool: Tool) -> None:
        """注册工具"""
        if tool.name in self._tools:
            raise ToolAlreadyRegisteredError(f"Tool {tool.name} already registered")
        self._tools[tool.name] = tool

    def unregister(self, name: str) -> None:
        """注销工具"""
        if name in self._tools:
            del self._tools[name]

    def get(self, name: str) -> Optional[Tool]:
        """获取工具"""
        return self._tools.get(name)

    def list_tools(self) -> List[str]:
        """列出所有工具"""
        return list(self._tools.keys())

    def register_builtin(self) -> None:
        """注册内置工具"""
        if self._builtin_registered:
            return

        # 基础工具
        self.register(ReadTool())
        self.register(WriteTool())
        self.register(EditTool())
        self.register(BashTool())
        self.register(GrepTool())
        self.register(GlobTool())

        self._builtin_registered = True

    def register_mcp_tools(self, tools: List[Tool]) -> None:
        """注册 MCP 工具"""
        for tool in tools:
            self.register(tool)

    def register_skill_tools(self, tools: List[Tool]) -> None:
        """注册 Skill 工具"""
        for tool in tools:
            self.register(tool)
```

## 5. 工具执行器

### 5.1 ToolExecutor

```python
class ToolExecutor:
    """工具执行器"""

    def __init__(
        self,
        registry: ToolRegistry,
        permission_checker: PermissionChecker,
        sandbox: Sandbox,
    ):
        self.registry = registry
        self.permission_checker = permission_checker
        self.sandbox = sandbox

    async def execute(
        self,
        tool_name: str,
        params: Dict[str, Any],
        context: ToolContext,
    ) -> ToolResult:
        """执行工具"""
        # 1. 获取工具
        tool = self.registry.get(tool_name)
        if not tool:
            return ToolResult(success=False, error=f"Tool {tool_name} not found")

        # 2. 权限检查
        permission = await self.permission_checker.check(tool_name, params)
        if permission.status == PermissionStatus.DENY:
            return ToolResult(success=False, error=f"Permission denied for {tool_name}")

        if permission.status == PermissionStatus.ASK:
            # HITL 场景
            return ToolResult(
                success=False,
                error="HITL confirmation required",
                metadata={"requires_approval": True}
            )

        # 3. 沙箱检查
        if not await self.sandbox.check(tool_name, params):
            return ToolResult(success=False, error=f"Sandbox restriction for {tool_name}")

        # 4. 参数验证
        if not tool.validate_params(params):
            return ToolResult(success=False, error="Invalid parameters")

        # 5. 执行
        try:
            result = await asyncio.wait_for(
                tool.execute(params, context),
                timeout=tool.timeout
            )
            return result
        except asyncio.TimeoutError:
            return ToolResult(success=False, error=f"Tool {tool_name} timed out")
        except Exception as e:
            return ToolResult(success=False, error=str(e))
```

## 6. 内置工具

### 6.1 ReadTool

```python
class ReadTool(Tool):
    """读取文件工具"""

    @property
    def name(self) -> str:
        return "read"

    @property
    def description(self) -> str:
        return "读取文件内容"

    @property
    def input_schema(self) -> Dict[str, Any]:
        return {
            "type": "object",
            "properties": {
                "path": {"type": "string", "description": "文件路径"},
                "offset": {"type": "integer", "description": "起始行号", "default": 0},
                "limit": {"type": "integer", "description": "读取行数", "default": 1000},
            },
            "required": ["path"]
        }

    async def execute(self, params: Dict[str, Any], context: ToolContext) -> ToolResult:
        try:
            path = Path(params["path"])
            if not path.exists():
                return ToolResult(success=False, error=f"File not found: {path}")

            content = path.read_text(encoding="utf-8")
            lines = content.split("\n")

            offset = params.get("offset", 0)
            limit = params.get("limit", 1000)
            lines = lines[offset:offset + limit]

            return ToolResult(
                success=True,
                content="\n".join(lines),
                metadata={"lines_read": len(lines), "total_lines": len(content.split("\n"))}
            )
        except Exception as e:
            return ToolResult(success=False, error=str(e))
```

```python
class WriteTool(Tool):
    """写入文件工具"""

    @property
    def name(self) -> str:
        return "write"

    @property
    def description(self) -> str:
        return "创建或覆盖文件"

    @property
    def input_schema(self) -> Dict[str, Any]:
        return {
            "type": "object",
            "properties": {
                "path": {"type": "string", "description": "文件路径"},
                "content": {"type": "string", "description": "文件内容"},
                "create_dirs": {"type": "boolean", "description": "创建目录", "default": False},
            },
            "required": ["path", "content"]
        }

    async def execute(self, params: Dict[str, Any], context: ToolContext) -> ToolResult:
        try:
            path = Path(params["path"])

            if params.get("create_dirs"):
                path.parent.mkdir(parents=True, exist_ok=True)

            path.write_text(params["content"], encoding="utf-8")

            return ToolResult(
                success=True,
                content=f"File written: {path}",
                metadata={"bytes_written": len(params["content"])}
            )
        except Exception as e:
            return ToolResult(success=False, error=str(e))
```

### 6.3 EditTool (HashAnchorEdit)

```python
class EditTool(Tool):
    """带哈希锚定的安全编辑工具"""

    @property
    def name(self) -> str:
        return "edit"

    @property
    def description(self) -> str:
        return "带哈希验证的安全编辑"

    @property
    def input_schema(self) -> Dict[str, Any]:
        return {
            "type": "object",
            "properties": {
                "path": {"type": "string", "description": "文件路径"},
                "old": {"type": "string", "description": "原内容（用于验证）"},
                "new": {"type": "string", "description": "新内容"},
            },
            "required": ["path", "old", "new"]
        }

    async def execute(self, params: Dict[str, Any], context: ToolContext) -> ToolResult:
        path = Path(params["path"])

        try:
            # 读取当前内容
            current = path.read_text(encoding="utf-8")

            # 验证哈希
            old_content = params["old"]
            if self.hash(current) != self.hash(old_content):
                return ToolResult(
                    success=False,
                    error="Hash mismatch: file content has changed since read"
                )

            # 执行替换
            new_content = params["new"]
            if old_content not in current:
                return ToolResult(success=False, error="Old content not found in file")

            updated = current.replace(old_content, new_content)
            path.write_text(updated, encoding="utf-8")

            return ToolResult(
                success=True,
                content=f"File edited: {path}",
                metadata={"bytes_changed": len(new_content) - len(old_content)}
            )

        except FileNotFoundError:
            return ToolResult(success=False, error=f"File not found: {path}")
        except Exception as e:
            return ToolResult(success=False, error=str(e))

    def hash(self, content: str) -> str:
        """计算内容哈希"""
        return hashlib.sha256(content.encode()).hexdigest()[:16]
```

### 6.4 BashTool

```python
class BashTool(Tool):
    """Shell 命令执行工具"""

    @property
    def name(self) -> str:
        return "bash"

    @property
    def description(self) -> str:
        return "执行 Shell 命令"

    @property
    def input_schema(self) -> Dict[str, Any]:
        return {
            "type": "object",
            "properties": {
                "command": {"type": "string", "description": "Shell 命令"},
                "timeout": {"type": "integer", "description": "超时时间(秒)", "default": 30},
                "cwd": {"type": "string", "description": "工作目录"},
            },
            "required": ["command"]
        }

    async def execute(self, params: Dict[str, Any], context: ToolContext) -> ToolResult:
        command = params["command"]
        timeout = params.get("timeout", 30)
        cwd = params.get("cwd")

        try:
            result = await asyncio.create_subprocess_shell(
                command,
                stdout=asyncio.subprocess.PIPE,
                stderr=asyncio.subprocess.PIPE,
                cwd=cwd,
            )

            try:
                stdout, stderr = await asyncio.wait_for(
                    result.communicate(),
                    timeout=timeout
                )
            except asyncio.TimeoutError:
                result.kill()
                return ToolResult(success=False, error=f"Command timed out after {timeout}s")

            return ToolResult(
                success=result.returncode == 0,
                content=stdout.decode() if stdout else "",
                error=stderr.decode() if stderr else None,
                metadata={
                    "return_code": result.returncode,
                    "command": command,
                }
            )

        except Exception as e:
            return ToolResult(success=False, error=str(e))
```

## 7. HashAnchorEdit 安全机制

### 7.1 工作原理

```
┌─────────────────────────────────────────────────────────────┐
│                    HashAnchorEdit                              │
│                                                             │
│  1. 读取文件                                                 │
│     └── current = read(path)                               │
│                                                             │
│  2. 计算哈希                                                 │
│     └── expected_hash = hash(old_content)                   │
│                                                             │
│  3. 验证                                                     │
│     └── current_hash = hash(current)                        │
│         └── current_hash == expected_hash?                   │
│                                                             │
│  4. 执行替换                                                 │
│     └── updated = current.replace(old, new)                 │
│                                                             │
│  5. 写入                                                     │
│     └── write(path, updated)                                │
│                                                             │
└─────────────────────────────────────────────────────────────┘
```

### 7.2 并发安全

HashAnchorEdit 通过哈希验证确保：
- 只有当文件内容与读取时一致时才执行编辑
- 防止多 Agent 并发编辑导致的内容覆盖
- 发现冲突时返回错误而非静默覆盖

│  5. 执行工具                                                 │
│     └── Tool.execute(params, context)                       │

│  6. 返回结果                                                 │
│     └── ToolResult                                          │
