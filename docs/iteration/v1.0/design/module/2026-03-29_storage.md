# Storage 模块设计文档

## 1. 模块概述

### 1.1 模块名称

Storage（存储层）

### 1.2 职责

Storage 模块是 Mozi 系统的基础设施层组件，负责所有持久化数据的存储和管理：
- 会话数据的持久化和恢复
- 长期记忆的向量化存储
- 产物文件（Artifact）的管理和存储
- 存储迁移和版本管理

### 1.3 核心能力

| 能力 | 说明 |
| --- | --- |
| 会话持久化 | 会话数据的创建、读取、更新、删除，支持跨会话恢复 |
| 记忆存储 | 长期记忆的向量化和持久化，支持语义检索 |
| 文件管理 | 产物文件的存储、检索、清理和空间管理 |
| 存储抽象 | 多后端存储支持（文件系统、云存储），统一接口 |
| 迁移管理 | 数据迁移、版本升级、向后兼容 |

### 1.4 核心问题

| 问题 | 说明 |
| --- | --- |
| 存储抽象 | 不同类型数据需要不同存储后端 |
| 数据持久化 | 会话、记忆、产物等数据的持久化 |
| 向量检索 | 长期记忆需要向量检索能力 |
| 存储迁移 | 数据格式升级如何平滑迁移 |

### 1.5 难点

| 难点 | 说明 |
| --- | --- |
| 多后端支持 | 如何抽象统一的存储接口支持多种后端 |
| 向量存储选型 | Faiss、Milvus 等向量化引擎的选择 |
| 数据一致性 | 读写分离架构下的数据一致性 |
| 存储空间管理 | 产物文件的清理策略和空间回收 |

### 1.6 解决方案

| 方案 | 说明 |
| --- | --- |
| StorageBase 抽象 | 统一接口定义，子类实现具体后端 |
| FileSessionStorage | 基于文件的会话持久化 |
| FileVectorStore | 基于文件的向量存储（轻量级） |
| ArtifactManager | 产物文件管理和自动清理 |

## 3. 存储抽象

### 3.1 基础存储接口

```python
from abc import ABC, abstractmethod
from dataclasses import dataclass, field
from datetime import datetime
from enum import Enum
from typing import Any, AsyncIterator


class StorageBackend(Enum):
    """存储后端类型"""
    FILE_SYSTEM = "file_system"
    SQLITE = "sqlite"
    POSTGRESQL = "postgresql"
    S3 = "s3"
    MEMORY = "memory"


@dataclass
class StorageConfig:
    """存储配置"""
    backend: StorageBackend = StorageBackend.FILE_SYSTEM
    base_path: str = "~/.mozi/storage"
    max_size_mb: int = 1024  # 最大存储大小（MB）
    auto_cleanup: bool = True
    cleanup_interval_hours: int = 24
    retention_days: int = 30


@dataclass
class StorageStats:
    """存储统计信息"""
    total_size_bytes: int = 0
    file_count: int = 0
    last_cleanup: datetime | None = None
    available_space_bytes: int = 0


class StorageBase(ABC):
    """存储抽象基类"""

    @abstractmethod
    async def save(
        self,
        key: str,
        data: bytes | str,
        metadata: dict[str, Any] | None = None,
    ) -> None:
        """
        保存数据

        Args:
            key: 存储键
            data: 数据内容
            metadata: 元数据
        """

    @abstractmethod
    async def load(self, key: str) -> bytes | str | None:
        """
        加载数据

        Args:
            key: 存储键

        Returns:
            数据内容，不存在返回 None
        """

    @abstractmethod
    async def delete(self, key: str) -> bool:
        """
        删除数据

        Args:
            key: 存储键

        Returns:
            是否删除成功
        """

    @abstractmethod
    async def exists(self, key: str) -> bool:
        """
        检查键是否存在

        Args:
            key: 存储键

        Returns:
            是否存在
        """

    @abstractmethod
    async def list_keys(
        self,
        prefix: str | None = None,
        limit: int | None = None,
    ) -> list[str]:
        """
        列出键

        Args:
            prefix: 键前缀过滤
            limit: 返回数量限制

        Returns:
            键列表
        """

    @abstractmethod
    async def get_metadata(self, key: str) -> dict[str, Any] | None:
        """
        获取元数据

        Args:
            key: 存储键

        Returns:
            元数据字典
        """

    @abstractmethod
    async def get_stats(self) -> StorageStats:
        """获取存储统计信息"""
```

### 3.2 文件系统存储实现

```python
import json
from pathlib import Path


class FileStorage(StorageBase):
    """文件系统存储实现"""

    def __init__(self, config: StorageConfig) -> None:
        self._config = config
        self._base_path = Path(config.base_path).expanduser()
        self._metadata_dir = self._base_path / ".metadata"

    async def save(
        self,
        key: str,
        data: bytes | str,
        metadata: dict[str, Any] | None = None,
    ) -> None:
        """保存数据到文件"""
        file_path = self._get_file_path(key)
        file_path.parent.mkdir(parents=True, exist_ok=True)

        if isinstance(data, str):
            data = data.encode("utf-8")

        await self._write_atomic(file_path, data)

        # 保存元数据
        if metadata:
            metadata_path = self._get_metadata_path(key)
            metadata_path.parent.mkdir(parents=True, exist_ok=True)
            await self._write_json(metadata_path, metadata)

    async def load(self, key: str) -> bytes | None:
        """从文件加载数据"""
        file_path = self._get_file_path(key)
        if not file_path.exists():
            return None

        return file_path.read_bytes()

    async def delete(self, key: str) -> bool:
        """删除文件"""
        file_path = self._get_file_path(key)
        metadata_path = self._get_metadata_path(key)

        deleted = False
        if file_path.exists():
            file_path.unlink()
            deleted = True

        if metadata_path.exists():
            metadata_path.unlink()

        return deleted

    async def exists(self, key: str) -> bool:
        """检查文件是否存在"""
        return self._get_file_path(key).exists()

    async def list_keys(
        self,
        prefix: str | None = None,
        limit: int | None = None,
    ) -> list[str]:
        """列出所有键"""
        all_files = list(self._base_path.rglob("*"))
        keys = [
            str(f.relative_to(self._base_path))
            for f in all_files
            if f.is_file() and not f.name.startswith(".")
        ]

        if prefix:
            keys = [k for k in keys if k.startswith(prefix)]

        keys.sort()
        if limit:
            keys = keys[:limit]

        return keys

    async def get_metadata(self, key: str) -> dict[str, Any] | None:
        """获取元数据"""
        metadata_path = self._get_metadata_path(key)
        if not metadata_path.exists():
            return None

        return await self._read_json(metadata_path)

    async def get_stats(self) -> StorageStats:
        """获取存储统计"""
        all_files = list(self._base_path.rglob("*"))
        total_size = sum(f.stat().st_size for f in all_files if f.is_file())

        return StorageStats(
            total_size_bytes=total_size,
            file_count=len([f for f in all_files if f.is_file()]),
        )

    def _get_file_path(self, key: str) -> Path:
        """获取文件路径"""
        return self._base_path / key

    def _get_metadata_path(self, key: str) -> Path:
        """获取元数据路径"""
        safe_key = key.replace("/", "_")
        return self._metadata_dir / f"{safe_key}.json"

    async def _write_atomic(self, path: Path, data: bytes) -> None:
        """原子写入"""
        temp_path = path.with_suffix(path.suffix + ".tmp")
        temp_path.write_bytes(data)
        temp_path.replace(path)

    async def _write_json(self, path: Path, data: dict) -> None:
        """写入 JSON"""
        json_str = json.dumps(data, indent=2, ensure_ascii=False)
        path.write_text(json_str, encoding="utf-8")

    async def _read_json(self, path: Path) -> dict:
        """读取 JSON"""
        return json.loads(path.read_text(encoding="utf-8"))
```

### 4.4 文件会话存储实现

```python
import json
from pathlib import Path


class FileSessionStorage(SessionStorage):
    """文件会话存储实现"""

    def __init__(
        self,
        storage_dir: str = "~/.mozi/sessions",
        schema_version: str = "1.0",
    ) -> None:
        self._storage_dir = Path(storage_dir).expanduser()
        self._schema_version = schema_version

    async def save(self, session: Session) -> None:
        """保存会话到文件"""
        session_path = self._get_session_path(session.id)
        session_path.parent.mkdir(parents=True, exist_ok=True)

        data = {
            "version": self._schema_version,
            "session": self._serialize_session(session),
        }

        await self._write_atomic(session_path, data)

    async def load(self, session_id: str) -> Session | None:
        """从文件加载会话"""
        session_path = self._get_session_path(session_id)
        if not session_path.exists():
            return None

        try:
            data = await self._read_json(session_path)
            return self._deserialize_session(data)
        except (json.JSONDecodeError, KeyError) as e:
            # 降级处理：损坏的文件
            raise SessionStorageError(
                f"Failed to load session '{session_id}': {e}"
            ) from e

    async def delete(self, session_id: str) -> bool:
        """删除会话文件"""
        session_path = self._get_session_path(session_id)
        if session_path.exists():
            session_path.unlink()
            return True
        return False

    async def list(
        self,
        status: SessionStatus | None = None,
        user_id: str | None = None,
        limit: int = 100,
        offset: int = 0,
    ) -> list[Session]:
        """列出存储的会话"""
        sessions = []

        for session_file in self._storage_dir.glob("*.json"):
            try:
                data = await self._read_json(session_file)
                session = self._deserialize_session(data)

                # 应用过滤
                if status is not None and session.status != status:
                    continue
                if user_id is not None and session.metadata.user_id != user_id:
                    continue

                sessions.append(session)
            except (json.JSONDecodeError, KeyError):
                # 跳过损坏的文件
                continue

        # 按最后活跃时间排序
        sessions.sort(
            key=lambda s: s.metadata.last_active_at,
            reverse=True,
        )

        return sessions[offset:offset + limit]

    async def exists(self, session_id: str) -> bool:
        """检查会话是否存在"""
        return self._get_session_path(session_id).exists()

    async def update_status(
        self,
        session_id: str,
        status: SessionStatus,
    ) -> bool:
        """更新会话状态"""
        session = await self.load(session_id)
        if not session:
            return False

        session.status = status
        session.metadata.updated_at = datetime.now()
        await self.save(session)
        return True

    async def stream_messages(
        self,
        session_id: str,
    ) -> AsyncIterator[Message]:
        """流式获取会话消息"""
        session = await self.load(session_id)
        if session:
            for message in session.messages:
                yield message

    def _get_session_path(self, session_id: str) -> Path:
        """获取会话文件路径"""
        return self._storage_dir / f"{session_id}.json"

    def _serialize_session(self, session: Session) -> dict:
        """序列化会话"""
        return {
            "id": session.id,
            "status": session.status.value,
            "parent_session_id": session.parent_session_id,
            "metadata": {
                "created_at": session.metadata.created_at.isoformat(),
                "updated_at": session.metadata.updated_at.isoformat(),
                "last_active_at": session.metadata.last_active_at.isoformat(),
                "message_count": session.metadata.message_count,
                "total_tokens": session.metadata.total_tokens,
                "user_id": session.metadata.user_id,
                "tags": session.metadata.tags,
                "custom_fields": session.metadata.custom_fields,
            },
            "messages": [
                {
                    "role": msg.role.value,
                    "content": msg.content,
                    "timestamp": msg.timestamp.isoformat(),
                    "tool_call_id": msg.tool_call_id,
                    "metadata": msg.metadata,
                }
                for msg in session.messages
            ],
        }

    def _deserialize_session(self, data: dict) -> Session:
        """反序列化会话"""
        session_data = data.get("session", data)

        messages = [
            Message(
                role=MessageRole(msg["role"]),
                content=msg["content"],
                timestamp=datetime.fromisoformat(msg["timestamp"]),
                tool_call_id=msg.get("tool_call_id"),
                metadata=msg.get("metadata", {}),
            )
            for msg in session_data.get("messages", [])
        ]

        metadata = session_data.get("metadata", {})
        session_metadata = SessionMetadata(
            created_at=datetime.fromisoformat(metadata["created_at"]),
            updated_at=datetime.fromisoformat(metadata["updated_at"]),
            last_active_at=datetime.fromisoformat(metadata["last_active_at"]),
            message_count=metadata.get("message_count", 0),
            total_tokens=metadata.get("total_tokens", 0),
            user_id=metadata.get("user_id"),
            tags=metadata.get("tags", []),
            custom_fields=metadata.get("custom_fields", {}),
        )

        return Session(
            id=session_data["id"],
            status=SessionStatus(session_data.get("status", "active")),
            parent_session_id=session_data.get("parent_session_id"),
            metadata=session_metadata,
            messages=messages,
        )

    async def _write_atomic(self, path: Path, data: dict) -> None:
        """原子写入"""
        temp_path = path.with_suffix(".tmp")
        json_str = json.dumps(data, indent=2, ensure_ascii=False)
        temp_path.write_text(json_str, encoding="utf-8")
        temp_path.replace(path)

    async def _read_json(self, path: Path) -> dict:
        """读取 JSON"""
        return json.loads(path.read_text(encoding="utf-8"))
```

class VectorStore(ABC):
    """向量存储抽象接口"""

    @abstractmethod
    async def initialize(self) -> None:
        """初始化向量存储"""

    @abstractmethod
    async def upsert(
        self,
        session_id: str,
        blocks: list[MemoryBlock],
    ) -> None:
        """
        插入或更新记忆块

        Args:
            session_id: 会话 ID
            blocks: 记忆块列表
        """

    @abstractmethod
    async def search(
        self,
        query_embedding: list[float],
        session_id: str | None = None,
        memory_type: MemoryType | None = None,
        top_k: int = 5,
        threshold: float = 0.7,
    ) -> list[tuple[MemoryBlock, float]]:
        """
        搜索相似记忆

        Args:
            query_embedding: 查询向量
            session_id: 会话 ID 过滤
            memory_type: 记忆类型过滤
            top_k: 返回数量
            threshold: 相似度阈值

        Returns:
            (记忆块, 相似度) 列表
        """

    @abstractmethod
    async def delete_session(self, session_id: str) -> None:
        """
        删除会话的所有记忆

        Args:
            session_id: 会话 ID
        """

    @abstractmethod
    async def delete_block(self, block_id: str) -> bool:
        """
        删除单个记忆块

        Args:
            block_id: 记忆块 ID

        Returns:
            是否删除成功
        """

    @abstractmethod
    async def get_stats(self) -> dict[str, Any]:
        """
        获取存储统计

        Returns:
            统计信息字典
        """
```

        self,
        storage_dir: str = "~/.mozi/memory/vectors",
        embedding_dim: int = 1536,
    ) -> None:
        self._storage_dir = Path(storage_dir).expanduser()
        self._embedding_dim = embedding_dim
        self._index_file = self._storage_dir / "vector_index.json"
        self._blocks_dir = self._storage_dir / "blocks"

    async def initialize(self) -> None:
        """初始化存储目录"""
        self._storage_dir.mkdir(parents=True, exist_ok=True)
        self._blocks_dir.mkdir(parents=True, exist_ok=True)

        if not self._index_file.exists():
            await self._write_json(self._index_file, {"blocks": {}})

    async def upsert(
        self,
        session_id: str,
        blocks: list[MemoryBlock],
    ) -> None:
        """插入或更新记忆块"""
        index_data = await self._read_json(self._index_file)

        for block in blocks:
            # 保存块数据
            block_file = self._blocks_dir / f"{block.id}.json"
            await self._write_json(block_file, self._serialize_block(block))

            # 更新索引
            index_data["blocks"][block.id] = {
                "session_id": block.session_id,
                "memory_type": block.memory_type.value,
                "embedding_norm": self._normalize_vector(block.embedding)
                    if block.embedding else None,
                "created_at": block.created_at.isoformat(),
            }

        await self._write_json(self._index_file, index_data)

    async def search(
        self,
        query_embedding: list[float],
        session_id: str | None = None,
        memory_type: MemoryType | None = None,
        top_k: int = 5,
        threshold: float = 0.7,
    ) -> list[tuple[MemoryBlock, float]]:
        """搜索相似记忆"""
        index_data = await self._read_json(self._index_file)
        query_norm = self._normalize_vector(query_embedding)

        results: list[tuple[MemoryBlock, float]] = []

        for block_id, index_entry in index_data["blocks"].items():
            # 应用过滤
            if session_id and index_entry["session_id"] != session_id:
                continue
            if memory_type and index_entry["memory_type"] != memory_type.value:
                continue
            if not index_entry["embedding_norm"]:
                continue

            # 计算余弦相似度
            similarity = self._cosine_similarity(
                query_norm,
                index_entry["embedding_norm"],
            )

            if similarity >= threshold:
                # 加载完整块数据
                block_file = self._blocks_dir / f"{block_id}.json"
                block_data = await self._read_json(block_file)
                block = self._deserialize_block(block_data)

                results.append((block, similarity))

        # 按相似度排序
        results.sort(key=lambda x: x[1], reverse=True)

        return results[:top_k]

    async def delete_session(self, session_id: str) -> None:
        """删除会话的所有记忆"""
        index_data = await self._read_json(self._index_file)
        blocks_to_delete = [
            block_id
            for block_id, entry in index_data["blocks"].items()
            if entry["session_id"] == session_id
        ]

        for block_id in blocks_to_delete:
            # 删除块文件
            block_file = self._blocks_dir / f"{block_id}.json"
            if block_file.exists():
                block_file.unlink()

            # 从索引中移除
            del index_data["blocks"][block_id]

        await self._write_json(self._index_file, index_data)

    async def delete_block(self, block_id: str) -> bool:
        """删除单个记忆块"""
        index_data = await self._read_json(self._index_file)

        if block_id not in index_data["blocks"]:
            return False

        # 删除块文件
        block_file = self._blocks_dir / f"{block_id}.json"
        if block_file.exists():
            block_file.unlink()

        # 从索引中移除
        del index_data["blocks"][block_id]
        await self._write_json(self._index_file, index_data)

        return True

    async def get_stats(self) -> dict[str, Any]:
        """获取存储统计"""
        index_data = await self._read_json(self._index_file)
        blocks = index_data["blocks"]

        session_ids = set(entry["session_id"] for entry in blocks.values())

        return {
            "total_blocks": len(blocks),
            "sessions": len(session_ids),
            "by_memory_type": {
                mt.value: sum(
                    1 for entry in blocks.values()
                    if entry["memory_type"] == mt.value
                )
                for mt in MemoryType
            },
        }

    def _normalize_vector(self, vector: list[float]) -> list[float]:
        """归一化向量"""
        norm = math.sqrt(sum(x * x for x in vector))
        if norm == 0:
            return vector
        return [x / norm for x in vector]

    def _cosine_similarity(
        self,
        vec1: list[float],
        vec2: list[float],
    ) -> float:
        """计算余弦相似度"""
        dot_product = sum(a * b for a, b in zip(vec1, vec2))
        return max(0.0, min(1.0, dot_product))  # 归一化到 [0, 1]

    def _serialize_block(self, block: MemoryBlock) -> dict:
        """序列化记忆块"""
        return {
            "id": block.id,
            "session_id": block.session_id,
            "content": block.content,
            "memory_type": block.memory_type.value,
            "embedding": block.embedding,
            "importance": block.importance,
            "created_at": block.created_at.isoformat(),
            "accessed_at": block.accessed_at.isoformat(),
            "metadata": block.metadata,
        }

    def _deserialize_block(self, data: dict) -> MemoryBlock:
        """反序列化记忆块"""
        return MemoryBlock(
            id=data["id"],
            session_id=data["session_id"],
            content=data["content"],
            memory_type=MemoryType(data["memory_type"]),
            embedding=data.get("embedding"),
            importance=data.get("importance", 0.5),
            created_at=datetime.fromisoformat(data["created_at"]),
            accessed_at=datetime.fromisoformat(data["accessed_at"]),
            metadata=data.get("metadata", {}),
        )

    async def _write_json(self, path: Path, data: dict) -> None:
        """写入 JSON"""
        json_str = json.dumps(data, indent=2, ensure_ascii=False)
        path.write_text(json_str, encoding="utf-8")

    async def _read_json(self, path: Path) -> dict:
        """读取 JSON"""
        return json.loads(path.read_text(encoding="utf-8"))
```

### 6.3 文件产物管理器实现

```python
import hashlib
import shutil
from pathlib import Path


class FileArtifactManager(ArtifactManager):
    """文件产物管理器实现"""

    def __init__(
        self,
        storage_dir: str = "~/.mozi/artifacts",
        max_size_bytes: int = 10 * 1024 * 1024 * 1024,  # 10GB
    ) -> None:
        self._storage_dir = Path(storage_dir).expanduser()
        self._max_size_bytes = max_size_bytes
        self._manifest_file = self._storage_dir / "manifest.json"

    async def save(
        self,
        content: bytes | str,
        metadata: ArtifactMetadata,
    ) -> Artifact:
        """保存产物到文件"""
        # 生成产物 ID
        artifact_id = self._generate_id(content)

        # 计算校验和
        if isinstance(content, str):
            content_bytes = content.encode("utf-8")
        else:
            content_bytes = content

        checksum = hashlib.sha256(content_bytes).hexdigest()
        size_bytes = len(content_bytes)

        # 确定文件路径
        file_ext = self._get_extension(metadata.mime_type, metadata.artifact_type)
        artifact_dir = self._storage_dir / metadata.artifact_type.value
        artifact_dir.mkdir(parents=True, exist_ok=True)
        file_path = artifact_dir / f"{artifact_id}{file_ext}"

        # 写入文件
        file_path.write_bytes(content_bytes)

        # 更新清单
        artifact = Artifact(
            id=artifact_id,
            metadata=metadata,
            content=content_bytes if len(content_bytes) < 1024 else None,
            file_path=file_path,
            status=ArtifactStatus.COMPLETED,
            created_at=datetime.now(),
        )

        await self._update_manifest(artifact, checksum, size_bytes)

        return artifact

    async def load(self, artifact_id: str) -> Artifact | None:
        """加载产物"""
        manifest = await self._read_manifest()
        entry = manifest.get("artifacts", {}).get(artifact_id)

        if not entry:
            return None

        # 读取文件
        file_path = Path(entry["file_path"])
        if not file_path.exists():
            return None

        content = file_path.read_bytes()

        metadata = ArtifactMetadata(
            name=entry["name"],
            artifact_type=ArtifactType(entry["artifact_type"]),
            size_bytes=entry["size_bytes"],
            mime_type=entry.get("mime_type"),
            session_id=entry.get("session_id"),
            task_id=entry.get("task_id"),
            tags=entry.get("tags", []),
            custom_fields=entry.get("custom_fields", {}),
        )

        return Artifact(
            id=artifact_id,
            metadata=metadata,
            content=content,
            file_path=file_path,
            status=ArtifactStatus(entry["status"]),
            created_at=datetime.fromisoformat(entry["created_at"]),
            updated_at=datetime.fromisoformat(entry["updated_at"]),
            expires_at=datetime.fromisoformat(entry["expires_at"])
                if entry.get("expires_at") else None,
        )

    async def delete(self, artifact_id: str) -> bool:
        """删除产物"""
        manifest = await self._read_manifest()
        entry = manifest.get("artifacts", {}).get(artifact_id)

        if not entry:
            return False

        # 删除文件
        file_path = Path(entry["file_path"])
        if file_path.exists():
            file_path.unlink()

        # 从清单中移除
        del manifest["artifacts"][artifact_id]
        await self._write_manifest(manifest)

        return True

    async def list(
        self,
        session_id: str | None = None,
        artifact_type: ArtifactType | None = None,
        status: ArtifactStatus | None = None,
        limit: int = 100,
    ) -> list[Artifact]:
        """列出产物"""
        manifest = await self._read_manifest()
        artifacts = []

        for artifact_id, entry in manifest.get("artifacts", {}).items():
            # 应用过滤
            if session_id and entry.get("session_id") != session_id:
                continue
            if artifact_type and entry["artifact_type"] != artifact_type.value:
                continue
            if status and entry["status"] != status.value:
                continue

            artifacts.append(artifact_id)

        # 排序并限制数量
        artifacts.sort(
            key=lambda a: manifest["artifacts"][a]["created_at"],
            reverse=True,
        )

        results = []
        for artifact_id in artifacts[:limit]:
            artifact = await self.load(artifact_id)
            if artifact:
                results.append(artifact)

        return results

    async def get_download_url(
        self,
        artifact_id: str,
        expires_in_seconds: int = 3600,
    ) -> str | None:
        """获取本地文件路径作为下载 URL"""
        artifact = await self.load(artifact_id)
        if not artifact or not artifact.file_path:
            return None

        return str(artifact.file_path.absolute())

    async def cleanup_expired(self) -> int:
        """清理过期产物"""
        manifest = await self._read_manifest()
        now = datetime.now()
        deleted_count = 0

        artifacts_to_delete = [
            artifact_id
            for artifact_id, entry in manifest.get("artifacts", {}).items()
            if entry.get("expires_at") and
            datetime.fromisoformat(entry["expires_at"]) < now
        ]

        for artifact_id in artifacts_to_delete:
            if await self.delete(artifact_id):
                deleted_count += 1

        return deleted_count

    def _generate_id(self, content: bytes | str) -> str:
        """生成产物 ID"""
        if isinstance(content, str):
            content = content.encode("utf-8")
        return hashlib.sha256(content).hexdigest()[:16]

    def _get_extension(
        self,
        mime_type: str | None,
        artifact_type: ArtifactType,
    ) -> str:
        """获取文件扩展名"""
        if mime_type:
            mime_to_ext = {
                "text/plain": ".txt",
                "text/html": ".html",
                "text/css": ".css",
                "text/javascript": ".js",
                "application/json": ".json",
                "application/python": ".py",
                "image/png": ".png",
                "image/jpeg": ".jpg",
                "image/gif": ".gif",
            }
            if mime_type in mime_to_ext:
                return mime_to_ext[mime_type]

        type_to_ext = {
            ArtifactType.CODE: ".txt",
            ArtifactType.DOCUMENT: ".md",
            ArtifactType.IMAGE: ".bin",
            ArtifactType.DATA: ".json",
            ArtifactType.LOG: ".log",
            ArtifactType.OTHER: ".bin",
        }
        return type_to_ext.get(artifact_type, ".bin")

    async def _update_manifest(
        self,
        artifact: Artifact,
        checksum: str,
        size_bytes: int,
    ) -> None:
        """更新清单"""
        manifest = await self._read_manifest()

        if "artifacts" not in manifest:
            manifest["artifacts"] = {}

        manifest["artifacts"][artifact.id] = {
            "name": artifact.metadata.name,
            "artifact_type": artifact.metadata.artifact_type.value,
            "file_path": str(artifact.file_path),
            "size_bytes": size_bytes,
            "checksum": checksum,
            "mime_type": artifact.metadata.mime_type,
            "session_id": artifact.metadata.session_id,
            "task_id": artifact.metadata.task_id,
            "status": artifact.status.value,
            "created_at": artifact.created_at.isoformat(),
            "updated_at": artifact.updated_at.isoformat(),
            "expires_at": artifact.expires_at.isoformat()
                if artifact.expires_at else None,
            "tags": artifact.metadata.tags,
            "custom_fields": artifact.metadata.custom_fields,
        }

        await self._write_manifest(manifest)

    async def _read_manifest(self) -> dict:
        """读取清单"""
        if not self._manifest_file.exists():
            return {"artifacts": {}}

        import json
        return json.loads(self._manifest_file.read_text(encoding="utf-8"))

    async def _write_manifest(self, manifest: dict) -> None:
        """写入清单"""
        import json
        self._manifest_file.parent.mkdir(parents=True, exist_ok=True)
        self._manifest_file.write_text(
            json.dumps(manifest, indent=2, ensure_ascii=False),
            encoding="utf-8",
        )
```
