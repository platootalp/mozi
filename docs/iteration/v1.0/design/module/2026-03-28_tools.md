## 1. 模块概述

Tools 模块是系统的工具层，为 Agent 执行任务提供能力支撑。通过 HashAnchorEdit 实现安全的文件编辑，通过内置工具提供基础能力，通过 MCPClient 和 LSPBridge 提供扩展能力。

### 1.1 核心职责

| 职责 | 说明 |
|------|------|
| 工具抽象 | 定义 Tool 基类，统一工具接口 |
| 工具注册 | 通过 ToolRegistry 管理工具生命周期 |
| 工具执行 | 通过 ToolExecutor 统一执行入口 |
| 安全编辑 | HashAnchorEdit 防止并发冲突 |
| 扩展机制 | 支持 MCP/LSP/Skill 扩展 |

### 1.2 差异化特性

| 特性 | 说明 |
|------|------|
| HashAnchorEdit | 带内容验证的安全编辑，防止并发编辑覆盖 |
| 权限检查 | PermissionChecker 实现细粒度权限控制 |
| 沙箱隔离 | Sandbox 限制危险操作 |
| MCP 集成 | 支持 Model Context Protocol 外部工具 |
| Skill 扩展 | 支持动态加载 Skill 工具 |

### 3.3 Tool 抽象基类

```python
class Tool(ABC):
    """工具抽象基类"""

    @property
    @abstractmethod
    def name(self) -> str:
        """工具名称"""
        pass

    @property
    @abstractmethod
    def description(self) -> str:
        """工具描述"""
        pass

    @property
    def input_schema(self) -> Dict[str, Any]:
        """输入 Schema (JSON Schema)"""
        return {}

    @property
    def output_schema(self) -> Dict[str, Any]:
        """输出 Schema"""
        return {}

    @property
    def timeout(self) -> int:
        """超时时间（秒）"""
        return 30

    @property
    def write_tool(self) -> bool:
        """是否为写操作工具"""
        return False

    @abstractmethod
    async def execute(self, params: Dict[str, Any], context: ToolContext) -> ToolResult:
        """执行工具"""
        pass

    def validate_params(self, params: Dict[str, Any]) -> bool:
        """验证参数"""
        try:
            jsonschema.validate(params, self.input_schema)
            return True
        except jsonschema.ValidationError:
            return False
```

## 4. 工具注册表

### 4.1 ToolRegistry

```python
class ToolRegistry:
    """工具注册表"""

    def __init__(self):
        self._tools: Dict[str, Tool] = {}
        self._builtin_registered = False

    def register(self, tool: Tool) -> None:
        """注册工具"""
        if tool.name in self._tools:
            raise ToolAlreadyRegisteredError(f"Tool {tool.name} already registered")
        self._tools[tool.name] = tool

    def unregister(self, name: str) -> None:
        """注销工具"""
        if name in self._tools:
            del self._tools[name]

    def get(self, name: str) -> Optional[Tool]:
        """获取工具"""
        return self._tools.get(name)

    def list_tools(self) -> List[ToolMetadata]:
        """列出所有工具（包含元数据）"""
        return [
            ToolMetadata(
                name=tool.name,
                description=tool.description,
                input_schema=tool.input_schema,
                write_tool=tool.write_tool,
            )
            for tool in self._tools.values()
        ]

    def register_builtin(self) -> None:
        """注册内置工具"""
        if self._builtin_registered:
            return

        self.register(ReadTool())
        self.register(WriteTool())
        self.register(EditTool())
        self.register(BashTool())
        self.register(GrepTool())
        self.register(GlobTool())

        self._builtin_registered = True

    def register_mcp_tools(self, tools: List[Tool]) -> None:
        """注册 MCP 工具"""
        for tool in tools:
            self.register(tool)

    def register_skill_tools(self, tools: List[Tool]) -> None:
        """注册 Skill 工具"""
        for tool in tools:
            self.register(tool)
```

## 5. 工具执行器

### 5.1 ToolExecutor

```python
class ToolExecutor:
    """工具执行器"""

    def __init__(
        self,
        registry: ToolRegistry,
        permission_checker: PermissionChecker,
        sandbox: Sandbox,
    ):
        self.registry = registry
        self.permission_checker = permission_checker
        self.sandbox = sandbox

    async def execute(
        self,
        tool_name: str,
        params: Dict[str, Any],
        context: ToolContext,
    ) -> ToolResult:
        """执行工具"""
        # 1. 获取工具
        tool = self.registry.get(tool_name)
        if not tool:
            return ToolResult(success=False, error=f"Tool {tool_name} not found")

        # 2. 权限检查
        permission = await self.permission_checker.check(tool_name, params)
        if permission.status == PermissionStatus.DENY:
            return ToolResult(success=False, error=f"Permission denied for {tool_name}")

        if permission.status == PermissionStatus.ASK:
            return ToolResult(
                success=False,
                error="HITL confirmation required",
                metadata={"requires_approval": True}
            )

        # 3. 沙箱检查
        if not await self.sandbox.check(tool_name, params):
            return ToolResult(success=False, error=f"Sandbox restriction for {tool_name}")

        # 4. 参数验证
        if not tool.validate_params(params):
            return ToolResult(success=False, error="Invalid parameters")

        # 5. 执行
        try:
            result = await asyncio.wait_for(
                tool.execute(params, context),
                timeout=tool.timeout
            )
            return result
        except asyncio.TimeoutError:
            return ToolResult(success=False, error=f"Tool {tool_name} timed out")
        except Exception as e:
            return ToolResult(success=False, error=str(e))
```

## 6. 内置工具

### 6.1 ReadTool

读取文件内容，支持行号分页。

```python
class ReadTool(Tool):
    """读取文件工具"""

    @property
    def name(self) -> str:
        return "read"

    @property
    def description(self) -> str:
        return "读取文件内容"

    @property
    def input_schema(self) -> Dict[str, Any]:
        return {
            "type": "object",
            "properties": {
                "path": {"type": "string", "description": "文件路径"},
                "offset": {"type": "integer", "description": "起始行号", "default": 0},
                "limit": {"type": "integer", "description": "读取行数", "default": 1000},
            },
            "required": ["path"]
        }

    @property
    def output_schema(self) -> Dict[str, Any]:
        return {
            "type": "object",
            "properties": {
                "content": {"type": "string"},
                "metadata": {
                    "type": "object",
                    "properties": {
                        "lines_read": {"type": "integer"},
                        "total_lines": {"type": "integer"}
                    }
                }
            }
        }

    async def execute(self, params: Dict[str, Any], context: ToolContext) -> ToolResult:
        try:
            path = Path(params["path"])
            if not path.exists():
                return ToolResult(success=False, error=f"File not found: {path}")

            content = path.read_text(encoding="utf-8")
            lines = content.split("\n")

            offset = params.get("offset", 0)
            limit = params.get("limit", 1000)
            lines = lines[offset:offset + limit]

            return ToolResult(
                success=True,
                content="\n".join(lines),
                metadata={"lines_read": len(lines), "total_lines": len(content.split("\n"))}
            )
        except Exception as e:
            return ToolResult(success=False, error=str(e))
```

                success=True,
                content=f"File written: {path}",
                metadata={"bytes_written": len(params["content"])}
            )

### 6.3 EditTool (HashAnchorEdit)

带哈希锚定的安全编辑，防止并发冲突。

```python
class EditTool(Tool):
    """带哈希锚定的安全编辑工具"""

    @property
    def name(self) -> str:
        return "edit"

    @property
    def description(self) -> str:
        return "带哈希验证的安全编辑"

    @property
    def write_tool(self) -> bool:
        return True

    @property
    def input_schema(self) -> Dict[str, Any]:
        return {
            "type": "object",
            "properties": {
                "path": {"type": "string", "description": "文件路径"},
                "old": {"type": "string", "description": "原内容（用于验证）"},
                "new": {"type": "string", "description": "新内容"},
            },
            "required": ["path", "old", "new"]
        }

    async def execute(self, params: Dict[str, Any], context: ToolContext) -> ToolResult:
        path = Path(params["path"])
        try:
            current = path.read_text(encoding="utf-8")
            old_content = params["old"]
            new_content = params["new"]

            # 验证 old_content 是否存在于当前文件
            if old_content not in current:
                return ToolResult(
                    success=False,
                    error="Old content not found in file"
                )

            # 执行替换（只替换第一个匹配）
            updated = current.replace(old_content, new_content, 1)
            path.write_text(updated, encoding="utf-8")

            return ToolResult(success=True, content=f"File edited: {path}")
        except FileNotFoundError:
            return ToolResult(success=False, error=f"File not found: {path}")
        except Exception as e:
            return ToolResult(success=False, error=str(e))
```

### 6.4 BashTool

```python
class BashTool(Tool):
    """Shell 命令执行工具"""

    @property
    def name(self) -> str:
        return "bash"

    @property
    def description(self) -> str:
        return "执行 Shell 命令"

    @property
    def write_tool(self) -> bool:
        return True

    @property
    def input_schema(self) -> Dict[str, Any]:
        return {
            "type": "object",
            "properties": {
                "command": {"type": "string", "description": "Shell 命令"},
                "timeout": {"type": "integer", "description": "超时时间(秒)", "default": 30},
                "cwd": {"type": "string", "description": "工作目录"},
            },
            "required": ["command"]
        }

    async def execute(self, params: Dict[str, Any], context: ToolContext) -> ToolResult:
        command = params["command"]
        timeout = params.get("timeout", 30)
        cwd = params.get("cwd")

        try:
            result = await asyncio.create_subprocess_shell(
                command,
                stdout=asyncio.subprocess.PIPE,
                stderr=asyncio.subprocess.PIPE,
                cwd=cwd,
            )

            try:
                stdout, stderr = await asyncio.wait_for(
                    result.communicate(),
                    timeout=timeout
                )
            except asyncio.TimeoutError:
                result.kill()
                return ToolResult(success=False, error=f"Command timed out after {timeout}s")

            return ToolResult(
                success=result.returncode == 0,
                content=stdout.decode() if stdout else "",
                error=stderr.decode() if stderr else None,
                metadata={
                    "return_code": result.returncode,
                    "command": command,
                }
            )

        except Exception as e:
            return ToolResult(success=False, error=str(e))
```

### 6.5 GrepTool

正则表达式内容搜索。

```python
class GrepTool(Tool):
    """内容搜索工具"""

    @property
    def name(self) -> str:
        return "grep"

    @property
    def description(self) -> str:
        return "正则表达式内容搜索"

    @property
    def input_schema(self) -> Dict[str, Any]:
        return {
            "type": "object",
            "properties": {
                "pattern": {"type": "string", "description": "正则表达式"},
                "path": {"type": "string", "description": "搜索路径"},
                "context": {"type": "integer", "description": "上下文行数", "default": 0},
                "case_sensitive": {"type": "boolean", "description": "大小写敏感", "default": True},
            },
            "required": ["pattern"]
        }

    async def execute(self, params: Dict[str, Any], context: ToolContext) -> ToolResult:
        try:
            pattern = params["pattern"]
            path = params.get("path", ".")
            context_lines = params.get("context", 0)
            case_sensitive = params.get("case_sensitive", True)

            cmd = ["grep"]
            if not case_sensitive:
                cmd.append("-i")
            if context_lines > 0:
                cmd.extend(["-C", str(context_lines)])
            cmd.extend([pattern, path])

            result = await asyncio.create_subprocess_shell(
                " ".join(cmd),
                stdout=asyncio.subprocess.PIPE,
                stderr=asyncio.subprocess.PIPE,
            )
            stdout, stderr = await result.communicate()

            return ToolResult(
                success=result.returncode in (0, 1),
                content=stdout.decode() if stdout else "",
                error=stderr.decode() if stderr else None,
                metadata={"matches": stdout.decode().count("\n") if stdout else 0}
            )
        except Exception as e:
            return ToolResult(success=False, error=str(e))
```

## 7. 扩展机制

### 7.1 MCP Client

Model Context Protocol 客户端，支持调用外部工具。

```python
class MCPClient:
    """MCP 协议客户端"""

    def __init__(self, transport: Transport):
        self.transport = transport
        self.servers: Dict[str, MCPServer] = {}

    async def connect(self, name: str, config: MCPConfig) -> None:
        """连接 MCP 服务器"""
        server = MCPServer(name=name, config=config)
        await server.connect()
        self.servers[name] = server

    async def call_tool(self, server: str, tool: str, args: dict) -> Any:
        """调用 MCP 工具"""
        return await self.servers[server].call_tool(tool, args)

    async def list_tools(self, server: str) -> List[Tool]:
        """列出服务器工具"""
        return await self.servers[server].list_tools()

    async def list_resources(self, server: str) -> List[Resource]:
        """列出服务器资源"""
        return await self.servers[server].list_resources()
```

### 7.3 Skill Loader

动态加载 Skill 定义的工具。

```python
class SkillLoader:
    """Skill 加载器"""

    def __init__(self, skill_dir: Path):
        self.skill_dir = skill_dir

    async def load_skill(self, skill_name: str) -> List[Tool]:
        """加载单个 Skill"""
        skill_path = self.skill_dir / f"{skill_name}.md"
        if not skill_path.exists():
            raise SkillNotFoundError(f"Skill {skill_name} not found")

        parser = SkillParser()
        skill_def = await parser.parse(skill_path)

        return self._tools_from_skill(skill_def)

    async def load_all(self) -> List[Tool]:
        """加载所有 Skills"""
        tools = []
        for skill_file in self.skill_dir.glob("*.md"):
            try:
                tools.extend(await self.load_skill(skill_file.stem))
            except Exception:
                continue
        return tools
```

### 8.2 权限模式映射

| Claude Code 模式 | Mozi PermissionChecker | 行为 |
|------------------|------------------------|------|
| `default` | `ASK` | 操作前确认 (HITL) |
| `acceptEdits` | `ALLOW` | 自动批准 |
| `dontAsk` | `DENY` | 自动拒绝 |
| `bypassPermissions` | `ADMIN` | 跳过所有检查 |
| `plan` | `READONLY` | 仅允许读操作 |

```python
class PermissionChecker:
    """权限检查器"""

    WRITE_TOOLS = {"write", "edit", "bash", "multiedit", "notebook"}

    async def check(self, tool_name: str, params: Dict[str, Any]) -> PermissionResult:
        """检查工具执行权限"""
        if self.mode == PermissionMode.BYPASS:
            return PermissionResult(status=PermissionStatus.ALLOW)

        tool_permission = self._get_tool_permission(tool_name)

        if tool_permission == PermissionMode.DENY:
            return PermissionResult(status=PermissionStatus.DENY)

        if tool_permission == PermissionMode.ASK:
            return PermissionResult(status=PermissionStatus.ASK)

        if self.mode == PermissionMode.PLAN:
            if tool_name in self.WRITE_TOOLS:
                return PermissionResult(status=PermissionStatus.DENY)

        return PermissionResult(status=PermissionStatus.ALLOW)
```
