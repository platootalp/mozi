## 1. 模块概述

Config 模块是系统的配置管理模块，负责配置的统一管理和错误处理。

### 1.1 核心职责

- 配置项定义
- 配置加载与验证
- 错误码管理
- 降级策略

### 1.2 差异化特性

| 特性 | 说明 |
|------|------|
| 分层配置 | 环境变量 > 用户配置 > 默认配置 |
| 类型安全 | Pydantic 模型验证 |
| 动态重载 | 支持运行时重载配置 |
| 作用域隔离 | 多租户配置隔离 |

## 4. 配置加载器

### 4.1 ConfigLoader

```python
class ConfigLoader:
    """配置加载器"""

    DEFAULT_CONFIG_DIR = Path("~/.mozi").expanduser()
    DEFAULT_CONFIG_FILE = "config.json"

    def __init__(self):
        self.config: Optional[MoziConfig] = None

    async def load(
        self,
        config_path: Optional[Path] = None,
        config_dir: Optional[Path] = None,
    ) -> MoziConfig:
        """
        加载配置
        优先级: 环境变量 > config_path > config_dir > 默认
        """
        # 1. 加载默认配置
        config = self._get_default_config()

        # 2. 加载用户配置
        config_dir = config_dir or self.DEFAULT_CONFIG_DIR
        user_config_path = config_dir / self.DEFAULT_CONFIG_FILE

        if user_config_path.exists():
            user_config = await self._load_from_file(user_config_path)
            config = self._merge_config(config, user_config)

        # 3. 加载命令行指定配置
        if config_path and config_path.exists():
            cmd_config = await self._load_from_file(config_path)
            config = self._merge_config(config, cmd_config)

        # 4. 加载环境变量
        env_config = self._load_from_env()
        config = self._merge_config(config, env_config)

        # 5. 验证配置
        validator = ConfigValidator()
        validator.validate(config)

        self.config = config
        return config

    def _get_default_config(self) -> MoziConfig:
        """获取默认配置"""
        return MoziConfig()

    async def _load_from_file(self, path: Path) -> Dict[str, Any]:
        """从文件加载"""
        content = path.read_text()

        if path.suffix == ".json":
            return json.loads(content)
        elif path.suffix in (".yaml", ".yml"):
            return yaml.safe_load(content)
        else:
            raise ConfigError(f"Unsupported config file format: {path.suffix}")

    def _load_from_env(self) -> Dict[str, Any]:
        """从环境变量加载"""
        env_config = {}

        if model := os.getenv("MOZI_MODEL"):
            env_config.setdefault("model", {})["name"] = model

        if provider := os.getenv("MOZI_PROVIDER"):
            env_config.setdefault("model", {})["provider"] = provider

        if api_key := os.getenv("ANTHROPIC_API_KEY"):
            env_config.setdefault("model", {})["api_key"] = api_key

        if sandbox := os.getenv("MOZI_SANDBOX_MODE"):
            env_config.setdefault("security", {})["sandbox_mode"] = sandbox

        if skills_dir := os.getenv("MOZI_SKILLS_DIR"):
            env_config.setdefault("extensions", {})["skills_dir"] = skills_dir

        return env_config

    def _merge_config(
        self,
        base: Union[MoziConfig, Dict[str, Any]],
        override: Dict[str, Any],
    ) -> MoziConfig:
        """合并配置"""
        if isinstance(base, MoziConfig):
            base_dict = base.model_dump()
        else:
            base_dict = base

        merged = self._deep_merge(base_dict, override)
        return MoziConfig(**merged)

    def _deep_merge(
        self,
        base: Dict[str, Any],
        override: Dict[str, Any],
    ) -> Dict[str, Any]:
        """深度合并字典"""
        result = base.copy()

        for key, value in override.items():
            if key in result and isinstance(result[key], dict) and isinstance(value, dict):
                result[key] = self._deep_merge(result[key], value)
            else:
                result[key] = value

        return result
```

### 4.2 配置验证器

```python
class ConfigValidator:
    """配置验证器"""

    def validate(self, config: MoziConfig) -> None:
        """验证配置"""
        errors = []

        # 验证模型配置
        if not config.model.provider:
            errors.append("model.provider is required")

        if config.model.timeout <= 0:
            errors.append("model.timeout must be positive")

        if config.model.max_retries < 0:
            errors.append("model.max_retries must be non-negative")

        # 验证记忆配置
        if config.memory.working_max_size <= 0:
            errors.append("memory.working_max_size must be positive")

        if config.memory.short_term_ttl_days <= 0:
            errors.append("memory.short_term_ttl_days must be positive")

        # 验证安全配置
        if config.security.sandbox_mode not in ("off", "non-main", "all"):
            errors.append("security.sandbox_mode must be one of: off, non-main, all")

        # 验证编排配置
        if config.orchestration.max_iterations <= 0:
            errors.append("orchestration.max_iterations must be positive")

        if errors:
            raise ConfigValidationError("\n".join(errors))
```

### 5.2 错误处理

```python
class ErrorHandler:
    """错误处理器"""

    def __init__(self, audit_logger: AuditLogger):
        self.audit_logger = audit_logger

    async def handle(self, error: Exception, context: Optional[Context] = None) -> ErrorResponse:
        """处理错误"""
        if isinstance(error, MoziError):
            return await self._handle_mozi_error(error, context)
        else:
            return await self._handle_unknown_error(error, context)

    async def _handle_mozi_error(
        self,
        error: MoziError,
        context: Optional[Context],
    ) -> ErrorResponse:
        """处理 Mozi 错误"""
        # 记录审计日志
        await self.audit_logger.log(
            action="error",
            session_id=context.session_id if context else None,
            result=f"{error.code.value[0]}: {error.message}",
        )

        return ErrorResponse(
            code=error.code.value[0],
            message=error.message,
            details=error.details,
        )

    async def _handle_unknown_error(
        self,
        error: Exception,
        context: Optional[Context],
    ) -> ErrorResponse:
        """处理未知错误"""
        # 记录审计日志
        await self.audit_logger.log(
            action="error",
            session_id=context.session_id if context else None,
            result=f"UNKNOWN: {str(error)}",
        )

        return ErrorResponse(
            code="E9999",
            message="An unexpected error occurred",
            details={"error_type": type(error).__name__},
        )

@dataclass
class ErrorResponse:
    code: str
    message: str
    details: Optional[Dict[str, Any]] = None
```

## 6. 降级策略

### 6.1 降级管理器

```python
class FallbackManager:
    """降级策略管理器"""

    def __init__(self, config: MoziConfig):
        self.config = config
        self._fallback_chain: Dict[str, FallbackStrategy] = {}

        self._init_fallbacks()

    def _init_fallbacks(self) -> None:
        """初始化降级策略"""
        # 模型降级
        self._fallback_chain["model"] = ModelFallback([
            ModelFallbackItem(
                provider="anthropic",
                model="claude-sonnet-4-20250514",
            ),
            ModelFallbackItem(
                provider="openai",
                model="gpt-4o",
            ),
            ModelFallbackItem(
                provider="openai",
                model="gpt-3.5-turbo",
            ),
        ])

        # 存储降级
        self._fallback_chain["storage"] = StorageFallback([
            StorageFallbackItem(type="memory"),
        ])

        # 记忆降级
        self._fallback_chain["memory"] = MemoryFallback([
            MemoryFallbackItem(type="compressed"),
            MemoryFallbackItem(type="minimal"),
        ])

    async def execute_with_fallback(
        self,
        operation: str,
        func: Callable,
        *args,
        **kwargs,
    ) -> Any:
        """使用降级策略执行操作"""
        if operation not in self._fallback_chain:
            return await func(*args, **kwargs)

        fallback = self._fallback_chain[operation]

        for attempt in range(fallback.max_attempts):
            try:
                return await func(*args, **kwargs)
            except Exception as e:
                if attempt == fallback.max_attempts - 1:
                    raise

                # 尝试降级
                fallback_strategy = fallback.strategies[attempt + 1]
                await fallback_strategy.apply(self.config)
                logger.warning(f"Fallback to {fallback_strategy} after error: {e}")

class FallbackStrategy(ABC):
    """降级策略基类"""

    @abstractmethod
    async def apply(self, config: MoziConfig) -> None:
        pass

class ModelFallbackItem:
    def __init__(self, provider: str, model: str):
        self.provider = provider
        self.model = model

class StorageFallbackItem:
    def __init__(self, type: str):
        self.type = type

class MemoryFallbackItem:
    def __init__(self, type: str):
        self.type = type
```

### 6.2 降级策略详情

#### 模型降级

```python
class ModelFallback(FallbackStrategy):
    """模型降级策略"""

    def __init__(self, strategies: List[ModelFallbackItem]):
        self.strategies = strategies
        self.max_attempts = len(strategies)

    async def apply(self, config: MoziConfig) -> None:
        """切换到下一个模型"""
        current_provider = config.model.provider
        current_model = config.model.name

        for i, strategy in enumerate(self.strategies):
            if (strategy.provider, strategy.model) == (current_provider, current_model):
                if i + 1 < len(self.strategies):
                    next_strategy = self.strategies[i + 1]
                    config.model.provider = next_strategy.provider
                    config.model.name = next_strategy.model
                    logger.info(f"Falling back to {next_strategy.provider}/{next_strategy.model}")
                break
```

#### 存储降级

```python
class StorageFallback(FallbackStrategy):
    """存储降级策略"""

    def __init__(self, strategies: List[StorageFallbackItem]):
        self.strategies = strategies
        self.max_attempts = len(strategies)

    async def apply(self, config: MoziConfig) -> None:
        """降级到备用存储"""
        # 在内存模式下降级
        config.storage_mode = "memory"
        logger.warning("Falling back to in-memory storage")
```

### 6.3 重试机制

```python
class RetryPolicy:
    """重试策略"""

    def __init__(
        self,
        max_retries: int = 3,
        base_delay: float = 1.0,
        max_delay: float = 60.0,
        exponential_base: float = 2.0,
    ):
        self.max_retries = max_retries
        self.base_delay = base_delay
        self.max_delay = max_delay
        self.exponential_base = exponential_base

    def get_delay(self, attempt: int) -> float:
        """计算延迟（指数退避）"""
        delay = self.base_delay * (self.exponential_base ** attempt)
        return min(delay, self.max_delay)

    async def execute(self, func: Callable, *args, **kwargs) -> Any:
        """执行带重试的函数"""
        last_error = None

        for attempt in range(self.max_retries + 1):
            try:
                return await func(*args, **kwargs)
            except Exception as e:
                last_error = e

                if attempt < self.max_retries:
                    delay = self.get_delay(attempt)
                    logger.warning(f"Retry {attempt + 1}/{self.max_retries} after {delay}s: {e}")
                    await asyncio.sleep(delay)
                else:
                    logger.error(f"All retries exhausted: {e}")

        raise last_error
```

## 8. Claude Code Configuration Comparison

### 8.1 Configuration System Overview

| 特性 | Claude Code | Mozi |
|------|-------------|------|
| 配置位置 | `~/.claude/settings.json` (用户级), `.claude/` (项目级) | `~/.mozi/config.json` (用户级) |
| 配置格式 | JSON | JSON/YAML |
| 类型验证 | 无内置验证 | Pydantic 模型验证 |
| 配置优先级 | 项目级 > 用户级 > 默认 | 环境变量 > 命令行 > 用户配置 > 默认 |

### 8.3 Mozi vs Claude Code Hooks

Claude Code 支持 `hooks` 配置用于在工具使用和消息发送前后执行自定义逻辑。Mozi 应借鉴此设计：

```python
# 推荐的 Hook 配置
class HookConfig(BaseModel):
    """Hook 配置"""
    pre_tool_hook: Optional[str] = None      # 工具执行前钩子
    post_tool_hook: Optional[str] = None    # 工具执行后钩子
    pre_message_hook: Optional[str] = None   # 消息发送前钩子
    post_message_hook: Optional[str] = None  # 消息接收后钩子

# Hook 调用示例
class HookManager:
    """Hook 管理器"""

    def __init__(self, config: HookConfig):
        self.config = config

    async def execute_pre_tool(self, tool_name: str, args: Dict[str, Any]) -> None:
        """执行工具前钩子"""
        if self.config.pre_tool_hook:
            await self._run_hook(self.config.pre_tool_hook, {
                "tool": tool_name,
                "args": args
            })

    async def execute_post_tool(
        self,
        tool_name: str,
        args: Dict[str, Any],
        result: Any
    ) -> None:
        """执行工具后钩子"""
        if self.config.post_tool_hook:
            await self._run_hook(self.config.post_tool_hook, {
                "tool": tool_name,
                "args": args,
                "result": result
            })
```

### 8.4 Recommendations for Mozi

1. **增加 Hook 支持**: 在 `ExtensionsConfig` 中增加 `hooks` 配置项，支持预定义钩子点
2. **配置文件分离**: 参考 Claude Code 的 `.claude/` 目录结构，将 hooks/scripts 从主配置中分离
3. **权限配置增强**: 借鉴 Claude Code 的细粒度权限控制，支持按工具类型授权
