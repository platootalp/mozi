## 1. 模块概述

Context 模块负责管理 Agent 执行时的上下文，包括上下文组装、Token 预算管理和记忆检索。Context 模块与 Memory 模块紧密协作，从 Memory 获取历史记忆并组装成可用的上下文。

### 1.1 核心职责

- 上下文组装
- Token 预算管理
- 上下文压缩
- 记忆检索协调

### 1.2 差异化特性

| 特性 | 说明 |
|------|------|
| 智能压缩 | 按语义压缩，而非简单截断 |
| 预算控制 | 精确的 Token 计数与预算 |
| 多源组合 | 组合 Working/Short-term/Long-term 记忆 |

## 3. ContextManager

### 3.1 核心接口

```python
class ContextManager:
    """上下文管理器"""

    def __init__(
        self,
        memory: MemoryManager,
        config: ContextConfig,
    ):
        self.memory = memory
        self.config = config
        self.assembler = ContextAssembler(memory, config)
        self.budget = TokenBudget(config.max_tokens)
        self.compressor = ContextCompressor(config.compression)

    async def assemble(self, task: Task) -> AssembledContext:
        """
        组装任务上下文
        1. 从 Memory 获取记忆
        2. 组合成完整上下文
        3. Token 预算检查
        4. 必要时压缩
        """
        # 1. 获取三层记忆
        memory_context = await self.memory.get_context(task)

        # 2. 组装上下文
        assembled = await self.assembler.assemble(task, memory_context)

        # 3. Token 预算检查
        token_count = self._count_tokens(assembled)
        remaining_budget = self.budget.get_remaining()

        if token_count > remaining_budget:
            # 压缩上下文
            assembled = await self.compressor.compress(assembled, remaining_budget)

        return AssembledContext(
            prompt=assembled,
            token_count=self._count_tokens(assembled),
            budget_used=self.budget.get_used(),
            budget_remaining=self.budget.get_remaining(),
        )

    def _count_tokens(self, context: PromptContext) -> int:
        """计算 Token 数量"""
        # 使用 tiktoken 或估算
        return self.config.tokenizer.count(context)
```

### 3.2 组装流程

```python
class ContextAssembler:
    """上下文组装器"""

    def __init__(self, memory: MemoryManager, config: ContextConfig):
        self.memory = memory
        self.config = config

    async def assemble(
        self,
        task: Task,
        memory_context: MemoryContext,
    ) -> PromptContext:
        """
        组装完整上下文
        优先级: system > working > short_term > long_term
        """
        parts = []

        # 1. 系统 Prompt
        if self.config.include_system_prompt:
            system = self._get_system_prompt(task)
            parts.append(system)

        # 2. Working Memory（滑动窗口）
        if memory_context.working:
            working = self._format_working_memory(memory_context.working)
            parts.append(working)

        # 3. Short-term Memory（RAG 检索结果）
        if memory_context.short_term:
            short_term = self._format_short_term(memory_context.short_term)
            parts.append(short_term)

        # 4. Long-term Memory（用户偏好）
        if memory_context.long_term:
            long_term = self._format_long_term(memory_context.long_term)
            parts.append(long_term)

        # 5. 当前任务
        parts.append(f"\n\n## Current Task\n{task.description}")

        return PromptContext(parts=parts)

    def _format_working_memory(self, messages: List[Message]) -> str:
        """格式化 Working Memory"""
        return "\n".join(f"{m.role}: {m.content}" for m in messages)

    def _format_short_term(self, entries: List[MemoryEntry]) -> str:
        """格式化 Short-term Memory"""
        if not entries:
            return ""
        formatted = "\n## Relevant History\n"
        for entry in entries:
            formatted += f"- {entry.content}\n"
        return formatted

    def _format_long_term(self, preferences: UserPreferences) -> str:
        """格式化 Long-term Memory"""
        return f"""
## User Preferences
- Preferred Model: {preferences.preferred_model or 'auto'}
- Language: {preferences.preferred_language}
- Coding Style: {preferences.coding_style or 'default'}
"""
```

## 4. TokenBudget

### 4.1 Token 预算管理

```python
class TokenBudget:
    """Token 预算管理器"""

    def __init__(
        self,
        max_tokens: int,
        reserved_tokens: int = 1000,
    ):
        self.max_tokens = max_tokens
        self.reserved_tokens = reserved_tokens
        self._used = 0

    def get_remaining(self) -> int:
        """获取剩余 Token 预算"""
        return self.max_tokens - self.reserved_tokens - self._used

    def add_usage(self, tokens: int) -> None:
        """增加 Token 使用"""
        self._used += tokens

    def get_used(self) -> int:
        """获取已使用 Token"""
        return self._used

    def reset(self) -> None:
        """重置预算"""
        self._used = 0

    def reserve(self, tokens: int) -> bool:
        """
        预留 Token
        返回是否预留成功
        """
        if self.get_remaining() >= tokens:
            self.reserved_tokens += tokens
            return True
        return False
```

## 5. ContextCompressor

### 5.1 上下文压缩

```python
class ContextCompressor:
    """上下文压缩器"""

    def __init__(self, config: CompressionConfig):
        self.config = config

    async def compress(
        self,
        context: PromptContext,
        target_tokens: int,
    ) -> PromptContext:
        """
        压缩上下文到目标 Token 数量
        使用语义压缩，而非简单截断
        """
        current_tokens = self._count_tokens(context)

        if current_tokens <= target_tokens:
            return context

        # 计算需要压缩的比例
        ratio = target_tokens / current_tokens

        # 分层压缩策略
        if ratio > 0.7:
            # 轻度压缩：移除最旧的 Working Memory
            context = await self._light_compress(context, ratio)
        elif ratio > 0.4:
            # 中度压缩：Summary + 保留关键记忆
            context = await self._medium_compress(context, ratio)
        else:
            # 重度压缩：生成摘要
            context = await self._heavy_compress(context, ratio)

        return context

    async def _light_compress(
        self,
        context: PromptContext,
        ratio: float,
    ) -> PromptContext:
        """轻度压缩：移除部分 Working Memory"""
        # 保留最近的 50% Working Memory
        keep_count = int(len(context.working) * ratio)
        context.working = context.working[-keep_count:]
        return context

    async def _medium_compress(
        self,
        context: PromptContext,
        ratio: float,
    ) -> PromptContext:
        """中度压缩：Summary + 关键记忆"""
        # 生成 Working Memory 摘要
        if context.working:
            summary = await self._summarize_messages(context.working)
            context.working = [summary] + context.working[-3:]

        # 保留最重要的 Short-term Memory
        if context.short_term:
            keep_count = max(1, int(len(context.short_term) * ratio))
            context.short_term = context.short_term[:keep_count]

        return context

    async def _heavy_compress(
        self,
        context: PromptContext,
        ratio: float,
    ) -> PromptContext:
        """重度压缩：只保留摘要和最新任务"""
        # 生成完整摘要
        full_summary = await self._summarize_context(context)

        context.parts = [full_summary, context.parts[-1]]  # 只保留任务
        return context

    async def _summarize_messages(self, messages: List[Message]) -> Message:
        """生成消息摘要"""
        prompt = f"Summarize this conversation concisely:\n{messages}"
        summary = await self.model.complete(prompt)
        return Message(role="system", content=f"Summary: {summary}")

    async def _summarize_context(self, context: PromptContext) -> str:
        """生成上下文摘要"""
        prompt = f"""Summarize the following context concisely, preserving key information:

Working Memory: {context.working}
Short-term Memory: {context.short_term}
Long-term Memory: {context.long_term}

Provide a concise summary of:
1. What has been discussed/worked on
2. Key decisions made
3. Current state
"""
        return await self.model.complete(prompt)
```

## 9. CLAUDE.md Project Context

CLAUDE.md 是 Claude Code 的项目上下文文件，用于向 AI 代理提供项目特定的项目概览、架构、规则和约定。本节定义如何在 Context 模块中加载和管理 CLAUDE.md 项目上下文。

### 9.1 CLAUDE.md 定位

| 属性 | 值 |
|------|---|
| 文件位置 | `.claude/CLAUDE.md` 或项目根目录 `CLAUDE.md` |
| 加载时机 | Session 启动时 |
| 上下文优先级 | 高于 Memory，低于 System Prompt |
| 更新频率 | 静态文件，人工维护 |

### 9.3 CLAUDE.md 加载机制

```python
class ClaudeMdLoader:
    """CLAUDE.md 加载器"""

    SEARCH_PATHS = [
        ".claude/CLAUDE.md",
        "CLAUDE.md",
        ".claude.md",
    ]

    def __init__(self, project_root: Path):
        self.project_root = project_root
        self._cached: Optional[ClaudeMdSchema] = None

    async def load(self, force_reload: bool = False) -> Optional[ClaudeMdSchema]:
        """
        加载 CLAUDE.md 文件
        1. 搜索标准路径
        2. 解析 YAML/TOML 前置元数据
        3. 验证结构
        4. 缓存结果
        """
        if self._cached and not force_reload:
            return self._cached

        for path in self.SEARCH_PATHS:
            full_path = self.project_root / path
            if full_path.exists():
                content = await self._read_file(full_path)
                parsed = self._parse(content)
                self._cached = parsed
                return parsed

        return None

    async def _read_file(self, path: Path) -> str:
        """读取文件内容"""
        return path.read_text(encoding="utf-8")

    def _parse(self, content: str) -> ClaudeMdSchema:
        """
        解析 CLAUDE.md 内容
        - 提取前置元数据（YAML/TOML 格式）
        - 解析 Markdown 主体
        - 构建结构化 schema
        """
        # 提取 frontmatter
        frontmatter, body = self._extract_frontmatter(content)

        # 解析前置元数据
        metadata = self._parse_frontmatter(frontmatter)

        # 构建 schema
        return ClaudeMdSchema(
            project_name=metadata.get("project_name", "Unknown"),
            project_overview=body,
            architecture=metadata.get("architecture"),
            conventions=metadata.get("conventions"),
            rules=self._parse_rule_references(metadata.get("rules", [])),
            build_commands=metadata.get("build_commands"),
            development_commands=metadata.get("development"),
            dependencies=metadata.get("dependencies"),
        )

    def _extract_frontmatter(self, content: str) -> Tuple[str, str]:
        """提取前置元数据"""
        if content.startswith("---"):
            parts = content.split("---", 2)
            if len(parts) == 3:
                return parts[1], parts[2]
        return "", content

    def _parse_frontmatter(self, frontmatter: str) -> Dict:
        """解析前置元数据（支持 YAML 格式）"""
        # 使用 ruamel.yaml 或 pyyaml 解析
        if not frontmatter.strip():
            return {}
        return yaml.safe_load(frontmatter) or {}

    def _parse_rule_references(self, rules: List[Dict]) -> List[RuleReference]:
        """解析规则引用"""
        return [RuleReference(**r) for r in rules]
```

### 9.4 上下文集成

```python
class ContextAssembler:
    """上下文组装器（扩展支持 CLAUDE.md）"""

    # CLAUDE.md 上下文优先级
    CLAUDE_MD_PRIORITY = 0.85  # 介于 System Prompt 和 Memory 之间

    async def assemble(
        self,
        task: Task,
        memory_context: MemoryContext,
        claude_md: Optional[ClaudeMdSchema] = None,
    ) -> PromptContext:
        """
        组装完整上下文
        优先级: system > claude_md > working > short_term > long_term
        """
        parts = []

        # 1. 系统 Prompt（最高优先级）
        if self.config.include_system_prompt:
            system = self._get_system_prompt(task)
            parts.append(system)

        # 2. CLAUDE.md 项目上下文
        if claude_md and self.config.include_claude_md:
            claude_context = self._format_claude_md(claude_md)
            parts.append(claude_context)

        # 3. Working Memory（滑动窗口）
        if memory_context.working:
            working = self._format_working_memory(memory_context.working)
            parts.append(working)

        # 4. Short-term Memory（RAG 检索结果）
        if memory_context.short_term:
            short_term = self._format_short_term(memory_context.short_term)
            parts.append(short_term)

        # 5. Long-term Memory（用户偏好）
        if memory_context.long_term:
            long_term = self._format_long_term(memory_context.long_term)
            parts.append(long_term)

        # 6. 当前任务
        parts.append(f"\n\n## Current Task\n{task.description}")

        return PromptContext(parts=parts)

    def _format_claude_md(self, claude_md: ClaudeMdSchema) -> str:
        """格式化 CLAUDE.md 内容"""
        sections = ["## Project Context"]

        # 项目概览
        sections.append(f"### {claude_md.project_name}")
        sections.append(claude_md.project_overview)

        # 架构信息
        if claude_md.architecture:
            arch = claude_md.architecture
            sections.append("\n### Architecture")
            if arch.layers:
                sections.append(f"**Layers**: {' -> '.join(arch.layers)}")
            if arch.key_terminology:
                sections.append("\n**Key Terms**:")
                for term, definition in arch.key_terminology.items():
                    sections.append(f"- {term}: {definition}")

        # 关键路径
        if claude_md.key_paths:
            sections.append("\n### Key Paths")
            for name, path in claude_md.key_paths.items():
                sections.append(f"- {name}: `{path}`")

        # 约定
        if claude_md.conventions:
            sections.append("\n### Conventions")
            conv = claude_md.conventions
            if conv.commit_format:
                sections.append(f"- **Commit Format**: {conv.commit_format}")
            if conv.python:
                sections.append(f"- **Python**: {conv.python.min_version}+, "
                               f"line length {conv.python.line_length}")

        # 构建命令
        if claude_md.build_commands:
            sections.append("\n### Build Commands")
            for name, cmd in claude_md.build_commands.items():
                sections.append(f"- `{name}`: {cmd}")

        return "\n".join(sections)
```

#### 9.6.1 动态规则加载

```python
class DynamicRuleLoader:
    """动态规则加载器（支持 CLAUDE.md 引用外部规则文件）"""

    def __init__(self, project_root: Path):
        self.project_root = project_root
        self._rule_cache: Dict[str, RuleContent] = {}

    async def load_rule(self, reference: RuleReference) -> RuleContent:
        """加载规则文件内容"""
        cache_key = reference.file_path

        if cache_key not in self._rule_cache:
            rule_path = self.project_root / reference.file_path
            if rule_path.exists():
                content = rule_path.read_text(encoding="utf-8")
                self._rule_cache[cache_key] = RuleContent(
                    name=reference.name,
                    content=content,
                    version=reference.version,
                )

        return self._rule_cache.get(cache_key)

    async def load_all_rules(
        self,
        references: List[RuleReference]
    ) -> List[RuleContent]:
        """批量加载规则"""
        return [await self.load_rule(ref) for ref in references]
```

#### 9.6.2 项目上下文组装

```python
class ProjectContextBuilder:
    """项目上下文构建器"""

    def __init__(
        self,
        claude_md_loader: ClaudeMdLoader,
        rule_loader: DynamicRuleLoader,
        config: ContextConfig,
    ):
        self.claude_md_loader = claude_md_loader
        self.rule_loader = rule_loader
        self.config = config

    async def build_project_context(
        self,
        project_root: Path,
    ) -> ProjectContext:
        """
        构建完整项目上下文
        1. 加载 CLAUDE.md
        2. 解析规则引用
        3. 加载规则内容
        4. 组装项目上下文
        """
        # 加载 CLAUDE.md
        claude_md = await self.claude_md_loader.load()

        if not claude_md:
            return ProjectContext.empty()

        # 加载规则
        rules_content = []
        if claude_md.rules:
            rules_content = await self.rule_loader.load_all_rules(claude_md.rules)

        # 计算 Token 占用
        token_count = self._estimate_tokens(claude_md, rules_content)

        # 检查预算
        if token_count > self.config.claude_md_max_tokens:
            claude_md, rules_content = await self._compress_project_context(
                claude_md, rules_content, self.config.claude_md_max_tokens
            )

        return ProjectContext(
            claude_md=claude_md,
            rules=rules_content,
            token_count=token_count,
        )

    def _estimate_tokens(
        self,
        claude_md: ClaudeMdSchema,
        rules: List[RuleContent],
    ) -> int:
        """估算 Token 数量"""
        # 使用 tokenizer 估算
        content = json.dumps({
            "project": claude_md.project_overview,
            "rules": [r.content for r in rules],
        })
        return len(content) // 4  # 粗略估算
```

### 9.7 推荐实践

| 场景 | 推荐配置 |
|------|----------|
| 小型项目 (<1000 行代码) | 启用完整 CLAUDE.md，保留所有约定 |
| 中型项目 (1000-10000 行) | 启用 CLAUDE.md + 核心规则文件 |
| 大型项目 (>10000 行) | CLAUDE.md 仅含元数据，规则文件独立管理 |
| 多语言项目 | 每个语言独立 conftest 或规则文件 |

### 9.8 安全考虑

- **规则文件隔离**: 敏感规则（如安全配置）应存储在受限目录
- **路径验证**: 所有引用的规则文件路径必须位于项目根目录下
- **内容审计**: 规则内容应经过静态分析，防止注入攻击

```python
class ProjectFileLoader:
    """项目文件加载器"""

    AI_DIR = ".ai"
    SUPPORTED_SUBDIRS = ["commands", "agents", "skills", "agent-memory", "rules"]

    def __init__(self, project_root: Path):
        self.project_root = project_root
        self.ai_dir = project_root / self.AI_DIR
        self._cache: Dict[str, Any] = {}

    async def load_all(self) -> ProjectFiles:

            commands=await self._load_commands(),

            skills=await self._load_skills(),
            agent_memory=await self._load_agent_memory(),
            rules=await self._load_rules(),

        """加载自定义命令"""
        commands_dir = self.ai_dir / "commands"
        if not commands_dir.exists():
            return []

        commands = []
        for file in commands_dir.glob("*.sh"):
            commands.append(Command(
                name=file.stem,
                script=file.read_text(),
                description=await self._read_description(file),
            ))
        return commands

    async def _load_agents(self) -> List[AgentDefinition]:
        """加载项目级 Agent 定义"""
        agents_dir = self.ai_dir / "agents"
        if not agents_dir.exists():
            return []

        agents = []
        for file in agents_dir.glob("*.md"):
            agents.append(AgentDefinition.from_markdown(file.read_text()))
        return agents

    async def _load_skills(self) -> List[SkillDefinition]:
        """加载项目级 Skills 定义"""
        skills_dir = self.ai_dir / "skills"
        if not skills_dir.exists():
            return []

        skills = []
        for file in skills_dir.glob("*.md"):
            skills.append(SkillDefinition.from_markdown(file.read_text()))
        return skills

    async def _load_agent_memory(self) -> Dict[str, List[MemoryEntry]]:
        """加载项目共享记忆"""
        memory_dir = self.ai_dir / "agent-memory"
        if not memory_dir.exists():
            return {}

        memory = {}
        for agent_dir in memory_dir.iterdir():
            if agent_dir.is_dir():
                memory[agent_dir.name] = await self._load_memory_entries(agent_dir)
        return memory

    async def _load_rules(self) -> List[RuleFile]:
        """加载项目级规则文件"""
        rules_dir = self.ai_dir / "rules"
        if not rules_dir.exists():
            return []

        rules = []
        for file in rules_dir.rglob("*.md"):
            rules.append(RuleFile(
                name=file.stem,
                path=str(file.relative_to(self.project_root)),
                content=file.read_text(),
            ))
        return rules
```

## 11. MCP 配置作用域 (MCPConfigScope)

### 11.1 四级配置作用域

Claude Code 支持 MCP 配置的四级作用域：

| 作用域 | 位置 | 共享范围 | 优先级 |
|--------|------|----------|--------|
| `local` | `.mcp.json`（项目路径下） | 仅当前用户当前项目 | 最高 |
| `project` | `.mcp.json`（Git 仓库根目录） | 团队共享，可提交到 Git | ↑ |
| `user` | `~/.ai.json` | 仅当前用户所有项目 | ↓ |
| `managed` | 系统目录（如 `/etc/ai/mcp.json`） | 管理员部署，不可修改 | 最低 |

### 11.2 MCPConfigScopeManager

```python
class MCPConfigScopeManager:
    """MCP 配置作用域管理器"""

    SCOPE_PRIORITY = ["local", "project", "user", "managed"]

    def __init__(self, project_root: Path, user_home: Path):
        self.project_root = project_root
        self.user_home = user_home
        self._configs: Dict[str, List[MCPConfig]] = {}

    async def load_configs(self) -> List[MCPConfig]:
        """
        加载所有作用域的 MCP 配置
        按优先级排序（local 最高）
        """
        configs = []

        # 1. Managed scope（最低优先级）
        managed_path = Path("/etc/ai/mcp.json")
        if managed_path.exists():
            configs.extend(await self._load_from_path(managed_path, "managed"))

        # 2. User scope
        user_path = self.user_home / ".ai.json"
        if user_path.exists():
            configs.extend(await self._load_from_path(user_path, "user"))

        # 3. Project scope（Git 仓库根目录）
        project_path = await self._find_git_root()
        if project_path and project_path != self.project_root:
            project_mcp = project_path / ".mcp.json"
            if project_mcp.exists():
                configs.extend(await self._load_from_path(project_mcp, "project"))

        # 4. Local scope（最高优先级）
        local_path = self.project_root / ".mcp.json"
        if local_path.exists():
            configs.extend(await self._load_from_path(local_path, "local"))

        return configs

    async def resolve_server(self, server_name: str) -> Optional[MCPConfig]:
        """
        解析 MCP 服务器配置
        按优先级返回第一个匹配的服务器配置
        """
        configs = await self.load_configs()

        # 按优先级排序
        configs.sort(key=lambda c: self.SCOPE_PRIORITY.index(c.scope))

        for config in configs:
            if config.server_name == server_name:
                return config

        return None

    async def get_effective_config(self) -> MCPConfig:
        """
        获取有效配置（合并所有作用域）
        高优先级配置覆盖低优先级
        """
        all_configs = await self.load_configs()
        merged: Dict[str, MCPConfig] = {}

        # 按优先级从低到高处理
        for scope in reversed(self.SCOPE_PRIORITY):
            for config in all_configs:
                if config.scope == scope:
                    merged[config.server_name] = config

        return MCPConfig(servers=merged)
```

## 12. Tool 延迟加载优化 (ToolOptimizer)

### 12.1 背景

Claude Code 支持工具描述延迟加载，当工具描述超过 context 一定比例时启用延迟加载，避免预先塞入 context。

### 12.2 ToolOptimizer 配置

```python
@dataclass
class ToolOptimizerConfig:
    """Tool 优化器配置"""
    enabled: bool = True
    threshold_ratio: float = 0.1  # 工具描述超过 context 10% 时启用延迟加载
    auto_mode_threshold: int = 10  # auto 模式下工具数量阈值
    lazy_load_by_default: bool = False  # 默认是否延迟加载

    # 预加载策略
    preload_strategies: List[str] = field(default_factory=lambda: [
        "essential",    # 预加载必需工具
        "recent",       # 预加载最近使用的工具
        "frequent",     # 预加载高频工具
    ])


class ToolStrategy(Enum):
    """Tool 加载策略"""
    EAGER = "eager"           # 预加载所有工具
    LAZY = "lazy"             # 延迟加载
    AUTO = "auto"             # 自动选择（基于阈值）
    ESSENTIAL_ONLY = "essential"  # 仅预加载必需工具
```

### 12.3 ToolOptimizer 实现

```python
class ToolOptimizer:
    """Tool 优化器"""

    ESSENTIAL_TOOLS = {"Read", "Glob", "Grep"}  # 始终预加载的工具

    def __init__(
        self,
        tools: List[Tool],
        config: ToolOptimizerConfig,
        context_budget: TokenBudget,
    ):
        self.tools = tools
        self.config = config
        self.context_budget = context_budget
        self._usage_stats: Dict[str, ToolUsageStats] = {}
        self._loaded_tools: Set[str] = set()

    async def determine_strategy(self, task: Task) -> ToolStrategy:
        """
        确定 Tool 加载策略
        """
        if not self.config.enabled:
            return ToolStrategy.EAGER

        if self.config.lazy_load_by_default:
            return ToolStrategy.LAZY

        # 计算工具描述总 Token
        tool_descriptions_tokens = sum(
            self._estimate_tokens(t.description) for t in self.tools
        )

        # 计算工具描述占 context 比例
        context_capacity = self.context_budget.get_remaining()
        ratio = tool_descriptions_tokens / context_capacity

        if ratio > self.config.threshold_ratio:
            return ToolStrategy.LAZY

        # 检查工具数量
        if len(self.tools) > self.config.auto_mode_threshold:
            return ToolStrategy.AUTO

        return ToolStrategy.EAGER

    async def get_tools_to_preload(self, strategy: ToolStrategy) -> List[Tool]:
        """
        获取需要预加载的工具
        """
        if strategy == ToolStrategy.EAGER:
            return self.tools

        if strategy == ToolStrategy.LAZY:
            # 延迟加载模式：只返回必需工具
            return [t for t in self.tools if t.name in self.ESSENTIAL_TOOLS]

        if strategy == ToolStrategy.ESSENTIAL_ONLY:
            return [t for t in self.tools if t.name in self.ESSENTIAL_TOOLS]

        if strategy == ToolStrategy.AUTO:
            # AUTO 模式：预加载必需工具 + 最近使用 + 高频工具
            return await self._get_auto_preload_tools()

        return self.tools

    async def _get_auto_preload_tools(self) -> List[Tool]:
        """AUTO 模式：智能选择预加载工具"""
        preload = set(self.ESSENTIAL_TOOLS)

        # 添加最近使用的工具（最近 3 个）
        recent = self._get_recent_tools(3)
        preload.update(recent)

        # 添加高频工具（使用频率最高的 5 个）
        frequent = self._get_frequent_tools(5)
        preload.update(frequent)

        return [t for t in self.tools if t.name in preload]

    def _get_recent_tools(self, limit: int) -> Set[str]:
        """获取最近使用的工具"""
        sorted_tools = sorted(
            self._usage_stats.items(),
            key=lambda x: x[1].last_used,
            reverse=True,
        )
        return {name for name, _ in sorted_tools[:limit]}

    def _get_frequent_tools(self, limit: int) -> Set[str]:
        """获取高频工具"""
        sorted_tools = sorted(
            self._usage_stats.items(),
            key=lambda x: x[1].use_count,
            reverse=True,
        )
        return {name for name, _ in sorted_tools[:limit]}

    async def on_tool_used(self, tool_name: str) -> None:
        """记录工具使用"""
        if tool_name not in self._usage_stats:
            self._usage_stats[tool_name] = ToolUsageStats()
        self._usage_stats[tool_name].increment()

    def _estimate_tokens(self, text: str) -> int:
        """估算 Token 数量"""
        return len(text) // 4  # 粗略估算

    def get_lazy_tools(self) -> List[Tool]:
        """获取延迟加载的工具列表"""
        return [t for t in self.tools if t.name not in self._loaded_tools]

    async def load_tool(self, tool_name: str) -> Optional[Tool]:
        """按需加载单个工具"""
        for tool in self.tools:
            if tool.name == tool_name:
                self._loaded_tools.add(tool_name)
                return tool
        return None


@dataclass
class ToolUsageStats:
    """工具使用统计"""
    use_count: int = 0
    last_used: datetime = field(default_factory=datetime.now)

    def increment(self) -> None:
        self.use_count += 1
        self.last_used = datetime.now()
```

## 13. 动态压缩触发 (DynamicCompressionTrigger)

### 13.1 Claude Code 压缩机制

Claude Code 在 context 达到 95% 容量时自动触发压缩，而非固定阈值。

### 13.2 DynamicCompressionTrigger 实现

```python
@dataclass
class CompressionTriggerConfig:
    """压缩触发配置"""
    trigger_ratio: float = 0.95  # 95% 容量触发
    min_messages_before_compact: int = 10  # 最小消息数
    compact_immediately: bool = False  # 是否立即压缩
    watch_interval: int = 5  # 检查间隔（消息数）


class DynamicCompressionTrigger:
    """动态压缩触发器"""

    def __init__(
        self,
        budget: TokenBudget,
        config: CompressionTriggerConfig,
    ):
        self.budget = budget
        self.config = config
        self._message_count = 0
        self._last_compact_ratio = 0.0

    def should_compact(self, message_count: int, current_ratio: float) -> bool:
        """
        判断是否应该触发压缩
        """
        self._message_count = message_count

        # 消息数不足
        if message_count < self.config.min_messages_before_compact:
            return False

        # 容量未达到阈值
        if current_ratio < self.config.trigger_ratio:
            return False

        # 避免频繁压缩（距离上次压缩容量变化小于 5%）
        if current_ratio - self._last_compact_ratio < 0.05:
            return False

        return True

    def record_compaction(self, ratio_after: float) -> None:
        """记录压缩后的容量比"""
        self._last_compact_ratio = ratio_after

    def get_trigger_info(self) -> CompressionTriggerInfo:
        """获取触发信息"""
        return CompressionTriggerInfo(
            current_ratio=self.budget.get_used_ratio(),
            trigger_threshold=self.config.trigger_ratio,
            message_count=self._message_count,
            should_compact=self.should_compact(
                self._message_count,
                self.budget.get_used_ratio()
            ),
        )


@dataclass
class CompressionTriggerInfo:
    """压缩触发信息"""
    current_ratio: float
    trigger_threshold: float
    message_count: int
    should_compact: bool

    @property
    def remaining_ratio(self) -> float:
        return 1.0 - self.current_ratio

    @property
    def capacity_until_trigger(self) -> float:
        return self.trigger_threshold - self.current_ratio
```
