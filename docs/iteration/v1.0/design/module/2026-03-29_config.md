# Config 模块设计文档

## 1. 模块概述

### 1.1 模块名称

Config（配置管理）

### 1.2 职责

Config 模块是 Mozi 系统的核心基础设施层组件，负责：
- 管理应用配置（JSON 格式）
- 配置加载与校验（JSON Schema 验证）
- 环境变量注入（运行时动态替换）
- 配置热更新（无需重启应用）

### 1.3 核心能力

| 能力 | 说明 |
| ---- | ---- |
| JSON 配置加载 | 从指定路径加载 JSON 配置文件 |
| Schema 校验 | 使用 JSON Schema 验证配置合法性 |
| 环境变量注入 | 支持 `${ENV_VAR}` 语法注入环境变量 |
| 热更新 | 监听文件变化，自动重载配置 |
| 多环境支持 | 支持 development/staging/production 环境配置 |
| 配置合并 | 支持 base + override 配置合并策略 |

### 1.4 核心问题

| 问题 | 说明 |
| ---- | ---- |
| 配置格式不统一 | 不同环境、不同模块使用不同配置格式，难以统一管理 |
| 配置校验缺失 | 缺少配置校验导致运行时错误难以定位 |
| 敏感信息泄露 | 密钥等敏感信息直接写在配置中，存在安全风险 |
| 配置不可热更新 | 修改配置需要重启应用，影响可用性 |

### 1.5 难点

| 难点 | 说明 |
| ---- | ---- |
| Schema 验证的完整性 | 配置结构复杂，Schema 需要覆盖所有边界情况 |
| 环境变量注入的安全性 | 避免注入攻击，确保变量替换后格式正确 |
| 热更新的原子性 | 配置重载过程中避免不一致状态 |
| 多环境配置切换 | 支持 development/staging/production 无缝切换 |

### 1.6 解决方案

| 方案 | 说明 |
| ---- | ---- |
| JSON Schema Draft 7 | 使用标准化的 JSON Schema 进行配置验证，支持嵌套引用和自定义规则 |
| `${ENV_VAR}` 语法 | 统一的环境变量注入语法，支持 `${VAR}` 和 `${VAR:default}` 格式 |
| 文件监听热更新 | 使用 mtime 监测文件变化，验证通过后原子性重载配置 |
| 配置合并策略 | 支持 base + override 深度合并，实现环境特定覆盖 |

### 4.2 环境变量注入

支持 `${ENV_VAR}` 和 `${ENV_VAR:default}` 语法：

```python
import os
import re

ENV_VAR_PATTERN = re.compile(r"\$\{([^}:]+)(?::([^}]*))?\}")


def inject_env_vars(config: dict[str, Any]) -> dict[str, Any]:
    """递归注入环境变量到配置"""
    result = {}

    for key, value in config.items():
        if isinstance(value, str):
            result[key] = _replace_env_var(value)
        elif isinstance(value, dict):
            result[key] = inject_env_vars(value)
        elif isinstance(value, list):
            result[key] = [
                _replace_env_var(item) if isinstance(item, str) else item
                for item in value
            ]
        else:
            result[key] = value

    return result


def _replace_env_var(value: str) -> str:
    """替换单个字符串中的环境变量"""
    def replacer(match: re.Match) -> str:
        env_var = match.group(1)
        default = match.group(2)
        return os.environ.get(env_var, default or match.group(0))

    return ENV_VAR_PATTERN.sub(replacer, value)
```

**示例配置：**

```json
{
    "model": {
        "api_key": "${ANTHROPIC_API_KEY}",
        "api_base": "${ANTHROPIC_API_BASE:https://api.anthropic.com}"
    }
}
```

### 4.3 配置合并策略

支持 base 配置与 override 配置合并：

```python
def merge_configs(base: dict[str, Any], override: dict[str, Any]) -> dict[str, Any]:
    """深度合并配置"""
    result = base.copy()

    for key, value in override.items():
        if key in result and isinstance(result[key], dict) and isinstance(value, dict):
            result[key] = merge_configs(result[key], value)
        else:
            result[key] = value

    return result
```

```
          ▼

    └─────┬──────┘

          ▼

    │ EventBus   │
    └────────────┘
```

### 5.2 热更新接口

```python
import asyncio
from pathlib import Path
from typing import Callable


class ConfigHotReloader:
    """配置热更新器"""

    def __init__(
        self,
        loader: "ConfigLoader",
        poll_interval_seconds: float = 1.0,
    ) -> None:
        self._loader = loader
        self._poll_interval = poll_interval_seconds
        self._last_mtime: float = 0.0
        self._running = False
        self._listeners: list[Callable[["AppConfig"], None]] = []

    def add_listener(self, listener: Callable[["AppConfig"], None]) -> None:
        """添加配置变更监听器"""
        self._listeners.append(listener)

    async def start(self, config_path: str | Path) -> None:
        """启动热更新监视"""
        self._running = True
        path = Path(config_path)

        # 初始化 mtime
        if path.exists():
            self._last_mtime = path.stat().st_mtime

        while self._running:
            await asyncio.sleep(self._poll_interval)
            await self._check_for_changes(path)

    async def stop(self) -> None:
        """停止热更新监视"""
        self._running = False

    async def _check_for_changes(self, path: Path) -> None:
        """检查配置变化"""
        if not path.exists():
            return

        current_mtime = path.stat().st_mtime
        if current_mtime != self._last_mtime:
            self._last_mtime = current_mtime
            await self._reload_config(path)

    async def _reload_config(self, path: Path) -> None:
        """重新加载配置"""
        try:
            new_config = await self._loader.load(str(path))
            for listener in self._listeners:
                listener(new_config)
        except ConfigError as e:
            # 记录错误但不中断运行
            # 使用旧配置继续运行
            pass
```

## 6. 接口设计

### 6.1 核心类型定义

```python
from __future__ import annotations

from abc import ABC, abstractmethod
from dataclasses import dataclass, field
from pathlib import Path
from typing import Any


class ConfigError(Exception):
    """Config 模块基础异常"""

    def __init__(self, code: str, message: str) -> None:
        self.code = code
        super().__init__(message)


class ConfigValidationError(ConfigError):
    """配置校验失败"""
    def __init__(self, message: str, errors: list[str] | None = None) -> None:
        super().__init__("C001", f"Validation failed: {message}")
        self.errors = errors or []


class ConfigNotFoundError(ConfigError):
    """配置文件不存在"""
    def __init__(self, path: str) -> None:
        super().__init__("C002", f"Config file not found: {path}")


class ConfigLoadError(ConfigError):
    """配置加载失败"""
    def __init__(self, path: str, cause: str) -> None:
        super().__init__("C003", f"Failed to load config from {path}: {cause}")


@dataclass
class AppConfig:
    """应用配置"""
    version: str
    environment: str = "development"
    app: dict[str, Any] = field(default_factory=dict)
    model: dict[str, Any] = field(default_factory=dict)
    session: dict[str, Any] = field(default_factory=dict)
    tools: dict[str, Any] = field(default_factory=dict)
    security: dict[str, Any] = field(default_factory=dict)

    def get(self, key: str, default: Any = None) -> Any:
        """获取配置值，支持点号分隔路径"""
        keys = key.split(".")
        value = self.__dict__
        for k in keys:
            if isinstance(value, dict):
                value = value.get(k)
            else:
                return default
            if value is None:
                return default
        return value


class BaseConfigLoader(ABC):
    """配置加载器抽象基类"""

    @abstractmethod
    async def load(self, config_path: str | Path) -> AppConfig:
        """加载配置"""
        raise NotImplementedError

    @abstractmethod
    async def validate(self, config: dict[str, Any]) -> list[str]:
        """校验配置，返回错误列表"""
        raise NotImplementedError
```

### 6.2 ConfigLoader 实现

```python
import json
from pathlib import Path

from jsonschema import validate, ValidationError, Draft7Validator


class ConfigLoader(BaseConfigLoader):
    """配置加载器"""

    def __init__(self, schema: dict[str, Any]) -> None:
        self._schema = schema
        self._validator = Draft7Validator(schema)

    async def load(self, config_path: str | Path) -> AppConfig:
        """加载并校验配置文件"""
        path = Path(config_path)

        if not path.exists():
            raise ConfigNotFoundError(str(path))

        try:
            with open(path, "r", encoding="utf-8") as f:
                raw_config = json.load(f)
        except json.JSONDecodeError as e:
            raise ConfigLoadError(str(path), str(e))

        # 环境变量注入
        config = inject_env_vars(raw_config)

        # Schema 校验
        errors = await self.validate(config)
        if errors:
            raise ConfigValidationError(
                "Config validation failed",
                errors=errors,
            )

        return AppConfig(**config)

    async def validate(self, config: dict[str, Any]) -> list[str]:
        """校验配置，返回错误列表"""
        errors = []
        for error in self._validator.iter_errors(config):
            path = ".".join(str(p) for p in error.path)
            errors.append(f"{path}: {error.message}")
        return errors

    async def load_with_override(
        self,
        base_path: str | Path,
        override_path: str | Path,
    ) -> AppConfig:
        """加载配置并应用 override"""
        base_config = await self.load(base_path)
        override_config = await self.load(override_path)

        merged = merge_configs(
            base=base_config.__dict__,
            override=override_config.__dict__,
        )
        return AppConfig(**merged)
```

### 6.3 配置访问接口

```python
class ConfigAccessor:
    """配置访问器（单例模式）"""

    _instance: ConfigAccessor | None = None

    def __init__(self, config: AppConfig) -> None:
        self._config = config

    @classmethod
    def initialize(cls, config: AppConfig) -> None:
        """初始化配置访问器"""
        cls._instance = cls(config)

    @classmethod
    def get_instance(cls) -> AppConfig:
        """获取配置实例"""
        if cls._instance is None:
            raise ConfigError("C004", "Config not initialized")
        return cls._instance._config

    @classmethod
    def get(cls, key: str, default: Any = None) -> Any:
        """获取配置值"""
        return cls.get_instance().get(key, default)
```
