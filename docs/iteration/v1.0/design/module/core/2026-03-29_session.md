# Session 模块设计文档

## 1. 模块概述

### 1.1 模块名称

Session（会话管理）

### 1.2 职责

Session 模块是 Mozi 系统的核心层组件，负责管理用户与 Agent 之间的会话生命周期，
维护会话元数据，控制上下文窗口，以及提供会话持久化能力。

### 1.3 核心能力

| 能力                   | 说明                                                     |
| ---------------------- | -------------------------------------------------------- |
| 会话生命周期管理       | 创建、获取、更新、删除会话                               |
| 元数据存储             | 会话状态、创建时间、最后活跃时间、用户偏好               |
| 消息存储               | 消息追加、获取、列表管理                                 |
| 会话持久化             | 支持文件存储、跨会话恢复                                 |

### 1.4 核心问题

| 问题 | 说明 |
| ---- | ---- |
| 会话状态管理 | 如何管理会话的创建、更新、删除生命周期 |
| 会话恢复 | 应用重启后如何恢复之前的会话状态 |
| 多会话并发 | 支持多用户多会话并发，彼此隔离 |
| 消息存储效率 | 如何高效存储和检索大量消息 |

### 1.5 难点

| 难点 | 说明 |
| ---- | ---- |
| 会话数据一致性 | 高并发下的会话数据一致性问题 |
| 历史消息清理 | 何时清理、清理哪些历史消息 |
| 存储性能 | 大量会话和消息的存储性能 |

### 1.6 解决方案

| 方案 | 说明 |
| ---- | ---- |
| SessionManager | 会话生命周期管理，CRUD 操作 |
| 消息索引 | 高效的消息存储和检索 |
| 文件持久化 | FileSessionStorage 支持跨会话恢复 |

## 3. 会话模型

### 3.1 核心模型定义

```python
from dataclasses import dataclass, field
from datetime import datetime
from enum import Enum
from typing import Optional
from uuid import UUID, uuid4


class SessionStatus(Enum):
    """会话状态"""
    ACTIVE = "active"       # 活跃会话
    IDLE = "idle"           # 空闲会话（超过阈值未交互）
    ARCHIVED = "archived"   # 已归档会话
    EXPIRED = "expired"     # 已过期会话


class MessageRole(Enum):
    """消息角色"""
    SYSTEM = "system"        # 系统消息
    USER = "user"            # 用户消息
    ASSISTANT = "assistant" # Assistant 消息
    TOOL = "tool"            # 工具调用结果


@dataclass
class Message:
    """消息结构"""
    role: MessageRole
    content: str
    timestamp: datetime = field(default_factory=datetime.now)
    tool_call_id: Optional[str] = None
    metadata: dict = field(default_factory=dict)


@dataclass
class SessionMetadata:
    """会话元数据"""
    created_at: datetime = field(default_factory=datetime.now)
    updated_at: datetime = field(default_factory=datetime.now)
    last_active_at: datetime = field(default_factory=datetime.now)
    message_count: int = 0
    total_tokens: int = 0
    user_id: Optional[str] = None
    tags: list[str] = field(default_factory=list)
    custom_fields: dict = field(default_factory=dict)


@dataclass
class Session:
    """会话结构"""
    id: str
    metadata: SessionMetadata
    messages: list[Message] = field(default_factory=list)
    status: SessionStatus = SessionStatus.ACTIVE
    parent_session_id: Optional[str] = None  # 支持会话分支
```

### 3.2 生命周期状态机

```
                    创建
                      │
                      ▼
    ┌──────────────────────────────────────┐
    │              ACTIVE                   │
    │  (收到消息/发送消息时更新 last_active)│
    └──────────────┬───────────────────────┘
                   │ 超过 idle_timeout 无交互
                   ▼
    ┌──────────────────────────────────────┐
    │               IDLE                   │◄────────┐
    │  (可恢复，保留完整上下文)             │         │
    └──────────────┬───────────────────────┘         │
                   │ 超过 max_lifetime               │ 用户继续交互
                   ▼                                 │
    ┌──────────────────────────────────────┐         │
    │             ARCHIVED                 │─────────┘
    │  (压缩存储，可手动恢复)
    └──────────────┬───────────────────────┘
                   │ 超过 archive_lifetime
                   ▼
    ┌──────────────────────────────────────┐
    │             EXPIRED                  │
    │  (自动清理)
    └──────────────────────────────────────┘
```

### 3.3 元数据 Schema

| 字段             | 类型          | 说明                           |
| ---------------- | ------------- | ------------------------------ |
| `id`             | str           | 会话唯一标识（UUID）           |
| `status`         | SessionStatus | 当前状态                       |
| `metadata`       | SessionMetadata | 会话元数据                   |
| `messages`       | list[Message] | 消息列表                       |

### 4.2 存储接口

```python
from abc import ABC, abstractmethod
from typing import AsyncIterator


class SessionStorage(ABC):
    """会话存储抽象接口"""

    @abstractmethod
    async def save(self, session: Session) -> None:
        """保存会话"""
        ...

    @abstractmethod
    async def load(self, session_id: str) -> Optional[Session]:
        """加载会话"""
        ...

    @abstractmethod
    async def delete(self, session_id: str) -> bool:
        """删除会话"""
        ...

    @abstractmethod
    async def list(
        self,
        status: Optional[SessionStatus] = None,
        limit: int = 100,
    ) -> list[Session]:
        """列出会话"""
        ...

    @abstractmethod
    async def exists(self, session_id: str) -> bool:
        """检查会话是否存在"""
        ...


class FileSessionStorage(SessionStorage):
    """文件存储实现"""

    def __init__(
        self,
        storage_dir: str = "~/.mozi/sessions",
    ) -> None:
        self._storage_dir = Path(storage_dir).expanduser()

    async def save(self, session: Session) -> None:
        """保存会话到文件"""
        session_path = self._get_session_path(session.id)
        session_path.parent.mkdir(parents=True, exist_ok=True)

        data = {
            "version": "1.0",
            "session": self._serialize_session(session),
        }
        await write_json(session_path, data)

    async def load(self, session_id: str) -> Optional[Session]:
        """从文件加载会话"""
        session_path = self._get_session_path(session_id)
        if not session_path.exists():
            return None

        data = await read_json(session_path)
        return self._deserialize_session(data)

    async def delete(self, session_id: str) -> bool:
        """删除会话文件"""
        session_path = self._get_session_path(session_id)
        if session_path.exists():
            session_path.unlink()
            return True
        return False

    async def list(
        self,
        status: Optional[SessionStatus] = None,
        limit: int = 100,
    ) -> list[Session]:
        """列出存储的会话"""
        sessions = []
        for session_file in self._storage_dir.glob("*.json"):
            data = await read_json(session_file)
            session = self._deserialize_session(data)
            if status is None or session.status == status:
                sessions.append(session)

        sessions.sort(
            key=lambda s: s.metadata.last_active_at,
            reverse=True,
        )
        return sessions[:limit]

    async def exists(self, session_id: str) -> bool:
        """检查会话是否存在"""
        return self._get_session_path(session_id).exists()

    def _get_session_path(self, session_id: str) -> Path:
        """获取会话文件路径"""
        return self._storage_dir / f"{session_id}.json"
```

### 4.3 自动保存策略

| 策略       | 触发条件                    | 说明                   |
| ---------- | --------------------------- | ---------------------- |
| 实时保存   | 每次消息追加后              | 最安全，可能影响性能   |
| 定期保存   | 每 N 条消息或 N 秒          | 平衡安全和性能         |
| 延迟保存   | 批量写入，定时刷新          | 最佳性能，可能丢失数据 |
| 退出保存   | 会话结束时                  | 最差安全性             |

默认采用**延迟保存**策略：消息追加后写入缓冲区，每 30 秒或每 10 条消息刷新一次。

## 5. 接口设计

### 6.1 核心类定义

```python
from abc import ABC, abstractmethod
from typing import Optional


class BaseSessionManager(ABC):
    """会话管理器抽象基类"""

    @abstractmethod
    async def create(
        self,
        user_id: Optional[str] = None,
        parent_session_id: Optional[str] = None,
        metadata: Optional[dict] = None,
    ) -> Session:
        """创建新会话"""
        ...

    @abstractmethod
    async def get(self, session_id: str) -> Optional[Session]:
        """获取会话"""
        ...

    @abstractmethod
    async def update(self, session: Session) -> Session:
        """更新会话"""
        ...

    @abstractmethod
    async def delete(self, session_id: str) -> bool:
        """删除会话"""
        ...

    @abstractmethod
    async def list(
        self,
        status: Optional[SessionStatus] = None,
        limit: int = 100,
    ) -> list[Session]:
        """列出会话"""
        ...


class SessionManager(BaseSessionManager):
    """会话管理器实现"""

    def __init__(
        self,
        storage: SessionStorage,
        config: "SessionConfig",
    ) -> None:
        self._storage = storage
        self._config = config
        self._buffer: dict[str, list[Message]] = {}  # 延迟保存缓冲

    async def create(
        self,
        user_id: Optional[str] = None,
        parent_session_id: Optional[str] = None,
        metadata: Optional[dict] = None,
    ) -> Session:
        """创建新会话"""
        session_id = str(uuid4())
        session_metadata = SessionMetadata(
            user_id=user_id,
            custom_fields=metadata or {},
        )
        session = Session(
            id=session_id,
            metadata=session_metadata,
            parent_session_id=parent_session_id,
        )
        await self._storage.save(session)
        return session

    async def get(self, session_id: str) -> Optional[Session]:
        """获取会话"""
        session = await self._storage.load(session_id)
        if session:
            # 更新最后活跃时间
            session.metadata.last_active_at = datetime.now()
        return session

    async def update(self, session: Session) -> Session:
        """更新会话"""
        session.metadata.updated_at = datetime.now()
        await self._storage.save(session)
        return session

    async def delete(self, session_id: str) -> bool:
        """删除会话"""
        return await self._storage.delete(session_id)

    async def list(
        self,
        status: Optional[SessionStatus] = None,
        limit: int = 100,
    ) -> list[Session]:
        """列出会话"""
        return await self._storage.list(status=status, limit=limit)
```

### 7.1 与 Orchestrator 的交互

```
Orchestrator
    │
    ├──► session_manager.create() ──► 创建新会话
    │
    ├──► session_manager.get() ──► 获取会话
    │
    ├──► session_manager.update() ──► 保存会话更新
    │
    └──► Session.append_message() ──► 追加消息
              │
              ▼
         延迟保存缓冲
              │
              ▼
         Storage.save() ──► 持久化到磁盘
```
