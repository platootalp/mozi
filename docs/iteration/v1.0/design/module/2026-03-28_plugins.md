## 1. 模块概述

Plugins 模块是系统的插件层，对标 Claude Code 的插件系统。插件是自包含的功能单元，可打包 Skill、Agent、Hook、MCP Server、LSP Server 五类组件，支持用户和项目级别的扩展。

### 1.1 核心职责

- 插件发现与加载
- 插件生命周期管理（安装/卸载/启用/禁用）
- 五类组件的注册与调度
- 插件市场（Marketplace）支持

### 1.2 组件类型

| 组件类型 | 用途 |
|----------|------|
| **Skill** | 自定义 slash 命令，按需触发 |
| **Agent** | 专用子代理，处理特定任务 |
| **Hook** | 事件钩子，响应 Claude Code 事件 |
| **MCP Server** | Model Context Protocol 连接外部工具 |
| **LSP Server** | Language Server Protocol 提供代码智能 |

### 1.3 差异化特性

| 特性 | 说明 |
|------|------|
| 按需加载 | 懒加载组件，只在触发时加载 |
| Markdown 定义 | Skill 使用 Markdown 定义，易读易写 |
| 多传输支持 | MCP 支持 stdio/sse/http/websocket |
| 热插拔 | 插件动态注册，reload 无需重启 |
| 安装作用域 | 支持 user/project/local/managed 四级作用域 |

## 4. 插件注册表（Plugin Registry）

### 4.1 PluginRegistry

```python
@dataclass
class PluginManifest:
    name: str
    version: str
    description: str
    author: Optional[Author]
    homepage: Optional[str]
    commands: Optional[Path]
    agents: Optional[Path]
    skills: Optional[Path]
    hooks: Optional[Path]
    mcp_servers: Optional[Path]
    lsp_servers: Optional[Path]
    dependencies: Dict[str, str]
    permissions: List[str]
    root: Path

@dataclass
class Plugin:
    manifest: PluginManifest
    enabled: bool
    loaded: bool
    components: Dict[str, Any]

class PluginRegistry:
    """插件注册表"""

    def __init__(self, plugin_dir: Path):
        self.plugin_dir = plugin_dir
        self.plugins: Dict[str, Plugin] = {}
        self._enabled: Dict[str, bool] = {}

    async def discover(self) -> List[Plugin]:
        """发现所有插件"""
        discovered = []
        for path in self.plugin_dir.iterdir():
            if path.is_dir() and (path / "plugin.json").exists():
                try:
                    manifest = await PluginManifest.load(path)
                    plugin = Plugin(manifest=manifest, enabled=True, loaded=False, components={})
                    discovered.append(plugin)
                except Exception as e:
                    logger.warning(f"Failed to load plugin {path}: {e}")
        return discovered

    async def register(self, plugin: Plugin) -> None:
        """注册插件"""
        self.plugins[plugin.manifest.name] = plugin
        logger.info(f"Registered plugin: {plugin.manifest.name}")

    async def unregister(self, name: str) -> None:
        """注销插件"""
        if name in self.plugins:
            await self._unload(name)
            del self.plugins[name]
            logger.info(f"Unregistered plugin: {name}")

    async def enable(self, name: str) -> None:
        """启用插件"""
        if name in self.plugins:
            self.plugins[name].enabled = True
            await self._load(name)

    async def disable(self, name: str) -> None:
        """禁用插件"""
        if name in self.plugins:
            self.plugins[name].enabled = False
            await self._unload(name)

    async def load_all(self) -> None:
        """加载所有已启用插件"""
        for name, plugin in self.plugins.items():
            if plugin.enabled and not plugin.loaded:
                await self._load(name)

    async def reload(self, name: str) -> None:
        """热重载插件"""
        await self._unload(name)
        await self._load(name)

    async def _load(self, name: str) -> None:
        """加载插件组件"""
        plugin = self.plugins[name]
        # 加载 Skills
        if plugin.manifest.skills:
            skills = await self._load_skills(plugin)
            plugin.components["skills"] = skills
        # 加载 Agents
        if plugin.manifest.agents:
            agents = await self._load_agents(plugin)
            plugin.components["agents"] = agents
        # 加载 Hooks
        if plugin.manifest.hooks:
            hooks = await self._load_hooks(plugin)
            plugin.components["hooks"] = hooks
        # 加载 MCP Servers
        if plugin.manifest.mcp_servers:
            mcp = await self._load_mcp(plugin)
            plugin.components["mcp"] = mcp
        # 加载 LSP Servers
        if plugin.manifest.lsp_servers:
            lsp = await self._load_lsp(plugin)
            plugin.components["lsp"] = lsp
        plugin.loaded = True

    async def _unload(self, name: str) -> None:
        """卸载插件组件"""
        plugin = self.plugins[name]
        plugin.components.clear()
        plugin.loaded = False
```

### 5.3 SkillParser

```python
@dataclass
class SkillMetadata:
    name: str
    triggers: List[str]
    description: str
    version: str
    author: str
    tags: List[str]

@dataclass
class Skill:
    metadata: SkillMetadata
    content: str
    steps: List[str]
    parameters: Dict[str, Any]

class SkillParser:
    """Skill 解析器"""

    FRONT_MATTER_PATTERN = re.compile(
        r"^---\n(.*?)\n---\n(.*)",
        re.DOTALL
    )

    def parse(self, content: str) -> Skill:
        """解析 Skill 文档"""
        # 1. 解析 front matter
        match = self.FRONT_MATTER_PATTERN.match(content.strip())
        if not match:
            raise SkillParseError("Invalid skill format: missing front matter")

        front_matter_raw = match.group(1)
        body = match.group(2)

        # 2. 解析元数据
        metadata = self._parse_front_matter(front_matter_raw)

        # 3. 解析执行步骤
        steps = self._parse_steps(body)

        # 4. 解析参数
        parameters = self._parse_parameters(body)

        return Skill(
            metadata=metadata,
            content=content,
            steps=steps,
            parameters=parameters,
        )

    def _parse_front_matter(self, raw: str) -> SkillMetadata:
        """解析 YAML front matter"""
        data = yaml.safe_load(raw)
        return SkillMetadata(
            name=data["name"],
            triggers=data.get("triggers", []),
            description=data.get("description", ""),
            version=data.get("version", "1.0.0"),
            author=data.get("author", ""),
            tags=data.get("tags", []),
        )

    def _parse_steps(self, body: str) -> List[str]:
        """解析执行步骤"""
        steps = []
        for line in body.split("\n"):
            line = line.strip()
            if line.startswith("## 执行步骤") or line.startswith("## Steps"):
                continue
            if re.match(r"^\d+\.", line):
                steps.append(re.sub(r"^\d+\.\s*", "", line))
        return steps

    def _parse_parameters(self, body: str) -> Dict[str, Any]:
        """解析参数定义"""
        params = {}
        in_params = False
        for line in body.split("\n"):
            line = line.strip()
            if line.startswith("## 参数") or line.startswith("## Parameters"):
                in_params = True
                continue
            if in_params and line.startswith("-"):
                match = re.match(r"- `(\w+)`: (.+)", line)
                if match:
                    params[match.group(1)] = match.group(2)
        return params
```

### 5.4 SkillMatcher

```python
class SkillMatcher:
    """Skill 触发器匹配器"""

    def __init__(self, skills: List[Skill]):
        self.skills = {s.metadata.name: s for s in skills}
        self._build_index()

    def _build_index(self) -> None:
        """构建触发器索引"""
        self.trigger_index: Dict[str, List[str]] = {}  # trigger -> skill names

        for skill in self.skills.values():
            for trigger in skill.metadata.triggers:
                if trigger not in self.trigger_index:
                    self.trigger_index[trigger] = []
                self.trigger_index[trigger].append(skill.metadata.name)

    def match(self, user_input: str) -> Optional[Skill]:
        """匹配用户输入到 Skill"""
        user_input_lower = user_input.lower()

        # 精确匹配
        for trigger, skill_names in self.trigger_index.items():
            if trigger.lower() in user_input_lower:
                return self.skills[skill_names[0]]

        # 模糊匹配
        for skill_name, skill in self.skills.items():
            if skill_name.lower().replace("-", " ") in user_input_lower:
                return skill

        return None

    def get_skill(self, name: str) -> Optional[Skill]:
        """按名称获取 Skill"""
        return self.skills.get(name)
```

### 5.5 SkillExecutor

```python
class SkillExecutor:
    """Skill 执行器"""

    def __init__(
        self,
        tool_executor: ToolExecutor,
        model: ModelGateway,
    ):
        self.tool_executor = tool_executor
        self.model = model

    async def execute(
        self,
        skill: Skill,
        params: Dict[str, Any],
        context: ToolContext,
    ) -> SkillResult:
        """执行 Skill"""
        results = []

        for step in skill.steps:
            # 1. 渲染步骤模板
            rendered = self._render_step(step, params, context)

            # 2. 判断步骤类型
            if rendered.startswith("bash:"):
                command = rendered[5:].strip()
                result = await self.tool_executor.execute(
                    "bash",
                    {"command": command},
                    context,
                )
            elif rendered.startswith("llm:"):
                prompt = rendered[4:].strip()
                response = await self.model.complete([
                    Message(role="user", content=prompt)
                ])
                result = ToolResult(success=True, content=response.content)
            else:
                result = await self.tool_executor.execute(
                    "bash",
                    {"command": rendered},
                    context,
                )

            results.append(SkillStepResult(step=step, result=result))

            # 3. 如果步骤失败，停止执行
            if not result.success:
                break

        return SkillResult(
            skill_name=skill.metadata.name,
            success=all(r.result.success for r in results),
            step_results=results,
        )

    def _render_step(
        self,
        step: str,
        params: Dict[str, Any],
        context: ToolContext,
    ) -> str:
        """渲染步骤模板"""
        # 替换参数
        for key, value in params.items():
            step = step.replace(f"{{{key}}}", str(value))

        return step

@dataclass
class SkillStepResult:
    step: str
    result: ToolResult

@dataclass
class SkillResult:
    skill_name: str
    success: bool
    step_results: List[SkillStepResult]
```

### 5.6 SkillsEngine

```python
class SkillsEngine:
    """Skills 引擎"""

    def __init__(
        self,
        skill_dir: Path,
        tool_executor: ToolExecutor,
        model: ModelGateway,
    ):
        self.skill_dir = skill_dir
        self.tool_executor = tool_executor
        self.model = model

        self.parser = SkillParser()
        self.executor = SkillExecutor(tool_executor, model)
        self.matcher: Optional[SkillMatcher] = None

    async def load(self) -> None:
        """加载所有 Skill"""
        skills = []

        for path in self.skill_dir.glob("**/*.md"):
            try:
                content = path.read_text()
                skill = self.parser.parse(content)
                skills.append(skill)
            except Exception as e:
                logger.warning(f"Failed to load skill {path}: {e}")

        self.matcher = SkillMatcher(skills)
        logger.info(f"Loaded {len(skills)} skills")

    async def execute_from_input(
        self,
        user_input: str,
        context: ToolContext,
        params: Optional[Dict[str, Any]] = None,
    ) -> Optional[SkillResult]:
        """从用户输入执行 Skill"""
        if not self.matcher:
            await self.load()

        # 1. 匹配 Skill
        skill = self.matcher.match(user_input)
        if not skill:
            return None

        # 2. 执行 Skill
        return await self.executor.execute(
            skill,
            params or {},
            context,
        )
```

## 7. MCP Server Approval Workflow

### 7.1 Overview

MCP Server 审批工作流确保只有经过授权的 MCP 服务器才能与 Mozi 交互。这是安全的关键层，防止恶意或未验证的 MCP 服务器执行敏感操作。

Mozi 的 MCP 集成采用 Model Context Protocol (MCP)，支持多种传输类型：
- **stdio**: 本地进程通信
- **HTTP**: 远程服务调用
- **SSE**: Server-Sent Events 实时推送
- **WebSocket**: 双向实时通信

### 7.2 Approval States

MCP 服务器审批有三种状态：

| 状态 | 说明 | 行为 |
|------|------|------|
| **PENDING** | 待审批 | 服务器已注册但未授权，需要用户或管理员批准 |
| **APPROVED** | 已批准 | 服务器已授权，可以正常连接和使用 |
| **REJECTED** | 已拒绝 | 服务器被拒绝，无法连接或使用 |

#### 7.2.1 State Transitions

```
                    ┌─────────────┐
                    │   PENDING   │
                    └──────┬──────┘
                           │
            ┌──────────────┴──────────────┐
            │                             │
            ▼                             ▼
    ┌───────────────┐             ┌───────────────┐
    │   APPROVED    │             │   REJECTED    │
    └───────────────┘             └───────────────┘
            │                             │
            └──────────────┬──────────────┘
                           │
                           ▼
                    (可重新申请)
```

### 7.3 Approval Configuration

审批配置支持基于服务器名称和 URL 模式的匹配规则：

```json
{
  "allowedMcpServers": [
    { "serverName": "github" },
    { "serverName": "filesystem" },
    { "serverUrl": "https://mcp.company.com/*" }
  ],
  "deniedMcpServers": [
    { "serverName": "untrusted-tool" },
    { "serverUrl": "https://evil-*.com/*" }
  ]
}
```

#### 7.3.1 Matching Rules

| 规则类型 | 匹配方式 | 示例 |
|----------|----------|------|
| serverName | 精确匹配 | `"github"` 匹配 `github` |
| serverUrl | Glob 模式 | `"https://mcp.company.com/*"` 匹配 `https://mcp.company.com/server1` |

#### 7.3.2 Decision Flow

```
MCP Server 连接请求
    │
    ▼
检查 deniedMcpServers（优先匹配）
    ├── 命中 → REJECTED（直接拒绝）
    └── 未命中 → 继续检查
    │
    ▼
检查 allowedMcpServers
    ├── 命中 → APPROVED（允许连接）
    └── 未命中 → PENDING（请求审批）
    │
    ▼
用户审批 / 自动过期
```

### 7.4 MCP Tools Integration with Mozi Permission System

MCP 工具通过 Mozi 的权限系统集成，确保工具调用符合 Mozi 的安全策略。

#### 7.4.1 Permission Categories

| 权限类别 | 说明 | 示例 |
|----------|------|------|
| `tool:*` | 工具调用权限 | `tool:bash`, `tool:edit` |
| `file:read` | 文件读取权限 | 读取项目文件 |
| `file:write` | 文件写入权限 | 修改或创建文件 |
| `network:http` | HTTP 请求权限 | 调用外部 API |
| `network:websocket` | WebSocket 权限 | 双向实时通信 |

#### 7.4.2 MCP Tool Permission Mapping

MCP 工具在注册时需要声明所需权限：

```python
@dataclass
class MCPToolPermission:
    """MCP 工具权限声明"""
    tool_name: str
    required_permissions: List[str]
    risk_level: RiskLevel

class RiskLevel(Enum):
    LOW = "low"       # 文件读取、只读操作
    MEDIUM = "medium" # 文件写入、网络请求
    HIGH = "high"     # 命令执行、数据删除

class MCPToolPermissionChecker:
    """MCP 工具权限检查器"""

    def __init__(self, permission_manager: PermissionManager):
        self.permission_manager = permission_manager

    async def check_tool_permissions(
        self,
        tool: MCPTool,
        context: ToolContext,
    ) -> PermissionCheckResult:
        """检查 MCP 工具权限"""
        # 1. 获取工具声明的权限
        declared_permissions = self._get_declared_permissions(tool)

        # 2. 检查用户是否有这些权限
        for permission in declared_permissions:
            if not await self.permission_manager.check(permission, context):
                return PermissionCheckResult(
                    allowed=False,
                    reason=f"Missing permission: {permission}",
                )

        # 3. 检查风险级别
        risk_level = self._assess_risk_level(tool)
        if risk_level == RiskLevel.HIGH:
            # 高风险操作需要额外确认
            if not await self._require_explicit_approval(context):
                return PermissionCheckResult(
                    allowed=False,
                    reason="High-risk operation requires explicit approval",
                )

        return PermissionCheckResult(allowed=True)

    def _get_declared_permissions(self, tool: MCPTool) -> List[str]:
        """获取工具声明的权限"""
        # MCP 工具应在 manifest 中声明权限
        return tool.metadata.permissions if hasattr(tool.metadata, "permissions") else []
```

```python
@dataclass
class Hook:
    name: str
    description: str
    trigger: str
    action: str
    enabled: bool = True

class HookManager:
    """Hook 管理器"""

    TRIGGERS = [
        "session_start",
        "session_end",
        "before_tool_use",
        "after_tool_use",
        "before_message",
        "after_message",
        "on_error",
    ]

    def __init__(self):
        self.hooks: Dict[str, List[Hook]] = {t: [] for t in self.TRIGGERS}

    async def register(self, hook: Hook) -> None:
        """注册 Hook"""
        if hook.trigger not in self.TRIGGERS:
            raise ValueError(f"Unknown trigger: {hook.trigger}")
        self.hooks[hook.trigger].append(hook)

    async def trigger(self, trigger: str, context: Dict[str, Any]) -> None:
        """触发 Hook"""
        for hook in self.hooks.get(trigger, []):
            if hook.enabled:
                await self._execute(hook, context)

    async def _execute(self, hook: Hook, context: Dict[str, Any]) -> None:
        """执行 Hook"""
        env = {
            "MOZI_HOOK_NAME": hook.name,
            "MOZI_HOOK_TRIGGER": hook.trigger,
            **os.environ,
        }
        result = await asyncio.create_subprocess_exec(
            hook.action,
            env=env,
            stdout=asyncio.subprocess.PIPE,
            stderr=asyncio.subprocess.PIPE,
        )
        await result.wait()
```

### 9.2 MCP 服务器审批工作流

MCP 服务器需要经过审批才能使用，确保安全可控。

#### 9.2.1 审批配置结构

```json
{
  "allowedMcpServers": [
    { "serverName": "github" },
    { "serverUrl": "https://mcp.company.com/*" }
  ],
  "deniedMcpServers": [
    { "serverName": "untrusted-tool" }
  ]
}
```

#### 9.2.2 审批规则

| 规则类型 | 匹配方式 | 说明 |
|----------|----------|------|
| allowedMcpServers | serverName / serverUrl | 允许的 MCP 服务器 |
| deniedMcpServers | serverName / serverUrl | 拒绝的 MCP 服务器 |

#### 9.2.3 审批决策流程

```
MCP Server 连接请求
    │
    ▼
检查 deniedMcpServers（优先匹配）
    ├── 命中 → 直接拒绝
    └── 未命中 → 继续检查
    │
    ▼
检查 allowedMcpServers
    ├── 命中 → 允许连接
    └── 未命中 → 拒绝连接（默认拒绝）
```

#### 9.2.4 MCPApprovalManager

```python
@dataclass
class ApprovalRule:
    """审批规则"""
    server_name: Optional[str] = None
    server_url: Optional[str] = None

    def matches(self, server_name: str, server_url: Optional[str] = None) -> bool:
        """检查是否匹配"""
        if self.server_name and self.server_name == server_name:
            return True
        if self.server_url and server_url:
            import fnmatch
            return fnmatch.fnmatch(server_url, self.server_url)
        return False

@dataclass
class MCPApprovalConfig:
    """MCP 审批配置"""
    allowed: List[ApprovalRule] = field(default_factory=list)
    denied: List[ApprovalRule] = field(default_factory=list)

class MCPApprovalManager:
    """MCP 服务器审批管理器"""

    def __init__(self, config: MCPApprovalConfig):
        self.config = config

    def is_approved(self, server_name: str, server_url: Optional[str] = None) -> bool:
        """检查 MCP 服务器是否已审批"""
        # 1. 检查拒绝列表（优先）
        for rule in self.config.denied:
            if rule.matches(server_name, server_url):
                logger.warning(f"MCP server {server_name} is denied")
                return False

        # 2. 检查允许列表
        for rule in self.config.allowed:
            if rule.matches(server_name, server_url):
                logger.info(f"MCP server {server_name} is allowed")
                return True

        # 3. 默认拒绝
        logger.warning(f"MCP server {server_name} not in allowlist, denying by default")
        return False

    async def request_approval(
        self,
        server_name: str,
        server_url: Optional[str] = None,
        metadata: Optional[Dict[str, Any]] = None,
    ) -> ApprovalRequest:
        """请求审批"""
        if self.is_approved(server_name, server_url):
            return ApprovalRequest(
                server_name=server_name,
                server_url=server_url,
                status=ApprovalStatus.APPROVED,
            )

        # 创建待审批请求
        return ApprovalRequest(
            server_name=server_name,
            server_url=server_url,
            status=ApprovalStatus.PENDING,
            metadata=metadata or {},
        )

    def load_config(self, config_path: Path) -> MCPApprovalConfig:
        """从文件加载审批配置"""
        with open(config_path) as f:
            data = json.load(f)

        allowed = []
        for item in data.get("allowedMcpServers", []):
            allowed.append(ApprovalRule(
                server_name=item.get("serverName"),
                server_url=item.get("serverUrl"),
            ))

        denied = []
        for item in data.get("deniedMcpServers", []):
            denied.append(ApprovalRule(
                server_name=item.get("serverName"),
                server_url=item.get("serverUrl"),
            ))

        return MCPApprovalConfig(allowed=allowed, denied=denied)

    def save_config(self, config: MCPApprovalConfig, config_path: Path) -> None:
        """保存审批配置到文件"""
        data = {
            "allowedMcpServers": [
                {"serverName": r.server_name, "serverUrl": r.server_url}
                for r in config.allowed if r.server_name
            ],
            "deniedMcpServers": [
                {"serverName": r.server_name, "serverUrl": r.server_url}
                for r in config.denied if r.server_name
            ],
        }
        with open(config_path, "w") as f:
            json.dump(data, f, indent=2)

class ApprovalStatus(Enum):
    """MCP 服务器审批状态"""
    APPROVED = "approved"    # 已批准
    REJECTED = "rejected"   # 已拒绝
    PENDING = "pending"     # 待审批

@dataclass
class ApprovalRequest:
    server_name: str
    server_url: Optional[str]
    status: ApprovalStatus
    metadata: Dict[str, Any] = field(default_factory=dict)
```

### 9.5 MCPClient

```python
class MCPClient:
    """MCP 客户端"""

    def __init__(self, config: MCPConfig, approval_manager: Optional[MCPApprovalManager] = None):
        self.config = config
        self.approval_manager = approval_manager or MCPApprovalManager(MCPApprovalConfig())
        self.transport: Optional[Transport] = None
        self.tools: Dict[str, MCPTool] = {}
        self._connected = False

    async def connect(self) -> None:
        """连接 MCP Server"""
        # 1. 审批检查
        if not self.approval_manager.is_approved(self.config.name, self.config.url):
            raise MCPApprovalError(
                f"MCP server {self.config.name} is not approved. "
                f"Add it to allowedMcpServers in your MCP config."
            )

        # 2. 创建传输层
        if self.config.transport == "stdio":
            self.transport = StdioTransport(self.config.command, self.config.args)
        elif self.config.transport == "sse":
            self.transport = SSETransport(self.config.url)
        elif self.config.transport == "http":
            self.transport = HTTPTransport(self.config.url)
        elif self.config.transport == "websocket":
            self.transport = WebSocketTransport(self.config.url)

        await self.transport.connect()

        # 3. 初始化
        await self._initialize()

        # 4. 获取工具列表
        await self._list_tools()

        self._connected = True

    async def disconnect(self) -> None:
        """断开连接"""
        if self.transport:
            await self.transport.close()
            self._connected = False

    async def _initialize(self) -> None:
        """发送初始化请求"""
        request = MCPProtocol.MCPMessage(
            id=str(uuid.uuid4()),
            method=MCPProtocol.MessageType.INITIALIZE.value,
            params={
                "protocolVersion": MCPProtocol.VERSION,
                "capabilities": {
                    "tools": True,
                    "resources": True,
                    "prompts": True,
                },
                "clientInfo": {
                    "name": "mozi",
                    "version": "1.0.0",
                },
            },
        )

        response = await self.transport.send(request)
        if response.error:
            raise MCPError(f"Initialize failed: {response.error}")

    async def _list_tools(self) -> None:
        """获取工具列表"""
        request = MCPProtocol.MCPMessage(
            id=str(uuid.uuid4()),
            method=MCPProtocol.MessageType.TOOLS_LIST.value,
        )

        response = await self.transport.send(request)
        if response.result and "tools" in response.result:
            for tool_def in response.result["tools"]:
                self.tools[tool_def["name"]] = MCPTool(
                    name=tool_def["name"],
                    description=tool_def.get("description", ""),
                    input_schema=tool_def.get("inputSchema", {}),
                )

    async def call_tool(
        self,
        tool_name: str,
        params: Dict[str, Any],
    ) -> ToolResult:
        """调用工具"""
        if tool_name not in self.tools:
            return ToolResult(success=False, error=f"Tool {tool_name} not found")

        request = MCPProtocol.MCPMessage(
            id=str(uuid.uuid4()),
            method=MCPProtocol.MessageType.TOOLS_CALL.value,
            params={
                "name": tool_name,
                "arguments": params,
            },
        )

        response = await self.transport.send(request)

        if response.error:
            return ToolResult(success=False, error=str(response.error))

        return ToolResult(
            success=True,
            content=str(response.result),
        )

    def list_tools(self) -> List[MCPTool]:
        """列出所有工具"""
        return list(self.tools.values())

@dataclass
class MCPTool:
    name: str
    description: str
    input_schema: Dict[str, Any]

    def to_tool(self) -> Tool:
        """转换为 Mozi Tool"""
```

#### 9.6.3 StdioTransport

```python
class StdioTransport(Transport):
    """STDIO 传输 - 用于本地进程"""

    def __init__(self, command: str, args: List[str]):
        self.command = command
        self.args = args
        self.process: Optional[asyncio.subprocess.Process] = None
        self._reader: Optional[asyncio.StreamReader] = None
        self._writer: Optional[asyncio.StreamWriter] = None

    async def connect(self) -> None:
        self.process = await asyncio.create_subprocess_exec(
            self.command,
            *self.args,
            stdin=asyncio.subprocess.PIPE,
            stdout=asyncio.subprocess.PIPE,
        )
        self._reader = self.process.stdout
        self._writer = self.process.stdin

    async def send(self, message: MCPProtocol.MCPMessage) -> MCPProtocol.MCPMessage:
        data = json.dumps(message.__dict__).encode() + b"\n"
        self._writer.write(data)
        await self._writer.drain()

        response_data = await self._reader.readline()
        return MCPProtocol.MCPMessage(**json.loads(response_data))

    async def close(self) -> None:
        if self.process:
            self.process.terminate()
            await self.process.wait()
```

```python
class HTTPTransport(Transport):
    """HTTP 传输 - 用于远程服务"""

    def __init__(self, url: str, headers: Optional[Dict[str, str]] = None):
        self.url = url
        self.headers = headers or {}
        self._session: Optional[aiohttp.ClientSession] = None

        self._session = aiohttp.ClientSession(headers=self.headers)

    async def send(self, message: MCPProtocol.MCPMessage) -> MCPProtocol.MCPMessage:
        if not self._session:
            raise MCPError("Not connected")

        async with self._session.post(
            self.url,
            json=message.__dict__,
        ) as response:

            return MCPProtocol.MCPMessage(**data)

    async def close(self) -> None:

            await self._session.close()
```

#### 9.6.5 SSETransport

```python
class SSETransport(Transport):
    """SSE 传输 - 用于 Server-Sent Events 实时推送"""

    def __init__(self, url: str, headers: Optional[Dict[str, str]] = None):
        self.url = url
        self.headers = headers or {}
        self._session: Optional[aiohttp.ClientSession] = None
        self._response: Optional[aiohttp.ClientResponse] = None
        self._queue: asyncio.Queue = asyncio.Queue()

    async def connect(self) -> None:
        self._session = aiohttp.ClientSession(headers=self.headers)
        self._response = await self._session.get(self.url)

        # 启动事件监听任务
        asyncio.create_task(self._listen_events())

    async def _listen_events(self) -> None:
        """监听 SSE 事件"""
        async for line in self._response.content:
            line = line.decode().strip()
            if line.startswith("data: "):
                data = json.loads(line[6:])
                await self._queue.put(data)

    async def send(self, message: MCPProtocol.MCPMessage) -> MCPProtocol.MCPMessage:
        # SSE 是单向的，这里通过 HTTP POST 发送请求
        async with self._session.post(self.url + "/send", json=message.__dict__) as response:
            data = await response.json()
            return MCPProtocol.MCPMessage(**data)

    async def receive(self) -> MCPProtocol.MCPMessage:
        """接收事件"""
        data = await self._queue.get()
        return MCPProtocol.MCPMessage(**data)

    async def close(self) -> None:
        if self._response:
            await self._response.release()
        if self._session:
            await self._session.close()
```

#### 9.6.6 WebSocketTransport

```python
class WebSocketTransport(Transport):
    """WebSocket 传输 - 用于双向实时通信"""

    def __init__(self, url: str, headers: Optional[Dict[str, str]] = None):
        self.url = url
        self.headers = headers or {}
        self._ws: Optional[aiohttp.ClientWebSocketResponse] = None
        self._session: Optional[aiohttp.ClientSession] = None
        self._pending: Dict[str, asyncio.Future] = {}

    async def connect(self) -> None:
        self._session = aiohttp.ClientSession(headers=self.headers)
        self._ws = await self._session.ws_connect(self.url)

        # 启动消息监听任务
        asyncio.create_task(self._listen_messages())

    async def _listen_messages(self) -> None:
        """监听 WebSocket 消息"""
        async for msg in self._ws:
            if msg.type == aiohttp.WSMsgType.TEXT:
                data = json.loads(msg.data)
                msg_id = data.get("id")
                if msg_id in self._pending:
                    self._pending[msg_id].set_result(data)
                else:
                    # 处理通知消息
                    await self._handle_notification(data)
            elif msg.type == aiohttp.WSMsgType.ERROR:
                logger.error(f"WebSocket error: {msg.data}")

    async def send(self, message: MCPProtocol.MCPMessage) -> MCPProtocol.MCPMessage:
        if not self._ws:
            raise MCPError("Not connected")

        future = asyncio.Future()
        if message.id:
            self._pending[message.id] = future

        await self._ws.send_json(message.__dict__)

        return await future

    async def _handle_notification(self, data: Dict[str, Any]) -> None:
        """处理通知消息"""
        pass

    async def close(self) -> None:
```

## 10. MCP 工具搜索优化

### 9.1 概述

MCP 工具搜索优化通过延迟加载和智能缓存提升性能，避免在启动时加载所有工具描述。

### 9.2 搜索模式

| 模式 | 行为 | 环境变量 |
|------|------|----------|
| lazy | 仅加载工具名称 | `ENABLE_TOOL_SEARCH=lazy` |
| auto | 描述超上下文阈值时懒加载 | `ENABLE_TOOL_SEARCH=auto:10` |
| eager | 启动时加载所有工具 | `ENABLE_TOOL_SEARCH=eager` |

### 9.3 延迟加载策略

```bash
# 启动时只加载工具名称（默认）
ai

# 当描述超过 10% 上下文时自动懒加载
ENABLE_TOOL_SEARCH=auto:10 ai

# 强制立即加载所有工具描述
ENABLE_TOOL_SEARCH=eager ai
```

### 9.4 ToolSearchOptimizer 实现

```python
class ToolSearchOptimizer:
    """工具搜索优化器"""

    # 默认配置
    DEFAULT_LAZY_THRESHOLD = 10  # 描述超过上下文 10% 时触发懒加载

    def __init__(
        self,
        mode: str = "lazy",
        lazy_threshold: int = DEFAULT_LAZY_THRESHOLD,
    ):
        self.mode = mode
        self.lazy_threshold = lazy_threshold
        self._tool_names_loaded = False
        self._full_descriptions: Dict[str, ToolDescription] = {}

    async def load_tool_names(self, mcp_client: MCPClient) -> List[str]:
        """加载工具名称（快速）"""
        if self._tool_names_loaded:
            return list(mcp_client.tools.keys())

        tools = mcp_client.list_tools()
        self._tool_names_loaded = True
        return [t.name for t in tools]

    async def load_full_description(
        self,
        mcp_client: MCPClient,
        tool_name: str,
    ) -> Optional[ToolDescription]:
        """加载完整工具描述（按需）"""
        # 检查缓存
        if tool_name in self._full_descriptions:
            return self._full_descriptions[tool_name]

        # 获取工具定义
        tool = mcp_client.tools.get(tool_name)
        if not tool:
            return None

        # 缓存描述
        description = ToolDescription(
            name=tool.name,
            description=tool.description,
            input_schema=tool.input_schema,
        )
        self._full_descriptions[tool_name] = description
        return description

    def should_load_full(self, context_size: int, descriptions_size: int) -> bool:
        """判断是否应加载完整描述"""
        if self.mode == "eager":
            return True
        if self.mode == "lazy":
            return False
        if self.mode == "auto":
            ratio = descriptions_size / max(context_size, 1)
            return ratio > (self.lazy_threshold / 100)
        return False

    async def search_tools(
        self,
        mcp_client: MCPClient,
        query: str,
        context_size: int,
    ) -> SearchResult:
        """搜索工具"""
        # 1. 获取工具名称列表
        tool_names = await self.load_tool_names(mcp_client)

        # 2. 过滤匹配的工具
        matched_names = [
            name for name in tool_names
            if query.lower() in name.lower()
        ]

        # 3. 检查是否需要加载完整描述
        estimated_desc_size = len(matched_names) * 200  # 估算
        should_load = self.should_load_full(context_size, estimated_desc_size)

        if should_load:
            # 加载完整描述
            results = []
            for name in matched_names:
                desc = await self.load_full_description(mcp_client, name)
                if desc:
                    results.append(desc)
            return SearchResult(tools=results, fully_loaded=True)
        else:
            # 只返回名称
            return SearchResult(
                tools=[ToolNameOnly(name=n) for n in matched_names],
                fully_loaded=False,
            )

@dataclass
class ToolDescription:
    name: str
    description: str
    input_schema: Dict[str, Any]

@dataclass
class ToolNameOnly:
    name: str

@dataclass
class SearchResult:
    tools: List[Any]
    fully_loaded: bool
```

```python
@dataclass
class LSPConfig:
    name: str
    command: str
    args: List[str]
    languages: List[str]
    root_patterns: List[str]

class LSPClient:
    """LSP 客户端"""

    def __init__(self, config: LSPConfig):
        self.config = config
        self.process: Optional[asyncio.subprocess.Process] = None
        self._reader: Optional[asyncio.StreamReader] = None
        self._writer: Optional[asyncio.StreamWriter] = None
        self._request_id = 0
        self._callbacks: Dict[str, asyncio.Future] = {}

    async def connect(self) -> None:
        """连接 LSP Server"""
        self.process = await asyncio.create_subprocess_exec(
            self.config.command,
            *self.config.args,
            stdin=asyncio.subprocess.PIPE,
            stdout=asyncio.subprocess.PIPE,
        )
        self._reader = self.process.stdout
        self._writer = self.process.stdin

        # 发送初始化
        await self._send_initialize()

    async def _send_initialize(self) -> None:
        """发送初始化请求"""
        response = await self._send_request(
            "initialize",
            {
                "processId": os.getpid(),
                "rootUri": str(Path.cwd()),
                "capabilities": {},
            },
        )
        # 发送 initialized
        await self._send_notification("initialized", {})

    async def _send_request(self, method: str, params: Dict[str, Any]) -> Dict[str, Any]:
        """发送请求"""
        self._request_id += 1
        request_id = str(self._request_id)

        message = {
            "jsonrpc": "2.0",
            "id": request_id,
            "method": method,
            "params": params,
        }

        future = asyncio.Future()
        self._callbacks[request_id] = future

        data = json.dumps(message).encode() + b"\n"
        self._writer.write(data)
        await self._writer.drain()

        return await future

    async def _send_notification(self, method: str, params: Dict[str, Any]) -> None:
        """发送通知"""
        message = {
            "jsonrpc": "2.0",
            "method": method,
            "params": params,
        }
        data = json.dumps(message).encode() + b"\n"
        self._writer.write(data)
        await self._writer.drain()

    async def diagnostics(self, uri: str) -> List[Diagnostic]:
        """获取诊断信息"""
        response = await self._send_request("textDocument/diagnostic", {
            "textDocument": {"uri": uri}
        })
        return [Diagnostic(**d) for d in response.get("items", [])]

    async def close(self) -> None:
        """关闭连接"""
        await self._send_notification("shutdown", None)
        if self.process:
            self.process.terminate()
            await self.process.wait()
```

```python
class MarketplacePlugin:
    name: str
    version: str
    description: str
    source: str
    tags: List[str]
    license: str

@dataclass
class Marketplace:
    name: str
    owner: str
    plugins: List[MarketplacePlugin]

class MarketplaceResolver:
    """市场解析器"""

    async def resolve(self, source: str) -> Marketplace:
        """解析市场清单"""
        if source.startswith("github:"):
            return await self._resolve_github(source)
        elif source.startswith("npm:"):
            return await self._resolve_npm(source)
        else:
            return await self._resolve_local(source)

    async def _resolve_github(self, source: str) -> Marketplace:
        """从 GitHub 解析"""
        # 解析 github:owner/repo
        match = re.match(r"github:([^/]+)/([^#]+)(#(.+))?", source)
        if not match:
            raise ValueError(f"Invalid GitHub source: {source}")

        owner, repo, _, branch = match.groups()
        branch = branch or "main"

        # 克隆仓库
        clone_dir = self._cache_dir / f"github-{owner}-{repo}"
        if not clone_dir.exists():
            await self._clone_github(owner, repo, clone_dir, branch)

        # 读取 marketplace.json
        marketplace_path = clone_dir / "marketplace.json"
        return await self._parse_marketplace(marketplace_path)

```

```
    │
    ▼

    ▼
```
