## 1. 模块概述

Storage 模块是系统的存储层，负责所有数据的持久化。包括 Session 会话持久化、Config 配置存储、Artifact 工件存储。

### 1.1 核心职责

- Session 持久化
- Config 配置管理
- Artifact 大文件存储
- 数据库迁移

### 1.2 差异化特性

| 特性 | 说明 |
|------|------|
| SQLite 本地存储 | 数据不离开本地环境 |
| 异步操作 | 所有操作异步执行 |
| 连接池 | 高效的数据库连接管理 |
| 迁移管理 | 自动数据库迁移 |

## 3. SessionStore

### 3.1 概述

SessionStore 负责会话的持久化管理，包括创建、读取、更新、列表查询等操作。

### 3.2 核心接口

```python
class SessionStore:
    """Session 存储管理器"""

    def __init__(self, db_path: str):
        self.db_path = db_path
        self._init_db()

    def _init_db(self) -> None:
        """初始化数据库"""
        conn = sqlite3.connect(self.db_path)
        conn.execute("""
            CREATE TABLE IF NOT EXISTS sessions (
                id TEXT PRIMARY KEY,
                created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
                updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
                status TEXT NOT NULL,
                complexity_level TEXT,
                complexity_score INTEGER,
                model TEXT,
                message_count INTEGER DEFAULT 0,
                metadata JSON,
                last_activity TIMESTAMP
            )
        """)
        conn.execute("""
            CREATE INDEX IF NOT EXISTS idx_sessions_status ON sessions(status)
        """)
        conn.execute("""
            CREATE INDEX IF NOT EXISTS idx_sessions_updated ON sessions(updated_at)
        """)
        conn.close()

    async def create(self, session: Session) -> Session:
        """创建新会话"""
        conn = sqlite3.connect(self.db_path)
        conn.execute(
            """
            INSERT INTO sessions (id, status, complexity_level, complexity_score, model, metadata)
            VALUES (?, ?, ?, ?, ?, ?)
            """,
            (
                session.id,
                session.status.value,
                session.complexity_level.value if session.complexity_level else None,
                session.complexity_score,
                session.model,
                json.dumps(session.metadata or {}),
            )
        )
        conn.commit()
        conn.close()
        return session

    async def get(self, session_id: str) -> Optional[Session]:
        """获取会话"""
        conn = sqlite3.connect(self.db_path)
        cursor = conn.execute(
            "SELECT * FROM sessions WHERE id = ?",
            (session_id,)
        )
        row = cursor.fetchone()
        conn.close()

        if not row:
            return None

        return self._row_to_session(row)

    async def update(self, session: Session) -> Session:
        """更新会话"""
        conn = sqlite3.connect(self.db_path)
        conn.execute(
            """
            UPDATE sessions
            SET status = ?, complexity_level = ?, complexity_score = ?,
                model = ?, message_count = ?, metadata = ?,
                last_activity = CURRENT_TIMESTAMP, updated_at = CURRENT_TIMESTAMP
            WHERE id = ?
            """,
            (
                session.status.value,
                session.complexity_level.value if session.complexity_level else None,
                session.complexity_score,
                session.model,
                session.message_count,
                json.dumps(session.metadata or {}),
                session.id,
            )
        )
        conn.commit()
        conn.close()
        return session

    async def list(
        self,
        limit: int = 10,
        status: Optional[SessionStatus] = None,
    ) -> List[Session]:
        """列出会话"""
        conn = sqlite3.connect(self.db_path)

        if status:
            cursor = conn.execute(
                """
                SELECT * FROM sessions
                WHERE status = ?
                ORDER BY updated_at DESC
                LIMIT ?
                """,
                (status.value, limit)
            )
        else:
            cursor = conn.execute(
                """
                SELECT * FROM sessions
                ORDER BY updated_at DESC
                LIMIT ?
                """,
                (limit,)
            )

        rows = cursor.fetchall()
        conn.close()

        return [self._row_to_session(row) for row in rows]

    async def delete(self, session_id: str) -> bool:
        """删除会话"""
        conn = sqlite3.connect(self.db_path)
        cursor = conn.execute(
            "DELETE FROM sessions WHERE id = ?",
            (session_id,)
        )
        conn.commit()
        deleted = cursor.rowcount > 0
        conn.close()
        return deleted

    async def get_active(self) -> Optional[Session]:
        """获取最近活跃会话"""
        conn = sqlite3.connect(self.db_path)
        cursor = conn.execute(
            """
            SELECT * FROM sessions
            WHERE status = 'ACTIVE'
            ORDER BY last_activity DESC
            LIMIT 1
            """
        )
        row = cursor.fetchone()
        conn.close()

        if not row:
            return None

        return self._row_to_session(row)

    def _row_to_session(self, row: tuple) -> Session:
        """行转 Session 对象"""
        return Session(
            id=row[0],
            created_at=datetime.fromisoformat(row[1]),
            updated_at=datetime.fromisoformat(row[2]),
            status=SessionStatus(row[3]),
            complexity_level=ComplexityLevel(row[4]) if row[4] else None,
            complexity_score=row[5],
            model=row[6],
            message_count=row[7],
            metadata=json.loads(row[8]) if row[8] else {},
            last_activity=datetime.fromisoformat(row[9]) if row[9] else None,
        )
```

### 3.3 Session Schema

```python
@dataclass
class Session:
    id: str
    created_at: datetime
    updated_at: datetime
    status: SessionStatus
    complexity_level: Optional[ComplexityLevel] = None
    complexity_score: Optional[int] = None
    model: Optional[str] = None

    last_activity: Optional[datetime] = None

    COMPLETED = "COMPLETED"
    ERROR = "ERROR"

    SIMPLE = "SIMPLE"
    MEDIUM = "MEDIUM"
    COMPLEX = "COMPLEX"
```

## 4. ConfigStore

### 4.1 概述

ConfigStore 负责配置的管理和加载，支持环境变量、配置文件、默认配置多层级管理。

### 4.2 核心接口

```python
class ConfigStore:
    """配置存储管理器"""

    def __init__(self, config_dir: Path):
        self.config_dir = config_dir
        self.config_path = config_dir / "config.json"
        self._config: Dict[str, Any] = {}
        self._load()

    def _load(self) -> None:
        """加载配置"""
        # 1. 加载默认配置
        self._config = self._get_default_config()

        # 2. 加载配置文件
        if self.config_path.exists():
            with open(self.config_path) as f:
                user_config = json.load(f)
                self._config.update(user_config)

        # 3. 加载环境变量（覆盖）
        self._config.update(self._load_env_config())

    def _get_default_config(self) -> Dict[str, Any]:
        """获取默认配置"""
        return {
            "model": {
                "provider": "anthropic",
                "name": "claude-sonnet-4-20250514",
            },
            "memory": {
                "working_max_size": 100000,
                "short_term_ttl_days": 180,
            },
            "security": {
                "sandbox_mode": "off",
                "hitl_enabled": False,
            },
            "tools": {
                "timeout": 30,
                "allowed_commands": ["git", "npm", "pip"],
            },
        }

    def _load_env_config(self) -> Dict[str, Any]:
        """从环境变量加载配置"""
        env_config = {}

        if model := os.getenv("MOZI_MODEL"):
            env_config.setdefault("model", {})["name"] = model

        if path := os.getenv("MOZI_CONFIG_PATH"):
            env_config["config_path"] = path

        return env_config

    def get(self, key: str, default: Any = None) -> Any:
        """获取配置项"""
        keys = key.split(".")
        value = self._config

        for k in keys:
            if isinstance(value, dict):
                value = value.get(k)
                if value is None:
                    return default
            else:
                return default

        return value

    def set(self, key: str, value: Any) -> None:
        """设置配置项"""
        keys = key.split(".")
        config = self._config

        for k in keys[:-1]:
            if k not in config:
                config[k] = {}
            config = config[k]

        config[keys[-1]] = value

    async def save(self) -> None:
        """保存配置到文件"""
        self.config_dir.mkdir(parents=True, exist_ok=True)

        with open(self.config_path, "w") as f:
            json.dump(self._config, f, indent=2)
```

## 5. ArtifactStore

### 5.1 概述

ArtifactStore 负责存储大型文件（如图标、生成的代码压缩包等），使用文件系统存储。

### 5.2 核心接口

```python
class ArtifactStore:
    """Artifact 存储管理器"""

    def __init__(self, artifact_dir: Path):
        self.artifact_dir = artifact_dir
        self.artifact_dir.mkdir(parents=True, exist_ok=True)
        self._init_db()

    def _init_db(self) -> None:
        """初始化元数据库"""
        conn = sqlite3.connect(self.artifact_dir / "artifacts.db")
        conn.execute("""
            CREATE TABLE IF NOT EXISTS artifacts (
                id TEXT PRIMARY KEY,
                session_id TEXT,
                name TEXT NOT NULL,
                file_path TEXT NOT NULL,
                size INTEGER,
                content_type TEXT,
                created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
                FOREIGN KEY (session_id) REFERENCES sessions(id)
            )
        """)
        conn.execute("""
            CREATE INDEX IF NOT EXISTS idx_artifacts_session ON artifacts(session_id)
        """)
        conn.close()

    async def store(
        self,
        data: bytes,
        name: str,
        session_id: str,
        content_type: str = "application/octet-stream",
    ) -> Artifact:
        """存储 Artifact"""
        artifact_id = str(uuid.uuid4())
        file_path = self.artifact_dir / f"{artifact_id}_{name}"

        # 写入文件
        with open(file_path, "wb") as f:
            f.write(data)

        # 记录元数据
        conn = sqlite3.connect(self.artifact_dir / "artifacts.db")
        conn.execute(
            """
            INSERT INTO artifacts (id, session_id, name, file_path, size, content_type)
            VALUES (?, ?, ?, ?, ?, ?)
            """,
            (artifact_id, session_id, name, str(file_path), len(data), content_type)
        )
        conn.commit()
        conn.close()

        return Artifact(
            id=artifact_id,
            session_id=session_id,
            name=name,
            file_path=file_path,
            size=len(data),
            content_type=content_type,
        )

    async def get(self, artifact_id: str) -> Optional[Artifact]:
        """获取 Artifact 元数据"""
        conn = sqlite3.connect(self.artifact_dir / "artifacts.db")
        cursor = conn.execute(
            "SELECT * FROM artifacts WHERE id = ?",
            (artifact_id,)
        )
        row = cursor.fetchone()
        conn.close()

        if not row:
            return None

        return Artifact(
            id=row[0],
            session_id=row[1],
            name=row[2],
            file_path=Path(row[3]),
            size=row[4],
            content_type=row[5],
            created_at=datetime.fromisoformat(row[6]),
        )

    async def read(self, artifact_id: str) -> Optional[bytes]:
        """读取 Artifact 内容"""
        artifact = await self.get(artifact_id)
        if not artifact:
            return None

        with open(artifact.file_path, "rb") as f:
            return f.read()

    async def delete(self, artifact_id: str) -> bool:
        """删除 Artifact"""
        artifact = await self.get(artifact_id)
        if not artifact:
            return False

        # 删除文件
        if artifact.file_path.exists():
            artifact.file_path.unlink()

        # 删除元数据
        conn = sqlite3.connect(self.artifact_dir / "artifacts.db")
        conn.execute("DELETE FROM artifacts WHERE id = ?", (artifact_id,))
        conn.commit()
        conn.close()

        return True

    async def list_by_session(self, session_id: str) -> List[Artifact]:
        """列出会话的所有 Artifact"""
        conn = sqlite3.connect(self.artifact_dir / "artifacts.db")
        cursor = conn.execute(
            "SELECT * FROM artifacts WHERE session_id = ? ORDER BY created_at DESC",
            (session_id,)
        )
        rows = cursor.fetchall()
        conn.close()

        return [
            Artifact(
                id=row[0],
                session_id=row[1],
                name=row[2],
                file_path=Path(row[3]),
                size=row[4],
                content_type=row[5],
                created_at=datetime.fromisoformat(row[6]),
            )
            for row in rows
        ]
```

## 6. 数据库迁移

### 6.1 迁移管理

```python
class MigrationManager:
    """数据库迁移管理器"""

    def __init__(self, db_path: str):
        self.db_path = db_path

    async def migrate(self) -> None:
        """执行所有待执行迁移"""
        conn = sqlite3.connect(self.db_path)

        # 确保迁移表存在
        conn.execute("""
            CREATE TABLE IF NOT EXISTS schema_migrations (
                version INTEGER PRIMARY KEY,
                applied_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
            )
        """)

        # 获取当前版本
        cursor = conn.execute("SELECT MAX(version) FROM schema_migrations")
        current_version = cursor.fetchone()[0] or 0

        # 执行待执行迁移
        for version, migration in MIGRATIONS.items():
            if version > current_version:
                await self._apply_migration(conn, version, migration)

        conn.close()

    async def _apply_migration(
        self,
        conn: sqlite3.Connection,
        version: int,
        migration: Migration,
    ) -> None:
        """应用单个迁移"""
        try:
            migration.up(conn)
            conn.execute(
                "INSERT INTO schema_migrations (version) VALUES (?)",
                (version,)
            )
            conn.commit()
        except Exception as e:
            conn.rollback()
            raise MigrationError(f"Migration {version} failed: {e}")
```
