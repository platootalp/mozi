## 1. 模块概述

Security 模块是系统的安全层，提供跨切面的安全保障。包括权限检查（Permission）、沙箱隔离（Sandbox）、人工介入确认（HITL）、审计日志（Audit）。

### 1.1 核心职责

- 权限检查与实施
- 沙箱隔离
- 人工介入确认
- 操作审计日志

### 1.2 差异化特性

| 特性 | 说明 |
|------|------|
| 通配符匹配 | 支持 glob 模式匹配工具名和路径 |
| 多级沙箱 | off/non-main/all 三级隔离 |
| HITL | 高风险操作需人工确认 |
| 完整审计 | 所有操作可追溯 |

## 3. PermissionChecker

### 3.1 概述

PermissionChecker 负责权限检查，支持通配符模式匹配工具调用请求。

### 3.2 核心接口

```python
class PermissionChecker:
    """权限检查器"""

    def __init__(self, config: PermissionConfig):
        self.config = config
        self._load_policies()

    def _load_policies(self) -> None:
        """加载权限策略"""
        self.deny_patterns = self._compile_patterns(self.config.deny_list)
        self.ask_patterns = self._compile_patterns(self.config.ask_list)

    def _compile_patterns(self, patterns: List[str]) -> List[Pattern]:
        """编译 glob 模式为正则表达式"""
        compiled = []
        for pattern in patterns:
            # 将 glob 模式转换为正则
            regex = fnmatch.translate(pattern)
            compiled.append(re.compile(regex))
        return compiled

    async def check(
        self,
        tool_name: str,
        params: Dict[str, Any],
    ) -> PermissionResult:
        """
        检查权限
        优先级: deny > ask > allow
        """
        # 1. 检查 deny 名单
        if self._matches_patterns(tool_name, params, self.deny_patterns):
            return PermissionResult(
                status=PermissionStatus.DENY,
                reason="Tool or path is in deny list",
                tool=tool_name,
            )

        # 2. 检查 ask 名单
        if self._matches_patterns(tool_name, params, self.ask_patterns):
            return PermissionResult(
                status=PermissionStatus.ASK,
                reason="Tool or path requires human confirmation",
                tool=tool_name,
                requires_approval=True,
            )

        # 3. 默认 allow
        return PermissionResult(
            status=PermissionStatus.ALLOW,
            tool=tool_name,
        )

    def _matches_patterns(
        self,
        tool_name: str,
        params: Dict[str, Any],
        patterns: List[Pattern],
    ) -> bool:
        """检查是否匹配任何模式"""
        for pattern in patterns:
            # 检查工具名
            if pattern.match(tool_name):
                return True

            # 检查参数中的路径
            if path := params.get("path"):
                if isinstance(path, str) and pattern.match(path):
                    return True

            # 检查命令参数
            if command := params.get("command"):
                if isinstance(command, str) and pattern.match(command):
                    return True

        return False

@dataclass
class PermissionResult:
    status: PermissionStatus
    tool: str
    reason: Optional[str] = None
    requires_approval: bool = False

class PermissionStatus(Enum):
    ALLOW = "allow"
    DENY = "deny"
    ASK = "ask"
```

### 3.3 权限策略配置

        "rm -rf /*",           # 危险命令

## 4. Sandbox

### 4.1 概述

Sandbox 提供沙箱隔离，防止危险操作影响系统。

### 4.2 核心接口

```python
class Sandbox:
    """沙箱隔离"""

    def __init__(self, config: SandboxConfig):
        self.config = config

    async def check(self, tool_name: str, params: Dict[str, Any]) -> bool:
        """
        检查是否允许执行
        返回 True 表示允许，False 表示拒绝
        """
        if self.config.mode == SandboxMode.OFF:
            return True

        if self.config.mode == SandboxMode.NON_MAIN:
            return self._check_non_main(tool_name, params)

        if self.config.mode == SandboxMode.ALL:
            return self._check_all(tool_name, params)

        return True

    def _check_non_main(self, tool_name: str, params: Dict[str, Any]) -> bool:
        """non-main 模式：禁止访问 ~/ 和项目目录"""
        forbidden_paths = [
            os.path.expanduser("~"),
            str(Path.cwd()),  # 当前工作目录
        ]

        # 添加用户配置的项目路径
        forbidden_paths.extend(self.config.extra_forbidden_paths)

        return not self._contains_forbidden_path(params, forbidden_paths)

    def _check_all(self, tool_name: str, params: Dict[str, Any]) -> bool:
        """all 模式：禁止所有敏感路径"""
        sensitive_patterns = [
            "/etc/",
            "/usr/",
            "/var/",
            "/boot/",
            "/proc/",
            "/sys/",
            "/dev/",
            "/root/",
        ]

        return not self._contains_sensitive_path(params, sensitive_patterns)

    def _contains_forbidden_path(self, params: Dict[str, Any], paths: List[str]) -> bool:
        """检查参数中是否包含禁止路径"""
        for key, value in params.items():
            if not isinstance(value, str):
                continue

            for path in paths:
                if path in value:
                    return True

        return False

    def _contains_sensitive_path(self, params: Dict[str, Any], patterns: List[str]) -> bool:
        """检查参数中是否包含敏感路径"""
        for key, value in params.items():
            if not isinstance(value, str):
                continue

            for pattern in patterns:
                if pattern in value:
                    return True

        return False

class SandboxMode(Enum):
    OFF = "off"       # 无隔离
    NON_MAIN = "non-main"  # 禁止访问 ~/ 和项目目录
    ALL = "all"       # 禁止所有敏感路径

@dataclass
class SandboxConfig:
    mode: SandboxMode = SandboxMode.OFF
    extra_forbidden_paths: List[str] = field(default_factory=list)
```

### 4.3 沙箱模式对比

| 模式 | 行为 | 适用场景 |
|------|------|----------|
| `off` | 无隔离 | 完全信任的环境 |
| `non-main` | 禁止访问 ~/ 和项目目录 | 开发环境 |
| `all` | 禁止所有敏感路径 | 生产环境 |

## 5. HITL (Human-in-the-Loop)

### 5.1 概述

HITL 机制用于高风险操作，需要人工确认后才能执行。

### 5.2 核心接口

```python
class HITLManager:
    """HITL 管理器"""

    def __init__(self, config: HITLConfig):
        self.config = config
        self.pending_approvals: Dict[str, PendingApproval] = {}

    async def request_approval(
        self,
        tool_name: str,
        params: Dict[str, Any],
        context: ToolContext,
    ) -> str:
        """
        请求人工确认
        返回 approval_id
        """
        approval_id = str(uuid.uuid4())

        approval = PendingApproval(
            id=approval_id,
            tool_name=tool_name,
            params=params,
            context=context,
            created_at=datetime.now(),
            status=ApprovalStatus.PENDING,
        )

        self.pending_approvals[approval_id] = approval

        # 触发通知（CLI 提示、WebSocket 等）
        await self._notify_user(approval)

        return approval_id

    async def approve(self, approval_id: str) -> bool:
        """批准操作"""
        if approval_id not in self.pending_approvals:
            return False

        approval = self.pending_approvals[approval_id]
        approval.status = ApprovalStatus.APPROVED
        approval.decided_at = datetime.now()

        # 通知等待的调用者
        await self._notify_approved(approval)

        return True

    async def reject(self, approval_id: str, reason: Optional[str] = None) -> bool:
        """拒绝操作"""
        if approval_id not in self.pending_approvals:
            return False

        approval = self.pending_approvals[approval_id]
        approval.status = ApprovalStatus.REJECTED
        approval.decided_at = datetime.now()
        approval.reject_reason = reason

        # 通知等待的调用者
        await self._notify_rejected(approval)

        return True

    async def get_status(self, approval_id: str) -> Optional[PendingApproval]:
        """获取审批状态"""
        return self.pending_approvals.get(approval_id)

@dataclass
class PendingApproval:
    id: str
    tool_name: str
    params: Dict[str, Any]
    context: ToolContext
    created_at: datetime
    status: ApprovalStatus
    decided_at: Optional[datetime] = None
    reject_reason: Optional[str] = None

class ApprovalStatus(Enum):
    PENDING = "pending"
    APPROVED = "approved"
    REJECTED = "rejected"
    TIMEOUT = "timeout"
```

### 5.3 HITL 工作流

```
工具调用请求
    │
    ▼
PermissionChecker.check()
    │
    ├── ALLOW → 继续执行
    │
    ├── DENY → 返回错误
    │
    └── ASK → HITL 请求
              │
              ▼
        ┌─────────────┐
        │ 用户确认？   │
        └──────┬──────┘
               │
        ┌──────┼──────┐
        │      │      │
        ▼      ▼      ▼
      批准    拒绝   超时
        │      │      │
        ▼      ▼      ▼
      执行   返回   返回
      结果   错误   超时
```

## 6. Audit

### 6.1 概述

Audit 模块记录所有操作的可审计日志。

### 6.2 核心接口

```python
class AuditLogger:
    """审计日志记录器"""

    def __init__(self, db_path: str):
        self.db_path = db_path
        self._init_db()

    def _init_db(self) -> None:
        """初始化审计数据库"""
        conn = sqlite3.connect(self.db_path)
        conn.execute("""
            CREATE TABLE IF NOT EXISTS audit_logs (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                session_id TEXT,
                action TEXT NOT NULL,
                tool TEXT,
                params JSON,
                result TEXT,
                user_decision TEXT,
                ip_address TEXT,
                user_agent TEXT,
                created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
            )
        """)
        conn.execute("""
            CREATE INDEX IF NOT EXISTS idx_audit_session ON audit_logs(session_id)
        """)
        conn.execute("""
            CREATE INDEX IF NOT EXISTS idx_audit_action ON audit_logs(action)
        """)
        conn.execute("""
            CREATE INDEX IF NOT EXISTS idx_audit_created ON audit_logs(created_at)
        """)
        conn.close()

    async def log(
        self,
        action: str,
        session_id: Optional[str] = None,
        tool: Optional[str] = None,
        params: Optional[Dict[str, Any]] = None,
        result: Optional[str] = None,
        user_decision: Optional[str] = None,
    ) -> None:
        """记录审计日志"""
        conn = sqlite3.connect(self.db_path)
        conn.execute(
            """
            INSERT INTO audit_logs
            (session_id, action, tool, params, result, user_decision)
            VALUES (?, ?, ?, ?, ?, ?)
            """,
            (
                session_id,
                action,
                tool,
                json.dumps(params) if params else None,
                result,
                user_decision,
            )
        )
        conn.commit()
        conn.close()

    async def query(
        self,
        session_id: Optional[str] = None,
        action: Optional[str] = None,
        tool: Optional[str] = None,
        start_time: Optional[datetime] = None,
        end_time: Optional[datetime] = None,
        limit: int = 100,
    ) -> List[AuditLog]:
        """查询审计日志"""
        conn = sqlite3.connect(self.db_path)

        query = "SELECT * FROM audit_logs WHERE 1=1"
        params = []

        if session_id:
            query += " AND session_id = ?"
            params.append(session_id)

        if action:
            query += " AND action = ?"
            params.append(action)

        if tool:
            query += " AND tool = ?"
            params.append(tool)

        if start_time:
            query += " AND created_at >= ?"
            params.append(start_time.isoformat())

        if end_time:
            query += " AND created_at <= ?"
            params.append(end_time.isoformat())

        query += " ORDER BY created_at DESC LIMIT ?"
        params.append(limit)

        cursor = conn.execute(query, params)
        rows = cursor.fetchall()
        conn.close()

        return [
            AuditLog(
                id=row[0],
                session_id=row[1],
                action=row[2],
                tool=row[3],
                params=json.loads(row[4]) if row[4] else None,
                result=row[5],
                user_decision=row[6],
                created_at=datetime.fromisoformat(row[8]),
            )
            for row in rows
        ]

@dataclass
class AuditLog:
    id: int
    session_id: Optional[str]
    action: str
    tool: Optional[str]
    params: Optional[Dict[str, Any]]
    result: Optional[str]
    user_decision: Optional[str]
    created_at: datetime
```
