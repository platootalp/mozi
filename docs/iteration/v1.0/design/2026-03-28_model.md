## 1. 模块概述

Model 模块是系统的模型网关层，负责统一管理多个 LLM Provider。通过抽象接口，支持 Anthropic、OpenAI、Ollama 等多种模型，提供统一的调用方式和智能路由能力。

### 1.1 核心职责

- 多模型统一接口
- 模型智能路由
- Provider 适配
- 请求/响应处理
- Token 计费管理

### 1.2 差异化特性

| 特性 | 说明 |
|------|------|
| 多 Provider | 支持 Anthropic/OpenAI/Ollama |
| 统一接口 | 屏蔽 Provider 差异 |
| 智能路由 | 根据任务类型选择最佳模型 |
| 成本优化 | Token 用量追踪与控制 |

## 3. 组件详情

### 3.1 ModelGateway

**职责**: 多模型统一入口

**核心功能点**:

| 功能 | 说明 |
|------|------|
| complete() | 发送对话请求 |
| embeddings() | 获取文本嵌入 |
| 模型路由 | 根据配置/任务选择模型 |
| 响应缓存 | 缓存相同请求的响应 |

**接口定义**:

```python
class ModelGateway:
    PROVIDERS: Dict[str, Type[BaseProvider]] = {
        "anthropic": AnthropicProvider,
        "openai": OpenAIProvider,
        "ollama": OllamaProvider,
    }

    def __init__(
        self,
        router: Optional[Router] = None,
        cache: Optional[ResponseCache] = None,
    ):
        self.router = router or DefaultRouter()
        self.cache = cache or ResponseCache()
        self.providers: Dict[str, BaseProvider] = {}
        self._init_providers()

    def _init_providers(self) -> None:
        """初始化所有 Provider"""
        for name, provider_class in self.PROVIDERS.items():
            self.providers[name] = provider_class()

    async def complete(
        self,
        messages: List[Message],
        model: Optional[str] = None,
        **kwargs
    ) -> Response:
        """发送对话请求"""
        # 1. 解析模型名称
        provider_name, model_name = self.router.resolve(model)

        # 2. 检查缓存
        cache_key = self._make_cache_key(messages, model_name)
        if cached := await self.cache.get(cache_key):
            return cached

        # 3. 调用 Provider
        provider = self.providers[provider_name]
        response = await provider.complete(messages, model_name, **kwargs)

        # 4. 缓存响应
        await self.cache.set(cache_key, response)

        return response

    async def embeddings(self, texts: List[str], model: Optional[str] = None) -> List[float]:
        """获取文本嵌入"""
        provider_name, model_name = self.router.resolve(model or "embed")
        provider = self.providers.get(provider_name)
        if not provider:
            raise ModelNotFoundError(f"Provider {provider_name} not found")
        return await provider.embeddings(texts, model_name)
```

### 3.2 Router

**职责**: 模型路由决策

**路由策略**:

| 策略 | 说明 |
|------|------|
| 默认路由 | 使用配置中的默认模型 |
| 任务路由 | 根据任务类型选择模型 |
| 成本路由 | 选择最低成本模型 |
| 负载路由 | 均衡多个模型负载 |

```python
class Router(ABC):
    @abstractmethod
    def resolve(self, model: Optional[str]) -> Tuple[str, str]:
        """
        解析模型名称
        返回: (provider_name, model_name)
        """
        pass

class DefaultRouter(Router):
    def __init__(self, default_provider: str = "anthropic", default_model: str = "claude-sonnet-4-20250514"):
        self.default_provider = default_provider
        self.default_model = default_model

    def resolve(self, model: Optional[str]) -> Tuple[str, str]:
        if model:
            return self._parse_model(model)
        return self.default_provider, self.default_model

    def _parse_model(self, model: str) -> Tuple[str, str]:
        """解析 model 名称，支持 'provider/model' 格式"""
        if "/" in model:
            provider, name = model.split("/", 1)
            return provider, name
        return self.default_provider, model

class TaskRouter(Router):
    """基于任务类型的路由"""
    TASK_MODEL_MAP = {
        Intent.CODE: ("anthropic", "claude-sonnet-4-20250514"),
        Intent.REVIEW: ("anthropic", "claude-sonnet-4-20250514"),
        Intent.EXPLORE: ("openai", "gpt-4o"),
        Intent.PLAN: ("anthropic", "claude-opus-4-6"),
        Intent.RESEARCH: ("openai", "gpt-4o"),
    }

    def resolve(self, model: Optional[str]) -> Tuple[str, str]:
        if model:
            return self._parse_model(model)

        # 根据上下文中的意图选择模型
        intent = self._get_current_intent()
        return self.TASK_MODEL_MAP.get(intent, ("anthropic", "claude-sonnet-4-20250514"))
```

### 3.3 BaseProvider

**职责**: Provider 抽象基类

```python
class BaseProvider(ABC):
    """Provider 基类，所有 Provider 必须实现"""

    @property
    @abstractmethod
    def name(self) -> str:
        """Provider 名称"""
        pass

    @property
    @abstractmethod
    def supported_models(self) -> List[str]:
        """支持的模型列表"""
        pass

    @abstractmethod
    async def complete(
        self,
        messages: List[Message],
        model: str,
        **kwargs
    ) -> Response:
        """发送对话请求"""
        pass

    @abstractmethod
    async def embeddings(self, texts: List[str], model: str) -> List[float]:
        """获取文本嵌入"""
        pass

    def validate_model(self, model: str) -> bool:
        """验证模型是否支持"""
        return model in self.supported_models
```

### 3.4 AnthropicProvider

**职责**: Anthropic API 适配

```python
class AnthropicProvider(BaseProvider):
    BASE_URL = "https://api.anthropic.com/v1"

    def __init__(self, api_key: Optional[str] = None):
        self.api_key = api_key or os.getenv("ANTHROPIC_API_KEY")
        self.client = Anthropic(api_key=self.api_key)

    @property
    def name(self) -> str:
        return "anthropic"

    @property
    def supported_models(self) -> List[str]:
        return [
            "claude-opus-4-6",
            "claude-sonnet-4-20250514",
            "claude-haiku-4-5-20251001",
        ]

    async def complete(
        self,
        messages: List[Message],
        model: str,
        **kwargs
    ) -> Response:
        """调用 Anthropic API"""
        response = await self.client.messages.create(
            model=model,
            messages=[m.to_dict() for m in messages],
            **kwargs
        )
        return Response(
            content=response.content[0].text,
            model=model,
            usage=Usage(
                input_tokens=response.usage.input_tokens,
                output_tokens=response.usage.output_tokens,
            )
        )

    async def embeddings(self, texts: List[str], model: str) -> List[float]:
        raise NotImplementedError("Anthropic does not support embeddings")
```

### 3.5 OpenAIProvider

**职责**: OpenAI API 适配

```python
class OpenAIProvider(BaseProvider):
    BASE_URL = "https://api.openai.com/v1"

    def __init__(self, api_key: Optional[str] = None):
        self.api_key = api_key or os.getenv("OPENAI_API_KEY")
        self.client = OpenAI(api_key=self.api_key)

    @property
    def name(self) -> str:
        return "openai"

    @property
    def supported_models(self) -> List[str]:
        return [
            "gpt-4o",
            "gpt-4o-mini",
            "gpt-4-turbo",
            "gpt-3.5-turbo",
        ]

    async def complete(
        self,
        messages: List[Message],
        model: str,
        **kwargs
    ) -> Response:
        """调用 OpenAI API"""
        response = await self.client.chat.completions.create(
            model=model,
            messages=[m.to_dict() for m in messages],
            **kwargs
        )
        return Response(
            content=response.choices[0].message.content,
            model=model,
            usage=Usage(
                input_tokens=response.usage.prompt_tokens,
                output_tokens=response.usage.completion_tokens,
            )
        )

    async def embeddings(self, texts: List[str], model: str = "text-embedding-3-small") -> List[float]:
        """获取文本嵌入"""
        response = await self.client.embeddings.create(
            model=model,
            input=texts,
        )
        return [item.embedding for item in response.data]
```

### 3.6 OllamaProvider

**职责**: Ollama 本地模型适配

```python
class OllamaProvider(BaseProvider):
    BASE_URL = "http://localhost:11434"

    def __init__(self, base_url: Optional[str] = None):
        self.base_url = base_url or os.getenv("OLLAMA_BASE_URL", self.BASE_URL)
        self.client = AsyncHTTPClient(base_url=self.base_url)

    @property
    def name(self) -> str:
        return "ollama"

    @property
    def supported_models(self) -> List[str]:
        return ["llama2", "codellama", "mistral", "mixtral"]

    async def complete(
        self,
        messages: List[Message],
        model: str,
        **kwargs
    ) -> Response:
        """调用 Ollama API"""
        payload = {
            "model": model,
            "messages": [m.to_dict() for m in messages],
            **kwargs
        }
        response = await self.client.post("/api/chat", json=payload)
        data = response.json()
        return Response(
            content=data["message"]["content"],
            model=model,
            usage=Usage(
                input_tokens=data.get("prompt_eval_count", 0),
                output_tokens=data.get("eval_count", 0),
            )
        )

    async def embeddings(self, texts: List[str], model: str = "nomic-embed-text") -> List[float]:
        """获取文本嵌入"""
        embeddings = []
        for text in texts:
            payload = {"model": model, "prompt": text}
            response = await self.client.post("/api/embeddings", json=payload)
            embeddings.append(response.json()["embedding"])
        return embeddings
```

## 4. 响应缓存

```python
class ResponseCache:
    """响应缓存，支持 LRU 策略"""

    def __init__(self, max_size: int = 1000, ttl: int = 3600):
        self.max_size = max_size
        self.ttl = ttl
        self._cache: OrderedDict[str, CacheEntry] = OrderedDict()

    async def get(self, key: str) -> Optional[Response]:
        """获取缓存的响应"""
        if key not in self._cache:
            return None

        entry = self._cache[key]
        if time.time() - entry.created_at > self.ttl:
            del self._cache[key]
            return None

        # 移动到末尾（最近使用）
        self._cache.move_to_end(key)
        return entry.response

    async def set(self, key: str, response: Response) -> None:
        """设置缓存"""
        if len(self._cache) >= self.max_size:
            # 删除最旧的条目
            self._cache.popitem(last=False)

        self._cache[key] = CacheEntry(response=response, created_at=time.time())
```

## 5. 类型定义

```python
@dataclass
class Message:
    role: str
    content: str

    def to_dict(self) -> dict:
        return {"role": self.role, "content": self.content}

@dataclass
class Response:
    content: str
    model: str
    usage: Usage
    stop_reason: Optional[str] = None

@dataclass
class Usage:
    input_tokens: int
    output_tokens: int

    @property
    def total_tokens(self) -> int:
        return self.input_tokens + self.output_tokens
```
