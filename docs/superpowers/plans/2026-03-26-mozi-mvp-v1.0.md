```python
    """Base exception for all Mozi errors."""

    pass

    """Raised when configuration is invalid or cannot be loaded."""

    def __init__(self, path: str, message: str) -> None:

    pass
```

```python
"""SQLite database wrapper with async support."""

from __future__ import annotations

import json
import uuid
from contextlib import asynccontextmanager
from pathlib import Path
from typing import Any, AsyncIterator, Optional

import aiosqlite

from mozi.core.error import ConfigError, SessionError


class Database:
    """Async SQLite database wrapper."""

    def __init__(self, db_path: Path) -> None:
        self.db_path = Path(db_path).expanduser()
        self._conn: Optional[aiosqlite.Connection] = None

    async def initialize(self) -> None:
        """Initialize database connection and create schema."""
        self._conn = await aiosqlite.connect(self.db_path)
        self._conn.row_factory = aiosqlite.Row
        # Enable WAL mode for better concurrency
        await self._conn.execute("PRAGMA journal_mode=WAL")
        await self._conn.executescript(self._get_schema())
        await self._conn.commit()

    @staticmethod
    def _get_schema() -> str:
        """Return schema SQL."""
        from mozi.infrastructure.db.schema import SCHEMA_SQL
        return SCHEMA_SQL

    async def close(self) -> None:
        """Close database connection."""
        if self._conn:
            await self._conn.close()
            self._conn = None

    @asynccontextmanager
    async def execute(
        self, query: str, *args: Any
    ) -> AsyncIterator[aiosqlite.Cursor]:
        """Execute a query and return cursor."""
        if not self._conn:
            raise ConfigError(str(self.db_path), "Database not initialized")
        cursor = await self._conn.execute(query, args)
        try:
            yield cursor
        finally:
            await cursor.close()

    async def execute_many(self, query: str, args_list: list[tuple]) -> None:
        """Execute many queries in a transaction."""
        if not self._conn:
            raise ConfigError(str(self.db_path), "Database not initialized")
        await self._conn.executemany(query, args_list)
        await self._conn.commit()

    # Session operations
    async def create_session(self, user_input: str) -> str:
        """Create a new session and return its ID."""
        session_id = str(uuid.uuid4())
        async with self.execute(
            """
            INSERT INTO sessions (id, user_input, status)
            VALUES (?, ?, 'active')
            """,
            (session_id, user_input),
        ):
            pass
        return session_id

    async def get_session(self, session_id: str) -> Optional[dict[str, Any]]:
        """Get session by ID."""
        async with self.execute(
            "SELECT * FROM sessions WHERE id = ?", (session_id,)
        ) as cursor:
            row = await cursor.fetchone()
            if row:
                return dict(row)
            return None

    async def update_session_status(
        self, session_id: str, status: str
    ) -> None:
        """Update session status."""
        async with self.execute(
            """
            UPDATE sessions
            SET status = ?, updated_at = datetime('now'), ended_at = datetime('now')
            WHERE id = ?
            """,
            (status, session_id),
        ):
            pass

    async def update_session_intent(
        self,
        session_id: str,
        intent_data: dict[str, Any],
        complexity_score: float,
        complexity_level: str,
    ) -> None:
        """Update session with intent and complexity data."""
        async with self.execute(
            """
            UPDATE sessions
            SET intent_data = ?, complexity_score = ?, complexity_level = ?,
                updated_at = datetime('now')
            WHERE id = ?
            """,
            (json.dumps(intent_data), complexity_score, complexity_level, session_id),
        ):
            pass

    # Checkpoint operations
    async def create_checkpoint(
        self,
        session_id: str,
        agent_state: dict[str, Any],
        tool_results: Optional[list[dict[str, Any]]] = None,
    ) -> int:
        """Create a checkpoint and return its ID."""
        async with self.execute(
            """
            INSERT INTO checkpoints (session_id, agent_state, tool_results)
            VALUES (?, ?, ?)
            """,
            (session_id, json.dumps(agent_state), json.dumps(tool_results or [])),
        ):
            pass
        cursor = await self._conn.execute(
            "SELECT last_insert_rowid()"
        )
        row = await cursor.fetchone()
        return row[0] if row else 0

    async def get_latest_checkpoint(
        self, session_id: str
    ) -> Optional[dict[str, Any]]:
        """Get the latest checkpoint for a session."""
        async with self.execute(
            """
            SELECT * FROM checkpoints
            WHERE session_id = ?
            ORDER BY created_at DESC
            LIMIT 1
            """,
            (session_id,),
        ) as cursor:
            row = await cursor.fetchone()
            if row:
                result = dict(row)
                result["agent_state"] = json.loads(result["agent_state"])
                result["tool_results"] = json.loads(result["tool_results"])
                return result
            return None

    # Audit operations
    async def log_tool_call(
        self,
        session_id: str,
        tool_name: str,
        params: dict[str, Any],
        result: Optional[str],
        success: bool,
    ) -> None:
        """Log a tool call for audit."""
        async with self.execute(
            """
            INSERT INTO tool_audit (session_id, tool_name, params, result, success)
            VALUES (?, ?, ?, ?, ?)
            """,
            (session_id, tool_name, json.dumps(params), result, 1 if success else 0),
        ):
            pass
```

```python
"""Tool registry for managing available tools."""

from __future__ import annotations

from typing import Any, Optional

from mozi.core.error import ToolError


class Tool:
    """Base tool class."""

    name: str = ""
    description: str = ""
    parameters: dict[str, Any] = {}

    async def execute(self, **kwargs: Any) -> Any:
        """Execute the tool."""
        raise NotImplementedError


class ToolRegistry:
    """Registry for managing available tools."""

    def __init__(self) -> None:
        self._tools: dict[str, Tool] = {}

    def register(self, tool: Tool) -> None:
        """Register a tool."""
        if tool.name in self._tools:
            raise ToolError(tool.name, f"Tool already registered: {tool.name}")
        self._tools[tool.name] = tool

    def get(self, name: str) -> Optional[Tool]:
        """Get a tool by name."""
        return self._tools.get(name)

    def list_tools(self) -> list[Tool]:
        """List all registered tools."""
        return list(self._tools.values())

    def get_tool_schema(self, name: str) -> Optional[dict[str, Any]]:
        """Get tool schema for LLM function calling."""
        tool = self.get(name)
        if not tool:
            return None
        return {
            "name": tool.name,
            "description": tool.description,
            "input_schema": tool.parameters,
        }
```

```python
import hashlib
from pathlib import Path
from typing import Any

from mozi.capabilities.tools.registry import Tool


class EditTool(Tool):
    """Tool for editing existing files with conflict detection."""

    name = "edit"
    description = "Edit an existing file with hash anchoring for conflict detection"
    parameters = {
        "type": "object",
        "properties": {
            "path": {
                "type": "string",
                "description": "Path to the file to edit",
            },
            "old_string": {
                "type": "string",
                "description": "Exact content to replace",
            },
            "new_string": {
                "type": "string",
                "description": "Replacement content",
            },
            "anchor": {
                "type": "string",
                "description": "Hash anchor for conflict detection (optional)",
            },
        },
        "required": ["path", "old_string", "new_string"],
    }

    async def execute(
        self, path: str, old_string: str, new_string: str, anchor: str = ""
    ) -> str:
        """Edit file with hash anchoring."""
        file_path = Path(path)

        if not file_path.exists():
            return f"Error: File not found: {path}"

        try:
            with open(file_path, "r", encoding="utf-8") as f:
                content = f.read()

            # Verify anchor if provided
            if anchor:
                content_hash = hashlib.sha256(content.encode()).hexdigest()[:16]
                if content_hash != anchor:
                    return f"Error: Hash mismatch. Expected {anchor}, got {content_hash}"

            # Check if old_string exists
            if old_string not in content:
                return f"Error: old_string not found in file"

            # Perform replacement
            new_content = content.replace(old_string, new_string, 1)

            with open(file_path, "w", encoding="utf-8") as f:
                f.write(new_content)

            return f"Successfully edited file: {path}"
        except Exception as e:
            return f"Error editing file: {e}"
```

```python
from typing import Any

from mozi.capabilities.tools.registry import Tool

    """Tool for executing bash commands."""

    name = "bash"
    description = "Execute a bash command"
    parameters = {
        "type": "object",
        "properties": {
            "command": {
                "type": "string",
                "description": "The bash command to execute",
            },
            "cwd": {
                "type": "string",
                "description": "Working directory for the command",
            },
            "timeout": {
                "type": "integer",
                "description": "Command timeout in seconds",
            },
        },
        "required": ["command"],
    }

    async def execute(
        self, command: str, cwd: str = ".", timeout: int = 30
    ) -> str:
        """Execute bash command."""
        try:
            args = shlex.split(command)
            process = await asyncio.create_subprocess_exec(
                *args,
                stdout=asyncio.subprocess.PIPE,
                stderr=asyncio.subprocess.PIPE,
                cwd=cwd,
            )
            try:
                stdout, stderr = await asyncio.wait_for(
                    process.communicate(), timeout=timeout
                )
                result = stdout.decode() if stdout else ""
                if stderr:
                    result += "\nSTDERR:\n" + stderr.decode()
                return result
            except asyncio.TimeoutError:
                process.kill()
                return f"Error: Command timed out after {timeout} seconds"
        except Exception as e:
            return f"Error executing command: {e}"
```

```python
from __future__ import annotations

import re
from pathlib import Path
from typing import Any

from mozi.capabilities.tools.registry import Tool


class GrepTool(Tool):
    """Tool for searching file contents with regex."""

    name = "grep"
    description = "Search for patterns in files using regex"
    parameters = {
        "type": "object",
        "properties": {
            "pattern": {
                "type": "string",
                "description": "Regex pattern to search for",
            },
            "path": {
                "type": "string",
                "description": "Directory or file path to search in",
            },
            "file_pattern": {
                "type": "string",
                "description": "File pattern to match (e.g., *.py)",
            },
            "context": {
                "type": "integer",
                "description": "Number of context lines to show",
            },
        },
        "required": ["pattern", "path"],
    }

    async def execute(
        self,
        pattern: str,
        path: str,
        file_pattern: str = "*",
        context: int = 0,
    ) -> str:
        """Search for pattern in files."""
        try:
            regex = re.compile(pattern)
        except re.error as e:
            return f"Error: Invalid regex pattern: {e}"

        results = []
        search_path = Path(path)

        if search_path.is_file():
            files = [search_path]
        else:
            files = search_path.rglob(file_pattern)

        for file_path in files:
            if not file_path.is_file():
                continue
            try:
                with open(file_path, "r", encoding="utf-8") as f:
                    lines = f.readlines()
                for i, line in enumerate(lines, 1):
                    if regex.search(line):
                        if context > 0:
                            start = max(0, i - context - 1)
                            end = min(len(lines), i + context)
                            snippet = "".join(lines[start:end])
                            results.append(f"{file_path}:{i}:{snippet}")
                        else:
                            results.append(f"{file_path}:{i}:{line.rstrip()}")
            except Exception:
                continue

        if not results:
            return "No matches found"

        return "\n".join(results[:100])  # Limit output
```

```python
"""Tool framework orchestrating tool execution."""

from __future__ import annotations

from typing import Any, Optional

from mozi.capabilities.tools.registry import Tool, ToolRegistry
from mozi.capabilities.tools.builtin import (
    ReadTool,
    WriteTool,
    EditTool,
    BashTool,
    GrepTool,
    GlobTool,
)
from mozi.core.error import ToolError


class ToolFramework:
    """Framework for tool registration and execution."""

    def __init__(self) -> None:
        self.registry = ToolRegistry()
        self._register_builtin_tools()

    def _register_builtin_tools(self) -> None:
        """Register built-in tools."""
        tools = [
            ReadTool(),
            WriteTool(),
            EditTool(),
            BashTool(),
            GrepTool(),
            GlobTool(),
        ]
        for tool in tools:
            self.registry.register(tool)

    def get_tool_schema(self, name: str) -> Optional[dict[str, Any]]:
        """Get tool schema for LLM function calling."""
        return self.registry.get_tool_schema(name)

    def list_tool_schemas(self) -> list[dict[str, Any]]:
        """Get schemas for all tools."""
        return [
            schema
            for tool in self.registry.list_tools()
            if (schema := self.get_tool_schema(tool.name))
        ]

    async def execute(
        self, tool_name: str, params: dict[str, Any]
    ) -> dict[str, Any]:
        """Execute a tool by name with given parameters."""
        tool = self.registry.get(tool_name)
        if not tool:
            raise ToolError(tool_name, f"Tool not found: {tool_name}")

        try:
            result = await tool.execute(**params)
            return {"success": True, "output": result}
        except Exception as e:
            return {"success": False, "error": str(e)}
```

```python
    tool_results: list[dict[str, Any]] = field(default_factory=list)

                "step": self.current_step,
            }
        )
```

```python
"""Configuration loader with multi-source merging."""

from __future__ import annotations

import json
import os
from pathlib import Path
from typing import Optional

from mozi.capabilities.config.schemas.config import Config
from mozi.capabilities.config.schemas.agents import AgentsRegistry
from mozi.capabilities.config.schemas.tools import ToolsConfig
from mozi.core.error import ConfigError


class ConfigLoader:
    """Loads and merges configuration from multiple sources."""

    def __init__(
        self,
        config_dir: Optional[Path] = None,
        user_dir: Optional[Path] = None,
    ) -> None:
        self.config_dir = config_dir or Path.cwd() / ".mozi"
        self.user_dir = user_dir or Path.home() / ".mozi"
        self._config: Optional[Config] = None
        self._agents: Optional[AgentsRegistry] = None
        self._tools: Optional[ToolsConfig] = None

    def load(self) -> None:
        """Load all configuration files."""
        self._config = self._load_config()
        self._agents = self._load_agents()
        self._tools = self._load_tools()

    def _load_config(self) -> Config:
        """Load main config.json with priority: env > config_dir > user_dir > defaults."""
        # Start with defaults
        config = Config()

        # Check for user config
        user_config_path = self.user_dir / "config.json"
        if user_config_path.exists():
            config = self._merge_config(config, user_config_path)

        # Check for project config
        project_config_path = self.config_dir / "config.json"
        if project_config_path.exists():
            config = self._merge_config(config, project_config_path)

        # Environment variable overrides
        if host := os.getenv("MOZI_SERVER_HOST"):
            config.server.host = host
        if port := os.getenv("MOZI_SERVER_PORT"):
            config.server.port = int(port)
        if db_path := os.getenv("MOZI_DB_PATH"):
            config.storage.sqlite_path = Path(db_path)

        return config

    def _merge_config(self, base: Config, path: Path) -> Config:
        """Merge config from a file into base config."""
        with open(path, "r") as f:
            data = json.load(f)
        return Config.model_validate({**base.model_dump(), **data})

    def _load_agents(self) -> AgentsRegistry:
        """Load agents.json."""
        for path in [self.config_dir / "agents.json", self.user_dir / "agents.json"]:
            if path.exists():
                with open(path, "r") as f:
                    data = json.load(f)
                return AgentsRegistry.model_validate(data)
        return AgentsRegistry()

    def _load_tools(self) -> ToolsConfig:
        """Load tools.json."""
        for path in [self.config_dir / "tools.json", self.user_dir / "tools.json"]:
            if path.exists():
                with open(path, "r") as f:
                    data = json.load(f)
                return ToolsConfig.model_validate(data)
        return ToolsConfig()

    def get_config(self) -> Config:
        """Get main configuration."""
        if self._config is None:
            self.load()
        return self._config

    def get_agents(self) -> AgentsRegistry:
        """Get agents registry."""
        if self._agents is None:
            self.load()
        return self._agents

    def get_tools(self) -> ToolsConfig:
        """Get tools configuration."""
        if self._tools is None:
            self.load()
        return self._tools
```

```python
"""Intent recognition for user input."""

from __future__ import annotations

import re
from dataclasses import dataclass
from typing import Optional


@dataclass
class Intent:
    """Structured intent from user input."""

    task_type: str  # bug_fix, feature_add, refactor, docs, test, ci, ...
    scope: str  # single_file, multi_file, cross_module, cross_repo
    explicit_deps: list[str]
    implied_deps: list[str]
    clarification_needed: bool
    confidence: float

    # Extracted entities
    target_files: list[str] = None
    raw_input: str = ""

    def __post_init__(self) -> None:
        if self.target_files is None:
            self.target_files = []


class IntentClassifier:
    """Simple rule-based intent classifier for MVP."""

    TASK_PATTERNS = {
        "bug_fix": [
            r"fix",
            r"bug",
            r"crash",
            r"error",
            r"wrong",
            r"broken",
        ],
        "feature_add": [
            r"add",
            r"new",
            r"create",
            r"implement",
            r"feature",
        ],
        "refactor": [
            r"refactor",
            r"restructure",
            r"reorganize",
            r"clean up",
        ],
        "docs": [
            r"document",
            r"readme",
            r"comment",
            r"docs",
        ],
        "test": [
            r"test",
            r"spec",
            r"coverage",
        ],
        "ci": [
            r"ci",
            r"pipeline",
            r"github actions",
            r"workflow",
        ],
    }

    FILE_PATTERNS = [
        r"(\w+\.\w+)",  # filename.ext
        r"(\w+/[\w/]+\.\w+)",  # path/to/file.ext
    ]

    def classify(self, user_input: str) -> Intent:
        """Classify user input into structured intent."""
        task_type = self._classify_task_type(user_input)
        scope = self._classify_scope(user_input)
        target_files = self._extract_files(user_input)
        confidence = self._calculate_confidence(task_type, target_files)

        return Intent(
            task_type=task_type,
            scope=scope,
            explicit_deps=target_files,
            implied_deps=[],
            clarification_needed=confidence < 0.5,
            confidence=confidence,
            target_files=target_files,
            raw_input=user_input,
        )

    def _classify_task_type(self, user_input: str) -> str:
        """Classify the task type based on keywords."""
        user_input_lower = user_input.lower()

        for task_type, patterns in self.TASK_PATTERNS.items():
            for pattern in patterns:
                if re.search(pattern, user_input_lower):
                    return task_type

        return "general"  # Default

    def _classify_scope(self, user_input: str) -> str:
        """Classify the scope based on file references."""
        files = self._extract_files(user_input)
        if len(files) == 0:
            return "unknown"
        elif len(files) == 1:
            return "single_file"
        elif len(files) <= 3:
            return "multi_file"
        else:
            return "cross_module"

    def _extract_files(self, user_input: str) -> list[str]:
        """Extract file references from user input."""
        files = []
        for pattern in self.FILE_PATTERNS:
            matches = re.findall(pattern, user_input)
            files.extend(matches)
        return list(set(files))  # Deduplicate

    def _calculate_confidence(self, task_type: str, files: list[str]) -> float:
        """Calculate confidence score."""
        base_confidence = 0.5

        # Higher confidence if task type is specific
        if task_type != "general":
            base_confidence += 0.2

        # Higher confidence if files are identified
        if files:
            base_confidence += 0.15

        # Higher confidence if multiple signals match
        # (could add more heuristics)

        return min(base_confidence, 1.0)
```

```python
"""Complexity assessment engine."""

from __future__ import annotations

from dataclasses import dataclass
from typing import Any

from mozi.orchestrator.core.intent import Intent


@dataclass
class ComplexityScore:
    """Result of complexity assessment."""

    total: float
    level: str  # SIMPLE, MEDIUM, COMPLEX
    dimensions: dict[str, float]

    # Thresholds
    SIMPLE_THRESHOLD = 40
    MEDIUM_THRESHOLD = 70


class ComplexityEngine:
    """Assesses task complexity based on multiple dimensions."""

    def __init__(
        self,
        simple_threshold: int = ComplexityScore.SIMPLE_THRESHOLD,
        medium_threshold: int = ComplexityScore.MEDIUM_THRESHOLD,
    ) -> None:
        self.simple_threshold = simple_threshold
        self.medium_threshold = medium_threshold

    def assess(self, intent: Intent) -> ComplexityScore:
        """Assess complexity of a task."""
        dimensions = {
            "file_count": self._score_file_count(intent),
            "scope": self._score_scope(intent),
            "dependency": self._score_dependency(intent),
            "type": self._score_task_type(intent),
        }

        # Weighted average
        weights = {"file_count": 0.3, "scope": 0.3, "dependency": 0.2, "type": 0.2}
        total = sum(dimensions[k] * weights[k] for k in weights)

        level = self._determine_level(total)

        return ComplexityScore(
            total=total,
            level=level,
            dimensions=dimensions,
        )

    def _score_file_count(self, intent: Intent) -> float:
        """Score based on number of files."""
        file_count = len(intent.explicit_deps) + len(intent.implied_deps)
        if file_count == 0:
            return 20
        elif file_count == 1:
            return 25
        elif file_count == 2:
            return 40
        elif file_count <= 5:
            return 60
        else:
            return 80

    def _score_scope(self, intent: Intent) -> float:
        """Score based on scope."""
        scope_scores = {
            "single_file": 20,
            "multi_file": 50,
            "cross_module": 70,
            "cross_repo": 90,
            "unknown": 30,
        }
        return scope_scores.get(intent.scope, 30)

    def _score_dependency(self, intent: Intent) -> float:
        """Score based on dependency complexity."""
        implied = len(intent.implied_deps)
        if implied == 0:
            return 20
        elif implied <= 2:
            return 40
        elif implied <= 5:
            return 60
        else:
            return 80

    def _score_task_type(self, intent: Intent) -> float:
        """Score based on task type complexity."""
        type_scores = {
            "bug_fix": 30,
            "docs": 20,
            "test": 40,
            "feature_add": 55,
            "refactor": 65,
            "ci": 45,
            "general": 40,
        }
        return type_scores.get(intent.task_type, 40)

    def _determine_level(self, total: float) -> str:
        """Determine complexity level from total score."""
        if total <= self.simple_threshold:
            return "SIMPLE"
        elif total <= self.medium_threshold:
            return "MEDIUM"
        else:
            return "COMPLEX"
```

```python
"""Agent runtime with ReAct loop."""

from __future__ import annotations

import json
import logging
from typing import Any, Optional

from mozi.capabilities.tools.framework import ToolFramework
from mozi.core.error import ToolError
from mozi.infrastructure.model.adapter import ModelAdapter
from mozi.orchestrator.agent.base import Agent
from mozi.orchestrator.session.context import SessionContext

logger = logging.getLogger(__name__)


class AgentRuntime:
    """Runtime for executing agents with ReAct loop."""

    def __init__(
        self,
        model: ModelAdapter,
        tools: ToolFramework,
        max_steps: int = 50,
    ) -> None:
        self.model = model
        self.tools = tools
        self.max_steps = max_steps

    async def run_session(self, context: SessionContext) -> dict[str, Any]:
        """Run agent session until completion or max steps."""
        # Add initial user message
        context.add_message("user", context.user_input)

        while context.current_step < self.max_steps:
            result = await self.run_step(context)

            if result.get("is_final"):
                return result

            if not context.increment_step():
                return {"output": "Max steps reached", "is_final": True}

        return {"output": "Session ended", "is_final": True}

    async def run_step(self, context: SessionContext) -> dict[str, Any]:
        """Run a single ReAct step: Think -> Action -> Observe."""
        # Build prompt with tool schemas
        tool_schemas = self.tools.list_tool_schemas()

        # Convert messages for model
        messages = self._build_messages(context)

        try:
            response = await self.model.generate(
                prompt=self._build_prompt(context),
                system_prompt=self._get_system_prompt(),
                tools=tool_schemas if tool_schemas else None,
            )

            # Extract response text
            response_text = self._extract_response_text(response)
            context.add_message("assistant", response_text)

            # Check if model wants to use a tool
            tool_result = await self._execute_tools_if_any(response, context)

            if tool_result:
                return tool_result

            # Check if we're done
            if self._is_final_response(response):
                return {"output": response_text, "is_final": True}

            return {"output": response_text, "is_final": False}

        except Exception as e:
            logger.exception("Error in agent step")
            return {"error": str(e), "is_final": True}

    def _build_prompt(self, context: SessionContext) -> str:
        """Build the prompt for this step."""
        recent_messages = context.messages[-10:]  # Last 10 messages
        message_str = "\n".join(
            f"{msg['role']}: {msg['content']}" for msg in recent_messages
        )

        tool_results_str = ""
        if context.tool_results:
            recent_results = context.tool_results[-5:]
            tool_results_str = "\n".join(
                f"Tool: {r['tool']}\nResult: {r['result']}" for r in recent_results
            )
            tool_results_str = f"\n\nRecent tool results:\n{tool_results_str}"

        return f"""Current conversation:
{_message_str}
{tool_results_str}

Provide your next action or response."""

    def _build_messages(self, context: SessionContext) -> list[dict[str, str]]:
        """Build message list for the model."""
        return [
            {"role": msg["role"], "content": msg["content"]}
            for msg in context.messages[-10:]
        ]

    def _get_system_prompt(self) -> str:
        """Get system prompt for the agent."""
        return """You are Mozi, an AI coding assistant that helps users write and modify code.

You have access to tools: read, write, edit, bash, grep, glob.

When you need to use a tool, respond with a tool call in this format:
{
  "tool": "tool_name",
  "params": {"param1": "value1"}
}

Otherwise, respond directly with your analysis or result."""

    def _extract_response_text(self, response) -> str:
        """Extract text from model response."""
        if hasattr(response, "content"):
            for item in response.content:
                if hasattr(item, "type") and item.type == "text":
                    return item.text
        return str(response)

    def _is_final_response(self, response) -> bool:
        """Check if response is a final answer (no tool calls)."""
        if hasattr(response, "content"):
            for item in response.content:
                if hasattr(item, "type") and item.type == "tool_use":
                    return False
        return True

    async def _execute_tools_if_any(
        self, response, context: SessionContext
    ) -> Optional[dict[str, Any]]:
        """Execute any tool calls in the response."""
        if not hasattr(response, "content"):
            return None

        for item in response.content:
            if hasattr(item, "type") and item.type == "tool_use":
                tool_name = item.name
                tool_input = item.input

                # Execute tool
                result = await self.tools.execute(tool_name, tool_input)

                # Record result
                context.add_tool_result(
                    tool_name, tool_input, result.get("output", "")
                )

                # Add tool result to conversation
                context.add_message(
                    "user",
                    f"Tool result for {tool_name}: {result.get('output', result.get('error', 'No output'))}",
                )

                return {
                    "tool_executed": tool_name,
                    "result": result,
                    "is_final": False,
                }

        return None
```

```python
"""Main orchestrator coordinating all components."""

from __future__ import annotations

import logging
from pathlib import Path
from typing import Any, Optional

from mozi.capabilities.config.loader import ConfigLoader
from mozi.capabilities.tools.framework import ToolFramework
from mozi.infrastructure.db.sqlite import Database
from mozi.infrastructure.model.adapter import ModelAdapter
from mozi.orchestrator.agent.runtime import AgentRuntime
from mozi.orchestrator.core.intent import IntentClassifier
from mozi.orchestrator.core.complexity import ComplexityEngine
from mozi.orchestrator.core.router import TaskRouter
from mozi.orchestrator.session.manager import SessionManager

logger = logging.getLogger(__name__)


class Orchestrator:
    """Main orchestrator coordinating all layers."""

    def __init__(
        self,
        config_loader: ConfigLoader,
        model: Optional[ModelAdapter] = None,
    ) -> None:
        self.config_loader = config_loader
        self.config = config_loader.get_config()

        # Initialize components
        self.intent_classifier = IntentClassifier()
        self.complexity_engine = ComplexityEngine()
        self.router = TaskRouter()
        self.tool_framework = ToolFramework()

        # Initialize database
        self.db = Database(self.config.storage.sqlite_path.expanduser())

        # Initialize session manager
        self.session_manager = SessionManager(self.db)

        # Initialize agent runtime (will be set up after model is configured)
        self._model = model
        self._agent_runtime: Optional[AgentRuntime] = None

    async def initialize(self) -> None:
        """Initialize orchestrator components."""
        await self.db.initialize()
        await self.session_manager.initialize()

        if self._model is None:
            # TODO: Load from config
            raise ValueError("Model not configured")

        self._agent_runtime = AgentRuntime(
            model=self._model,
            tools=self.tool_framework,
        )

    async def close(self) -> None:
        """Clean up resources."""
        await self.session_manager.close()
        await self.db.close()

    async def process(
        self,
        user_input: str,
        session_id: Optional[str] = None,
    ) -> dict[str, Any]:
        """Process user input through the full pipeline."""
        # Create or resume session
        if session_id:
            context = await self.session_manager.get_context(session_id)
            if not context:
                raise ValueError(f"Session not found: {session_id}")
        else:
            session_id = await self.session_manager.create(user_input)
            context = await self.session_manager.get_context(session_id)

        # Step 1: Intent recognition
        intent = self.intent_classifier.classify(user_input)
        logger.info(f"Intent: {intent.task_type}, scope: {intent.scope}")

        # Step 2: Complexity assessment
        complexity = self.complexity_engine.assess(intent)
        logger.info(f"Complexity: {complexity.level} ({complexity.total})")

        # Update session with intent and complexity
        await self.session_manager.update_intent(
            session_id,
            {
                "task_type": intent.task_type,
                "scope": intent.scope,
                "target_files": intent.target_files,
            },
            complexity.total,
            complexity.level,
        )

        # Step 3: Route based on complexity
        route = self.router.route(complexity)
        logger.info(f"Routing to: {route}")

        # Step 4: Execute based on route
        if route == "fastpath":
            result = await self._execute_fastpath(context)
        elif route == "enhanced":
            result = await self._execute_enhanced(context)
        else:
            # DAG deferred to v1.1
            result = await self._execute_enhanced(context)

        # Update session status
        await self.session_manager.update_status(session_id, "completed")

        return {
            "session_id": session_id,
            "intent": intent,
            "complexity": complexity,
            "result": result,
        }

    async def _execute_fastpath(self, context) -> dict[str, Any]:
        """Execute simple task directly."""
        if not self._agent_runtime:
            raise ValueError("Agent runtime not initialized")

        result = await self._agent_runtime.run_session(context)
        return result

    async def _execute_enhanced(self, context) -> dict[str, Any]:
        """Execute with enhanced monitoring (MVP: same as fastpath)."""
        if not self._agent_runtime:
            raise ValueError("Agent runtime not initialized")

        result = await self._agent_runtime.run_session(context)
        return result
```

```python
"""CLI command implementations."""

from __future__ import annotations

import asyncio
import os
from pathlib import Path
from typing import Optional

import typer

from mozi import __version__
from mozi.cli.output import (
    console,
    print_welcome,
    print_result,
    print_error,
    print_success,
    print_info,
)
from mozi.capabilities.config.loader import ConfigLoader
from mozi.infrastructure.model.anthropic import AnthropicAdapter
from mozi.orchestrator.orchestrator import Orchestrator

cli = typer.Typer(help="Mozi AI Coding Agent")


@cli.command()
def version():
    """Show version information."""
    print_info(f"Mozi v{__version__}")


@cli.command()
def run(
    input: str = typer.Argument(..., help="User input/task description"),
    session_id: Optional[str] = typer.Option(None, help="Resume existing session"),
    config_dir: Optional[Path] = typer.Option(None, help="Config directory"),
    verbose: bool = typer.Option(False, "--verbose", "-v", help="Verbose output"),
):
    """Execute a task."""
    if verbose:
        import logging
        logging.basicConfig(level=logging.INFO)

    asyncio.run(_run_task(input, session_id, config_dir))


async def _run_task(
    user_input: str,
    session_id: Optional[str],
    config_dir: Optional[Path],
) -> None:
    """Internal async task runner."""
    try:
        # Determine config directory
        if config_dir is None:
            config_dir = Path.cwd() / ".mozi"

        # Load configuration
        config_loader = ConfigLoader(config_dir=config_dir)
        config_loader.load()

        # Get API key from environment
        api_key = os.getenv("ANTHROPIC_API_KEY")
        if not api_key:
            print_error("ANTHROPIC_API_KEY environment variable not set")
            raise typer.Exit(code=1)

        # Initialize model
        model = AnthropicAdapter(api_key=api_key)

        # Initialize orchestrator
        orchestrator = Orchestrator(config_loader=config_loader, model=model)
        await orchestrator.initialize()

        try:
            # Process input
            print_info("Processing...")
            result = await orchestrator.process(user_input, session_id)

            # Display result
            if "result" in result and "output" in result["result"]:
                print_result(result["result"]["output"])
            else:
                print_success("Task completed")

            # Show session ID for resumption
            print_info(f"Session ID: {result['session_id']}")

        finally:
            await orchestrator.close()

    except Exception as e:
        print_error(str(e))
        raise typer.Exit(code=1)


@cli.command()
def interactive():
    """Start interactive REPL mode."""
    from mozi.cli.output import print_welcome

    print_welcome()
    print_info("Interactive mode - type 'exit' to quit")

    session_id = None

    while True:
        user_input = console.input("\n[bold green]>[/bold green] ")

        if user_input.lower() in ["exit", "quit", "q"]:
            break

        if not user_input.strip():
            continue

        # Run task
        asyncio.run(_run_task(user_input, session_id, None))


if __name__ == "__main__":
    cli()
```
